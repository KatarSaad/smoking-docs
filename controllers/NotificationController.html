  @Get('mynotifications')
  @JwtAuth([RoleDecoratorEnum.USER, RoleDecoratorEnum.SYSADMIN, RoleDecoratorEnum.HEALTH_SPECIALIST])
  @ApiCreatedResponse({ type: PaginationCursorNotificationsResponse })
  @UseInterceptors(HttpTokenCacheInterceptor, BigIntTransformerInterceptor)
  async getMyNotifications(
    @HeaderLanguage() language: LanguageEnum,
    @User() user: UserPlayloadResponse,
    @Query() searchWithCursorPaginationDto: SearchWithCursorPaginationDto,
  ): Promise<PaginationCursorNotificationsResponse> {
    let where: Prisma.NotificationWhereInput = {
      recipientId: user.id,
    };
    if (user.isPatient) {
      const patient = await this.prismaService.user.findUnique({
        where: {
          id: user.id,
        },
        select: {
          patient: {
            select: {
              groups: { select: { id: true } },
            },
          },
        },
      });

      this.todayProgressesService.addCheckToTodaysProgress(user.id, DailyProgressTypeEnum.NOTIFICATION);

      if (patient.patient.groups.length === 0)
        return {
          data: [],
          endCursor: 0,
          hasNextPage: false,
          hasPreviousPage: false,
          startCursor: 0,
        };
      console.log(patient.patient.groups);

      where = {
        status: NotificationStatusEnum.SENT,
        targetGroupIds: {
          some: {
            id: {
              in: patient.patient.groups.map((group) => group.id),
            },
          },
        },
      };
    }

    return await this.notificationService.getListNotifications(language, user, where, searchWithCursorPaginationDto, user.isPatient);
  }

  @Get('mynotifications/offset-pagination')
  @JwtAuth([RoleDecoratorEnum.USER, RoleDecoratorEnum.SYSADMIN, RoleDecoratorEnum.HEALTH_SPECIALIST])
  @ApiCreatedResponse({ type: PaginationNotifications })
  @UseInterceptors(HttpTokenCacheInterceptor, BigIntTransformerInterceptor)
  async getMyNotificationsOffset(
    @HeaderLanguage() language: LanguageEnum,
    @User() user: UserPlayloadResponse,
    @Query() getNotificationsDto: GetNotificationsDto,
  ): Promise<PaginationNotifications> {
    let where: Prisma.NotificationWhereInput = {
      recipientId: user.id,
    };
    if (user.isPatient) {
      const patient = await this.prismaService.user.findUnique({
        where: {
          id: user.id,
        },
        select: {
          patient: {
            select: {
              groups: { select: { id: true } },
            },
          },
        },
      });

      //todayProgressesService to keep traking if user checked his notifs
      this.todayProgressesService.addCheckToTodaysProgress(user.id, DailyProgressTypeEnum.NOTIFICATION);

      if (patient.patient.groups.length === 0) {
        return {
          data: [],
          totalItems: 0,
        };
      }

      where = {
        status: NotificationStatusEnum.SENT,
        targetGroupIds: {
          some: {
            id: {
              in: patient.patient.groups.map((group) => group.id),
            },
          },
        },
      };
    }

    return await this.notificationService.getMyNotificationsOffset(language, user, where, getNotificationsDto, user.isPatient);
  }
