import { CacheTTL } from '@nestjs/cache-manager';
import { Controller, Get, Param, ParseIntPipe, Query, Post, Body, Patch, Delete, UseInterceptors, Put } from '@nestjs/common';
import { ApiCreatedResponse, ApiOkResponse, ApiTags } from '@nestjs/swagger';
import { LanguageEnum, PermissionEnum } from '@prisma/client';
import { User } from 'src/common/decorators/auth.decorator';
import { HeaderLanguage } from 'src/common/decorators/header-language';
import { JwtAuth } from 'src/common/decorators/jwt-auth.decorator';
import { Permissions } from 'src/common/decorators/permissions.decorator';
import { RoleDecoratorEnum } from 'src/common/enums/role-decorator.enum';
import { BigIntTransformerInterceptor } from 'src/common/interceptors/bigInt-transformer.interceptor';
import { HttpLanguageCacheInterceptor } from 'src/common/interceptors/http-language-cache.interceptor';
import { AddDraftFaqDto } from 'src/modules/faqs/dto/add-faq-draft.dto';
import { AddFaqDto } from 'src/modules/faqs/dto/add-faq.dto';
import { FetchFaqsWithCategoriesDto } from 'src/modules/faqs/dto/fetch-faqs-with-categories.dto copy';
import { FetchFaqsDto } from 'src/modules/faqs/dto/fetch-faqs.dto';
import { FaqsService } from 'src/modules/faqs/faqs.service';
import { FaqWithCategoryResponse } from 'src/modules/faqs/response/faq-with-category.response';
import { FaqWithMultiLanguage } from 'src/modules/faqs/response/faq-with-multi-language.response';
import { FaqResponse } from 'src/modules/faqs/response/faq.response';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';

@ApiTags('FAQ')
@Controller('faqs')
export class FaqsController {
  constructor(private faqsService: FaqsService) {}

  @Get('published')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_FAQ])
  @ApiOkResponse({ type: FaqResponse, isArray: true })
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  @CacheTTL(Number(process.env.REDIS_LONG_TTL))
  async fetchFaqs(@HeaderLanguage() language: LanguageEnum, @Query() fetchFaqsDto: FetchFaqsDto): Promise<FaqResponse[]> {
    return await this.faqsService.fetchFaqs(language, fetchFaqsDto);
  }

  @Post('published')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.ADD_FAQ])
  @ApiCreatedResponse({ type: FaqResponse })
  async createFaq(@HeaderLanguage() language: LanguageEnum, @User() user: UserPlayloadResponse, @Body() addFaqDto: AddFaqDto): Promise<FaqResponse> {
    return await this.faqsService.createFaq(language, user, addFaqDto);
  }

  @Get('faq-categories')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN, RoleDecoratorEnum.USER])
  @Permissions([PermissionEnum.DISPLAY_FAQ])
  @ApiOkResponse({ type: FaqWithCategoryResponse, isArray: true })
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  @CacheTTL(Number(process.env.REDIS_LONGER_TTL))
  async fetchfaqsByCategory(
    @HeaderLanguage() language: LanguageEnum,
    @Query() fetchFaqsWithCategoriesDto: FetchFaqsWithCategoriesDto,
  ): Promise<FaqWithCategoryResponse[]> {
    return await this.faqsService.fetchfaqsByCategory(language, fetchFaqsWithCategoriesDto);
  }

  @Get('drafts')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_FAQ])
  @ApiOkResponse({ type: FaqResponse, isArray: true })
  async fetchDraftFaqs(@HeaderLanguage() language: LanguageEnum, @User() user: UserPlayloadResponse, @Query() fetchFaqsDto: FetchFaqsDto): Promise<FaqResponse[]> {
    return await this.faqsService.fetchDraftFaqs(language, user, fetchFaqsDto);
  }

  @Post('drafts')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.ADD_FAQ])
  @ApiCreatedResponse({ type: FaqResponse })
  async createDraftFaq(@HeaderLanguage() language: LanguageEnum, @User() user: UserPlayloadResponse, @Body() addDraftFaqDto: AddDraftFaqDto): Promise<FaqResponse> {
    return await this.faqsService.createDraftFaq(language, user, addDraftFaqDto);
  }

  @Patch(':id')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.UPDATE_FAQ])
  @ApiOkResponse({ type: FaqResponse })
  async updateFaq(
    @HeaderLanguage() language: LanguageEnum,
    @User() user: UserPlayloadResponse,
    @Body() addFaqDto: AddFaqDto,
    @Param('id', ParseIntPipe) id: number,
  ): Promise<FaqResponse> {
    return await this.faqsService.updateFaq(language, user, addFaqDto, id);
  }

  @Delete(':id')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DELETE_FAQ])
  @ApiOkResponse({ type: FaqResponse })
  async deleteFaq(@HeaderLanguage() language: LanguageEnum, @User() user: UserPlayloadResponse, @Param('id', ParseIntPipe) id: number): Promise<FaqResponse> {
    return await this.faqsService.deleteFaq(language, user, id);
  }

  @Get('published/multi-language')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_FAQ])
  @ApiOkResponse({ type: FaqWithMultiLanguage, isArray: true })
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  async fetchFaqsWithMultiLanguage(@Query() fetchFaqsDto: FetchFaqsDto): Promise<FaqWithMultiLanguage[]> {
    return await this.faqsService.fetchFaqsWithMultiLanguage(fetchFaqsDto);
  }

  @Get('drafts/multi-language')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_FAQ])
  @ApiOkResponse({ type: FaqWithMultiLanguage, isArray: true })
  async fetchDraftFaqsWithMultiLanguage(@User() user: UserPlayloadResponse, @Query() fetchFaqsDto: FetchFaqsDto): Promise<FaqWithMultiLanguage[]> {
    return await this.faqsService.fetchDraftFaqsWithMultiLanguage(user, fetchFaqsDto);
  }

  @Put('published/:id/published')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.UPDATE_FAQ])
  @ApiOkResponse({ type: FaqResponse })
  async publishFaq(@HeaderLanguage() language: LanguageEnum, @Param('id', ParseIntPipe) id: number): Promise<FaqResponse> {
    return await this.faqsService.publishFaq(language, id);
  }

  @Put('published/:id/unpublished')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.UPDATE_FAQ])
  @ApiOkResponse({ type: FaqResponse })
  async unpublishFaq(@HeaderLanguage() language: LanguageEnum, @Param('id', ParseIntPipe) id: number): Promise<FaqResponse> {
    return await this.faqsService.unpublishFaq(language, id);
  }

  @Patch('drafts/:id')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.UPDATE_FAQ])
  @ApiOkResponse({ type: FaqResponse })
  async updateDraftFaq(
    @HeaderLanguage() language: LanguageEnum,
    @User() user: UserPlayloadResponse,
    @Body() addDraftFaqDto: AddDraftFaqDto,
    @Param('id', ParseIntPipe) id: number,
  ): Promise<FaqResponse> {
    return await this.faqsService.updateDraftFaq(language, user, addDraftFaqDto, id);
  }
}
