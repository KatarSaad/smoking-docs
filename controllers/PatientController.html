import { Controller, Delete, Get, Patch, Query, Body, Put, Param, ParseIntPipe, UseInterceptors } from '@nestjs/common';
import { PatientService } from './patient.service';
import { ApiOkResponse, ApiResponse, ApiTags } from '@nestjs/swagger';
import { JwtAuth } from 'src/common/decorators/jwt-auth.decorator';
import { User } from 'src/common/decorators/auth.decorator';
import { PatientProfileResponse } from './response/patient-pofile.response';
import { Serialize } from 'src/common/interceptors/response-serialization.interceptor';
import { Permissions } from 'src/common/decorators/permissions.decorator';
import { ShowPatientFilter } from './dto/show-patient-filter.dto';
import { ShowPatientProfileResponse } from './response/show-patient-profile.response';
import { UpdatePatientDto } from './dto/update-patient.dto';
import { UpdatePatientResponse } from './response/update-patient.response';
import { PaginationPatients } from 'src/common/responses/Paggination-data.response';
import { UserPlayloadResponse } from './response/user-playload.response';
import { UpdateUsernameDto } from './dto/update-username.dto';
import { UpdateProfileDto } from './dto/update-profile.dto';
import { FetchPatientDetailsResponse } from 'src/modules/patients/response/fetch-patient-details.response';
import { RoleDecoratorEnum } from 'src/common/enums/role-decorator.enum';
import { LanguageEnum, PermissionEnum } from '@prisma/client';
import { HeaderLanguage } from 'src/common/decorators/header-language';
import { Facilities } from 'src/common/decorators/facilities.decorator';
import { CacheTTL } from '@nestjs/cache-manager';
import { HttpTokenCacheInterceptor } from 'src/common/interceptors/http-token-cache.interceptor';
import { HttpLanguageCacheInterceptor } from 'src/common/interceptors/http-language-cache.interceptor';
import { BigIntTransformerInterceptor } from 'src/common/interceptors/bigInt-transformer.interceptor';
import { PatientNameOnlyResponse } from 'src/modules/patients/response/patient-name-only.response';
import { PatientOnlyNamesSearchDto } from 'src/modules/patients/dto/patient-only-names-search.dto';
import { UpdatePatientUsername } from 'src/modules/patients/response/update-patient-username.response';
import { SearchWithCursorPaginationDto } from 'src/common/dto/search-with-cursor-pagination.dto';
import { PaginationCursorNotBlockedPatientsResponse } from 'src/common/responses/paggination-cursor.response';
import { PatientNameQidResponse } from 'src/modules/patients/response/patient-name-qid.response';

@ApiTags('patients')
@Controller('patients')
export class PatientController {
  constructor(private patientService: PatientService) {}

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_PATIENTS])
  @ApiOkResponse({ type: PaginationPatients })
  @Get('')
  @UseInterceptors(HttpTokenCacheInterceptor, BigIntTransformerInterceptor)
  @CacheTTL(Number(process.env.REDIS_LONG_TTL))
  async fetchPatients(@Facilities() facilities: number[], @Query() showPatientFilter: ShowPatientFilter): Promise<PaginationPatients> {
    return await this.patientService.fetchPatients(facilities, showPatientFilter);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_PATIENTS])
  @ApiOkResponse({ type: [PatientNameOnlyResponse] })
  @Get('names-only')
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  @CacheTTL(Number(process.env.REDIS_LONG_TTL))
  async fetchPatientsNamesOnly(@Facilities() facilities: number[]): Promise<PatientNameOnlyResponse[]> {
    return await this.patientService.fetchPatientsNamesOnly(facilities);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: PatientNameQidResponse })
  @Delete('/my-account')
  async deleteAccount(@User() user: UserPlayloadResponse): Promise<PatientNameQidResponse> {
    return await this.patientService.deleteAccount(user);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: PatientProfileResponse })
  @Get('/profile')
  async patientProfile(@User() user: UserPlayloadResponse, @HeaderLanguage() language: LanguageEnum): Promise<PatientProfileResponse> {
    console.log(language);
    return await this.patientService.patientProfile(user, language);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: PatientProfileResponse })
  @Patch('/profile')
  async updatePatientProfile(@User() user: UserPlayloadResponse, @Body() updateProfileDto: UpdateProfileDto): Promise<boolean> {
    return await this.patientService.updatePatientProfile(user, updateProfileDto);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: UpdatePatientUsername })
  @Put('/username')
  async updatePatientUsername(@Body() updateUsernameDto: UpdateUsernameDto, @User() userPlayloadResponse: UserPlayloadResponse): Promise<UpdatePatientUsername> {
    return await this.patientService.updatePatientUsername(updateUsernameDto, userPlayloadResponse);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_PATIENTS])
  @ApiOkResponse({ type: ShowPatientProfileResponse })
  @Get('/:id')
  @UseInterceptors(HttpTokenCacheInterceptor, BigIntTransformerInterceptor)
  async fetchPatient(@Param('id', ParseIntPipe) id: number): Promise<ShowPatientProfileResponse> {
    return await this.patientService.fetchPatient(id);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.UPDATE_PATIENTS])
  @ApiOkResponse({ type: UpdatePatientResponse })
  @Patch('/:id')
  @ApiResponse({ status: 404, description: 'User not found!|Facility does not exist!|Support doctor does not exist!' })
  @ApiResponse({ status: 409, description: 'This support doctor does not work in this facility!' })
  async updatePatient(@Param('id', ParseIntPipe) id: number, @Body() updatePatientDto: UpdatePatientDto): Promise<UpdatePatientResponse> {
    return await this.patientService.updatePatient(id, updatePatientDto);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_PATIENTS])
  @ApiOkResponse({ type: [PatientNameQidResponse] })
  @Get('names-only/no-group-chat')
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  async fetchPatientsNamesOnlyNotInGroup(@Facilities() facilities: number[], @Query() searchName: PatientOnlyNamesSearchDto): Promise<PatientNameQidResponse[]> {
    return await this.patientService.fetchPatientsNamesOnlyNotInGroup(facilities, searchName);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_PATIENTS])
  @ApiOkResponse({ type: [PatientNameQidResponse] })
  @Get('names-only/no-private-chat')
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  async fetchPatientsNamesOnlyWithoutPrivateChat(@Facilities() facilities: number[], @Query() searchName: PatientOnlyNamesSearchDto): Promise<PatientNameQidResponse[]> {
    return await this.patientService.fetchPatientsNamesOnlyWithoutPrivateChat(facilities, searchName);
  }

  @JwtAuth()
  @ApiOkResponse({ type: PaginationCursorNotBlockedPatientsResponse })
  @Get('names-only/not-blocked')
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  @CacheTTL(Number(process.env.REDIS_TTL))
  async fetchNotBlockedPatientsNames(
    @Facilities() facilities: number[],
    @Query() searchWithCursorPaginationDto: SearchWithCursorPaginationDto,
  ): Promise<PaginationCursorNotBlockedPatientsResponse> {
    return await this.patientService.fetchNotBlockedPatientsNames(facilities, searchWithCursorPaginationDto);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_PATIENTS])
  @ApiOkResponse({ type: FetchPatientDetailsResponse })
  @Get('/details/:id')
  @Serialize(FetchPatientDetailsResponse)
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  async fetchPatientDetails(@Param('id', ParseIntPipe) id: number, @HeaderLanguage() language: LanguageEnum): Promise<FetchPatientDetailsResponse> {
    return await this.patientService.fetchPatientDetails(id, language);
  }
}
