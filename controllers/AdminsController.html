import { Body, Controller, Get, Param, ParseIntPipe, Post, Put, UseInterceptors } from '@nestjs/common';
import { ApiOkResponse, ApiTags } from '@nestjs/swagger';
import { LanguageEnum, PermissionEnum } from '@prisma/client';
import { User } from 'src/common/decorators/auth.decorator';
import { HeaderLanguage } from 'src/common/decorators/header-language';
import { JwtAuth } from 'src/common/decorators/jwt-auth.decorator';
import { RoleDecoratorEnum } from 'src/common/enums/role-decorator.enum';
import { Serialize } from 'src/common/interceptors/response-serialization.interceptor';
import { AdminsService } from 'src/modules/admins/admins.service';
import { UserAdminResponse } from 'src/modules/admins/response/fetch-admin.response';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';
import { CreateFacilityAdminDto } from './dtos/create-facility-admin.dto';
import { HttpTokenCacheInterceptor } from 'src/common/interceptors/http-token-cache.interceptor';
import { CacheTTL } from '@nestjs/cache-manager';
import { BigIntTransformerInterceptor } from 'src/common/interceptors/bigInt-transformer.interceptor';
import { UpdateFacilityAdminDto } from 'src/modules/admins/dtos/update-facility-admin.dto';
import { Permissions } from 'src/common/decorators/permissions.decorator';
import { CreateSystemAdminDto } from 'src/modules/admins/dtos/create-system-admin.dto';
import { UpdateSystemAdminDto } from 'src/modules/admins/dtos/update-system-admin.dto';
import { AdminSimpleResponse } from './response/fetch-admins-simple.response';

@ApiTags('Admins')
@Controller('admins')
export class AdminsController {
  constructor(private adminsService: AdminsService) {}

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Get('')
  @ApiOkResponse({ type: UserAdminResponse })
  @Serialize(UserAdminResponse)
  @UseInterceptors(HttpTokenCacheInterceptor, BigIntTransformerInterceptor)
  @CacheTTL(Number(process.env.REDIS_LONGER_TTL))
  async fetchAdminProfile(@HeaderLanguage() language: LanguageEnum, @User() user: UserPlayloadResponse): Promise<UserAdminResponse> {
    return await this.adminsService.fetchAdminProfile(language, user);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.ADD_ADMINS])
  @Post('/facility-admins')
  @ApiOkResponse({ type: UserAdminResponse })
  @Serialize(UserAdminResponse)
  async createFacilityAdmin(@HeaderLanguage() language: LanguageEnum, @Body() createFacilityAdminDto: CreateFacilityAdminDto): Promise<UserAdminResponse> {
    return await this.adminsService.createFacilityAdmin(language, createFacilityAdminDto);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.ADD_ADMINS])
  @Post('/system-admins')
  @ApiOkResponse({ type: UserAdminResponse })
  @Serialize(UserAdminResponse)
  async createSystemAdmin(@HeaderLanguage() language: LanguageEnum, @Body() createSystemAdminDto: CreateSystemAdminDto): Promise<UserAdminResponse> {
    return await this.adminsService.createSystemAdmin(language, createSystemAdminDto);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.UPDATE_ADMINS])
  @Put('/facility-admins/:id')
  @ApiOkResponse({ type: UserAdminResponse })
  @Serialize(UserAdminResponse)
  async updateFacilityAdmin(
    @HeaderLanguage() language: LanguageEnum,
    @Body() updateFacilityAdminDto: UpdateFacilityAdminDto,
    @Param('id', ParseIntPipe) id: number,
  ): Promise<UserAdminResponse> {
    return await this.adminsService.updateFacilityAdmin(language, updateFacilityAdminDto, id);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.UPDATE_ADMINS])
  @Put('/system-admins/:id')
  @ApiOkResponse({ type: UserAdminResponse })
  @Serialize(UserAdminResponse)
  async updateSystemAdmin(
    @HeaderLanguage() language: LanguageEnum,
    @Body() updateSystemAdminDto: UpdateSystemAdminDto,
    @Param('id', ParseIntPipe) id: number,
  ): Promise<UserAdminResponse> {
    return await this.adminsService.updateSystemAdmin(language, updateSystemAdminDto, id);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Get('simple-admins-list')
  @ApiOkResponse({ type: [AdminSimpleResponse] })
  @UseInterceptors(HttpTokenCacheInterceptor, BigIntTransformerInterceptor)
  @CacheTTL(Number(process.env.REDIS_LONG_TTL))
  async fetchAdminsSimple(@User() user: UserPlayloadResponse): Promise<AdminSimpleResponse[]> {
    return await this.adminsService.fetchAdminsSimple(user);
  }
}
