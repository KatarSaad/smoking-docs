import { CacheTTL } from '@nestjs/cache-manager';
import { Body, Controller, Get, Param, ParseIntPipe, Patch, Post, Query, UseInterceptors } from '@nestjs/common';
import { ApiCreatedResponse, ApiOkResponse, ApiTags } from '@nestjs/swagger';
import { LanguageEnum, PermissionEnum } from '@prisma/client';
import { HeaderLanguage } from 'src/common/decorators/header-language';
import { JwtAuth } from 'src/common/decorators/jwt-auth.decorator';
import { Permissions } from 'src/common/decorators/permissions.decorator';
import { PaginationOptionalDto } from 'src/common/dto/pagination-optional.dto';
import { RoleDecoratorEnum } from 'src/common/enums/role-decorator.enum';
import { BigIntTransformerInterceptor } from 'src/common/interceptors/bigInt-transformer.interceptor';
import { HttpLanguageCacheInterceptor } from 'src/common/interceptors/http-language-cache.interceptor';
import { PaginationArticleLabels, PaginationMultiLanguageArticlelabels } from 'src/common/responses/Paggination-data.response';
import { ArticlesLabelsService } from 'src/modules/articles-labels/articles-labels.service';
import { AddArticlesLabelDto } from 'src/modules/articles-labels/dto/add-articles-label.dto';
import { ArticleLabelResponse } from 'src/modules/articles-labels/response/article-label.response';
import { FetchMultiLanguageArticleLabelResponse } from 'src/modules/articles-labels/response/fetch-multi-language-article-label.response';

@Controller('articles-labels')
@ApiTags('Articles labels')
export class ArticlesLabelsController {
  constructor(private articlesLabelsService: ArticlesLabelsService) {}

  @Get('')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN, RoleDecoratorEnum.USER])
  @Permissions([PermissionEnum.DISPLAY_CONTENTS, PermissionEnum.GENERATE_CONTENT])
  @ApiOkResponse({ type: PaginationArticleLabels })
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  @CacheTTL(Number(process.env.REDIS_LONG_TTL))
  async fetchArticleLabels(@HeaderLanguage() language: LanguageEnum, @Query() paginationOptionalDto: PaginationOptionalDto): Promise<PaginationArticleLabels> {
    return await this.articlesLabelsService.fetchArticleLabels(language, paginationOptionalDto);
  }

  @Post()
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.GENERATE_CONTENT])
  @ApiCreatedResponse({ type: ArticleLabelResponse })
  async createArticleLabel(@Body() addArticlesLabelDto: AddArticlesLabelDto): Promise<ArticleLabelResponse> {
    return await this.articlesLabelsService.createArticleLabel(addArticlesLabelDto);
  }

  @Get('all-languages')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_CONTENTS, PermissionEnum.GENERATE_CONTENT])
  @ApiOkResponse({ type: PaginationMultiLanguageArticlelabels })
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  async fetchMultiLanguageArticleLabels(@Query() paginationOptionalDto: PaginationOptionalDto): Promise<PaginationMultiLanguageArticlelabels> {
    return await this.articlesLabelsService.fetchMultiLanguageArticleLabels(paginationOptionalDto);
  }

  @Patch(':id')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.GENERATE_CONTENT])
  @ApiOkResponse({ type: ArticleLabelResponse })
  async updateArticleLabel(@Param('id', ParseIntPipe) id: number, @Body() addArticlesLabelDto: AddArticlesLabelDto): Promise<ArticleLabelResponse> {
    return await this.articlesLabelsService.updateArticleLabel(id, addArticlesLabelDto);
  }

  @Get('all-languages/without-pagination')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_CONTENTS, PermissionEnum.GENERATE_CONTENT])
  @ApiOkResponse({ type: PaginationMultiLanguageArticlelabels })
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  async fetchMultiLanguageArticleLabelsWithoutPagination(): Promise<FetchMultiLanguageArticleLabelResponse[]> {
    return await this.articlesLabelsService.fetchMultiLanguageArticleLabelsWithoutPagination();
  }
}
