import { Body, Controller, Delete, Get, Param, ParseIntPipe, Post, Put, Query } from '@nestjs/common';
import { ApiCreatedResponse, ApiOkResponse, ApiTags } from '@nestjs/swagger';
import { PermissionEnum } from '@prisma/client';
import { JwtAuth } from 'src/common/decorators/jwt-auth.decorator';
import { Permissions } from 'src/common/decorators/permissions.decorator';
import { RoleDecoratorEnum } from 'src/common/enums/role-decorator.enum';
import { NameAndIdResponse } from 'src/common/responses/id-and-name.response';
import { GroupChatsService } from 'src/modules/group-chats/group-chats.service';
import { CreateGroupChatDto } from 'src/modules/group-chats/dto/create-group-chat.dto';
import { UpdateGroupChatDto } from 'src/modules/group-chats/dto/update-group-chat.dto';
import { GroupChatByIdResponse } from 'src/modules/group-chats/response/group-chat-by-id.response';
import { GroupChatResponse } from 'src/modules/group-chats/response/group-chat.response';
import { User } from 'src/common/decorators/auth.decorator';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';
import { PaginationPatientsMessages, PaginationShowFlatGroupChatResponse } from 'src/common/responses/Paggination-data.response';
import { UpdateGroupChatParticipantsDto } from 'src/modules/group-chats/dto/update-group-chat-participants.dto';
import {
  PaginationCursorMessages,
  PaginationCursorPatientsMessages,
  PaginationCursorPatientNameOnlyResponse,
  PaginationCursorMessagesSearchResponse,
} from 'src/common/responses/paggination-cursor.response';
import { FetchAuthedPatientChatDataResponse } from 'src/modules/group-chats/response/fetch-authed-patient-chat-data.response';
import { SearchWithCursorPaginationDto } from 'src/common/dto/search-with-cursor-pagination.dto';
import { SearchWithPaginationDto } from 'src/common/dto/search-with-pagination.dto';
import { SearchRequiredWithCursorPaginationDto } from 'src/common/dto/search-pagination-required.dto';

@Controller('group-chats')
@ApiTags('Group Chats')
export class GroupChatsController {
  constructor(private readonly groupChatsService: GroupChatsService) {}

  @Get('')
  @JwtAuth()
  @Permissions([PermissionEnum.DISPLAY_GROUP_CHAT, PermissionEnum.DISPLAY_GROUP_CHAT_ONLY])
  @ApiOkResponse({ type: PaginationShowFlatGroupChatResponse })
  async fetchGroupChats(@User() user: UserPlayloadResponse, @Query() searchWithPaginationDto: SearchWithPaginationDto): Promise<PaginationShowFlatGroupChatResponse> {
    return await this.groupChatsService.fetchGroupChats(user, searchWithPaginationDto);
  }

  @Post('')
  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST])
  @Permissions([PermissionEnum.ADD_GROUP_CHAT])
  @ApiCreatedResponse({ type: NameAndIdResponse })
  async createGroupChat(@Body() createGroupChatDto: CreateGroupChatDto): Promise<NameAndIdResponse> {
    return await this.groupChatsService.createGroupChat(createGroupChatDto);
  }

  @Put('quit')
  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: Boolean })
  async quitGroupChat(@User() user: UserPlayloadResponse): Promise<boolean> {
    return await this.groupChatsService.quitGroupChat(user);
  }

  @Get(':id')
  @JwtAuth()
  @Permissions([PermissionEnum.DISPLAY_GROUP_CHAT, PermissionEnum.DISPLAY_GROUP_CHAT_ONLY])
  @ApiOkResponse({ type: GroupChatByIdResponse })
  async fetchGroupChat(@Param('id', ParseIntPipe) id: number): Promise<GroupChatByIdResponse> {
    return await this.groupChatsService.fetchGroupChat(id);
  }

  @Put(':id')
  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST])
  @Permissions([PermissionEnum.UPDATE_GROUP_CHAT])
  @ApiOkResponse({ type: GroupChatResponse })
  async updateGroupChat(@Body() updateGroupChatDto: UpdateGroupChatDto, @Param('id', ParseIntPipe) id: number): Promise<GroupChatResponse> {
    return await this.groupChatsService.updateGroupChat(updateGroupChatDto, id);
  }

  @Delete(':id')
  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST])
  @Permissions([PermissionEnum.DELETE_GROUP_CHAT])
  @ApiOkResponse({ type: GroupChatResponse })
  async deleteGroupChat(@Param('id', ParseIntPipe) id: number): Promise<GroupChatResponse> {
    return await this.groupChatsService.deleteGroupChat(id);
  }

  @Get('my-chat/data')
  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: FetchAuthedPatientChatDataResponse })
  async fetchAuthedChatData(@User() user: UserPlayloadResponse): Promise<FetchAuthedPatientChatDataResponse> {
    return await this.groupChatsService.fetchAuthedChatData(user);
  }

  @Get('my-group-chat/messages')
  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: PaginationCursorPatientsMessages })
  async fetchPatientGroupChat(
    @User() user: UserPlayloadResponse,
    @Query() searchWithCursorPaginationDto: SearchWithCursorPaginationDto,
  ): Promise<PaginationCursorPatientsMessages> {
    return await this.groupChatsService.fetchPatientGroupChat(user, searchWithCursorPaginationDto);
  }

  @Get(':id/participants')
  @JwtAuth()
  @ApiOkResponse({ type: PaginationCursorPatientNameOnlyResponse })
  async fetchGroupChatParticipants(
    @Query() searchWithCursorPaginationDto: SearchWithCursorPaginationDto,
    @Param('id', ParseIntPipe) id: number,
  ): Promise<PaginationCursorPatientNameOnlyResponse> {
    return await this.groupChatsService.fetchGroupChatParticipants(searchWithCursorPaginationDto, id);
  }

  @Get(':id/messages')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_GROUP_CHAT, PermissionEnum.DISPLAY_GROUP_CHAT_ONLY])
  @ApiOkResponse({ type: PaginationCursorMessages })
  async fetchGroupChatMessages(
    @User() user: UserPlayloadResponse,
    @Query() searchWithCursorPaginationDto: SearchWithCursorPaginationDto,
    @Param('id', ParseIntPipe) id: number,
  ): Promise<PaginationCursorMessages> {
    return await this.groupChatsService.fetchGroupChatMessages(user, searchWithCursorPaginationDto, id);
  }

  @Post(':id/participants')
  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST])
  @Permissions([PermissionEnum.UPDATE_GROUP_CHAT])
  @ApiCreatedResponse({ type: GroupChatResponse })
  async addParticipantsToGroupChat(
    @Body() updateGroupChatParticipantsDto: UpdateGroupChatParticipantsDto,
    @Param('id', ParseIntPipe) id: number,
  ): Promise<GroupChatResponse> {
    return await this.groupChatsService.addParticipantsToGroupChat(updateGroupChatParticipantsDto, id);
  }

  @Get('my-group-chat/messages/offset-pagination')
  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: PaginationPatientsMessages })
  async fetchMyGroupChatMessagesOffsetPagination(
    @User() user: UserPlayloadResponse,
    @Query() searchWithPaginationDto: SearchWithPaginationDto,
  ): Promise<PaginationPatientsMessages> {
    return await this.groupChatsService.fetchMyGroupChatMessagesOffsetPagination(user, searchWithPaginationDto);
  }

  @Get(':id/messages/offset-pagination')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_GROUP_CHAT, PermissionEnum.DISPLAY_GROUP_CHAT_ONLY])
  @ApiOkResponse({ type: PaginationPatientsMessages })
  async fetchGroupChatMessagesOffsetPagination(
    @Query() searchWithPaginationDto: SearchWithPaginationDto,
    @Param('id', ParseIntPipe) id: number,
  ): Promise<PaginationPatientsMessages> {
    return await this.groupChatsService.fetchGroupChatMessagesOffsetPagination(searchWithPaginationDto, id);
  }

  @Post(':id/participants/:participantId')
  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST])
  @Permissions([PermissionEnum.UPDATE_GROUP_CHAT])
  @ApiOkResponse({ type: GroupChatResponse })
  async addParticipant(@Param('id', ParseIntPipe) id: number, @Param('participantId', ParseIntPipe) participantId: number): Promise<GroupChatResponse> {
    return await this.groupChatsService.addParticipant(id, participantId);
  }

  @Delete(':id/participants/:participantId')
  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST])
  @Permissions([PermissionEnum.UPDATE_GROUP_CHAT])
  @ApiOkResponse({ type: GroupChatResponse })
  async deleteParticipant(@Param('id', ParseIntPipe) id: number, @Param('participantId', ParseIntPipe) participantId: number): Promise<GroupChatResponse> {
    return await this.groupChatsService.deleteParticipant(id, participantId);
  }

  @Put(':id/health-specialists/:healthSpecialistId/connect')
  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST])
  @Permissions([PermissionEnum.UPDATE_GROUP_CHAT])
  @ApiOkResponse({ type: GroupChatResponse })
  async connectHealthSpecialistToGroupChat(
    @Param('id', ParseIntPipe) id: number,
    @Param('healthSpecialistId', ParseIntPipe) healthSpecialistId: number,
  ): Promise<GroupChatResponse> {
    return await this.groupChatsService.connectHealthSpecialistToGroupChat(id, healthSpecialistId);
  }

  @Put(':id/health-specialists/:healthSpecialistId/disconnect')
  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST])
  @Permissions([PermissionEnum.UPDATE_GROUP_CHAT])
  @ApiOkResponse({ type: GroupChatResponse })
  async diconnectHealthSpecialistFromGroupChat(
    @Param('id', ParseIntPipe) id: number,
    @Param('healthSpecialistId', ParseIntPipe) healthSpecialistId: number,
  ): Promise<GroupChatResponse> {
    return await this.groupChatsService.diconnectHealthSpecialistFromGroupChat(id, healthSpecialistId);
  }

  @Put(':oldChatGroupId/participants/:participantId/new-chat-group/:newChatGroupId')
  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST])
  @Permissions([PermissionEnum.UPDATE_GROUP_CHAT])
  @ApiOkResponse({ type: GroupChatResponse })
  async switchUserGroup(
    @Param('oldChatGroupId', ParseIntPipe) oldChatGroupId: number,
    @Param('participantId', ParseIntPipe) participantId: number,
    @Param('newChatGroupId', ParseIntPipe) newChatGroupId: number,
  ): Promise<GroupChatResponse> {
    return await this.groupChatsService.switchUserGroup(oldChatGroupId, participantId, newChatGroupId);
  }

  @Get(':id/search')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_GROUP_CHAT, PermissionEnum.DISPLAY_GROUP_CHAT_ONLY])
  @ApiOkResponse({ type: PaginationCursorMessagesSearchResponse })
  async searchMessageInGroupChat(
    @User() user: UserPlayloadResponse,
    @Query() searchWithCursorPaginationDto: SearchWithCursorPaginationDto,
    @Param('id', ParseIntPipe) id: number,
  ): Promise<PaginationCursorMessagesSearchResponse> {
    console.log(searchWithCursorPaginationDto);
    return await this.groupChatsService.searchMessage(user, searchWithCursorPaginationDto, id);
  }

  @Get('my-group-chat/messages/search')
  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: PaginationCursorMessagesSearchResponse })
  async searchMessageInMyGroupChat(
    @User() user: UserPlayloadResponse,
    @Query() searchWithCursorPaginationDto: SearchRequiredWithCursorPaginationDto,
  ): Promise<PaginationCursorMessagesSearchResponse> {
    console.log(user);
    return await this.groupChatsService.searchMessageMyGroupChat(user, searchWithCursorPaginationDto);
  }
}
