import { Body, Controller, Delete, Get, Param, Patch, Post, Query, UploadedFiles, UseInterceptors } from '@nestjs/common';
import { TicketsService } from './tickets.service';
import { FilesInterceptor } from '@nestjs/platform-express/multer';
import { PatientAddTicketDto } from './dto/patient-add-ticket.dto';
import { ImagesUploadPipe } from 'src/common/pipes/image-list-validation.pipe';
import { JwtAuth } from 'src/common/decorators/jwt-auth.decorator';
import { RoleDecoratorEnum } from 'src/common/enums/role-decorator.enum';
import { ApiTags, ApiOperation, ApiConsumes, ApiResponse, ApiParam } from '@nestjs/swagger';
import { User } from 'src/common/decorators/auth.decorator';
import { UserPlayloadResponse } from '../patients/response/user-playload.response';
import { UpdateTicketByAdminDto } from './dto/update-ticket-by-admin.dto';
import { UpdateTicketByPatientDto } from './dto/update-ticket-by-patient.dto';
import { TicketResponse } from './response/ticket.response';
import { PatientTicketResponse } from './response/patient-ticket.response';
import { TicketFilterDTO } from './dto/filter-ticket.dto';
import { PaginationTicketResponse } from 'src/common/responses/Paggination-data.response';
import { CountTicketByStatusAndAssignedToAdminResponse } from './response/count-tickets-stats.response';
import { PaginationDto } from 'src/common/dto/pagination.dto';

@ApiTags('tickets')
@Controller('tickets')
export class TicketsController {
  constructor(private readonly ticketService: TicketsService) {}

  @Post('')
  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOperation({ summary: 'Add a new ticket by a patient' })
  @ApiConsumes('multipart/form-data')
  @ApiResponse({
    status: 201,
    description: 'The ticket has been successfully created.',
    type: PatientTicketResponse,
  })
  @UseInterceptors(FilesInterceptor('attachments', 5))
  async addTicketByPatient(
    @User() user: UserPlayloadResponse,
    @Body() ticketDto: PatientAddTicketDto,
    @UploadedFiles(ImagesUploadPipe) attachments?: Express.Multer.File[],
  ): Promise<PatientTicketResponse> {
    return await this.ticketService.createTicket(user, ticketDto, attachments);
  }

  @Get()
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @ApiOperation({ summary: 'Get all tickets' })
  @ApiResponse({
    status: 200,
    description: 'The list of tickets has been successfully retrieved.',
    type: PaginationTicketResponse,
  })
  // @ApiQuery(TicketFilterDTO)
  async getTickets(@Query() ticketFilterDTO: TicketFilterDTO, @User() user: UserPlayloadResponse): Promise<PaginationTicketResponse> {
    return this.ticketService.getTickets(user, ticketFilterDTO);
  }
  //xx
  @Get('counts')
  @ApiOperation({ summary: 'get count of the status and assingned to the current admin tickets ' })
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @ApiResponse({ status: 200, description: 'Counts of tickets by status and assigned to admin', type: CountTicketByStatusAndAssignedToAdminResponse })
  async n(@User() user: UserPlayloadResponse): Promise<CountTicketByStatusAndAssignedToAdminResponse> {
    return this.ticketService.countTicketsByStatusAndAssignedToAdmin(user.adminId);
  }

  @Get(':id')
  @JwtAuth([RoleDecoratorEnum.USER, RoleDecoratorEnum.SYSADMIN])
  @ApiOperation({ summary: 'Get a ticket by ID' })
  @ApiResponse({
    status: 200,
    description: 'The ticket has been successfully retrieved.',
    type: TicketResponse,
  })
  @ApiResponse({ status: 404, description: 'Ticket not found.' })
  async getTicketById(@Param('id') id: bigint): Promise<TicketResponse> {
    return this.ticketService.getTicketById(id);
  }

  @Delete(':id')
  @JwtAuth([RoleDecoratorEnum.USER, RoleDecoratorEnum.SYSADMIN])
  @ApiResponse({ status: 204, description: 'Ticket and associated comments and attachments have been deleted.', type: Number })
  async deleteTicket(@Param('id') ticketId: bigint): Promise<number> {
    return await this.ticketService.deleteTicket(ticketId);
  }

  @Get('patients/authed-patient')
  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOperation({ summary: 'Get all tickets of the Patient' })
  @ApiResponse({
    status: 200,
    description: 'The list of tickets has been successfully retrieved.',
    type: PaginationTicketResponse,
  })
  async getTicketsOfPatient(@User() user: UserPlayloadResponse, @Query() paginationDto: PaginationDto): Promise<PaginationTicketResponse> {
    return this.ticketService.getTicketsOfPatient(user, paginationDto);
  }

  @Patch(':id/patients/authed-patient')
  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiConsumes('multipart/form-data')
  @ApiOperation({ summary: 'Update a ticket by patient' })
  @ApiResponse({ status: 200, type: PatientTicketResponse })
  @UseInterceptors(FilesInterceptor('attachments', 5))
  async updateTicketByPatient(
    @Param('id') id: bigint,
    @Body() dto: UpdateTicketByPatientDto,
    @UploadedFiles(ImagesUploadPipe) attachments?: Express.Multer.File[],
  ): Promise<PatientTicketResponse> {
    return this.ticketService.updateTicketByPatient(id, dto, attachments);
  }

  @Patch(':id/admins/authed-admin')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @ApiConsumes('multipart/form-data')
  @UseInterceptors(FilesInterceptor('attachments', 10))
  @ApiOperation({ summary: 'Update a ticket by admin' })
  @ApiParam({ name: 'id', type: 'integer', required: true, description: 'The ID of the ticket to update.' })
  @ApiResponse({
    status: 200,
    description: 'The ticket was successfully updated.',
    type: TicketResponse,
  })
  @ApiResponse({ status: 400, description: 'Bad Request.' })
  @ApiResponse({ status: 404, description: 'Ticket not found.' })
  async updateTicketByAdmin(
    @User() user: UserPlayloadResponse,
    @Param('id') id: bigint,
    @Body() dto: UpdateTicketByAdminDto,
    @UploadedFiles(ImagesUploadPipe) attachments?: Express.Multer.File[],
  ): Promise<TicketResponse> {
    return this.ticketService.updateTicketByAdmin(user, id, dto, attachments);
  }
}
