import { Body, Controller, Get, Param, ParseIntPipe, Post, Put, UseInterceptors } from '@nestjs/common';
import { ApiBody, ApiCreatedResponse, ApiOkResponse, ApiResponse, ApiTags } from '@nestjs/swagger';
import { User } from 'src/common/decorators/auth.decorator';
import { JwtAuth } from 'src/common/decorators/jwt-auth.decorator';
import { IdDto } from 'src/common/dto/id.dto';
import { Serialize } from 'src/common/interceptors/response-serialization.interceptor';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';
import { AssessmentsService } from './assessments.service';
import { AddAssessmentDto } from './dto/add-assessment.dto';
import { FetchAssessmentResponse } from './resposnes/fetch-assessment.response';
import { RoleDecoratorEnum } from 'src/common/enums/role-decorator.enum';
import { FetchAllAssessmentResponse } from 'src/modules/assessments/resposnes/fetch-all-assessment.response';
import { HabitInformationResponse } from '../onboarding/response/habit-info.response';
import { CurrentAssessmentResponse } from 'src/modules/assessments/resposnes/current-assessment.response';
import { HttpTokenCacheInterceptor } from 'src/common/interceptors/http-token-cache.interceptor';
import { BigIntTransformerInterceptor } from 'src/common/interceptors/bigInt-transformer.interceptor';
import { TobaccoResponse } from 'src/modules/tobaccos/response/tobacco.response';
import { AllergyResponse } from 'src/modules/allergies/response/allergy.response';
import { MotivationResponse } from 'src/modules/motivations/response/motivation.response';
import { RiskFactorResponse } from 'src/modules/risk-factors/response/risk-factor.response';
import { SmokingTriggerResponse } from 'src/modules/smoking-triggers/response/smoking-trigger.response';
import { PersonalInformationResponse } from 'src/modules/onboarding/response/personal-info.response';
import { CreatePersonalInfoAssessmentDto } from 'src/modules/assessments/dto/create-personal-info-assessment.dto';
import { CreateHabitInfoAssessmentDto } from 'src/modules/assessments/dto/create-habit-info-assessment.dto';
import { AddTobaccoAssessmentDto } from 'src/modules/assessments/dto/add-tobacco-assessment.dto';
import { AddAllergyAssessmentDto } from 'src/modules/assessments/dto/add-allergy-assessment.dto';
import { UpdateRiskFactorAssessmentDto } from 'src/modules/assessments/dto/update-risk-factor-assessment.dto';
import { UpdateSmokingTriggerAssessmentDto } from 'src/modules/assessments/dto/update-smoking-trigger-assessment.dto';
import { UpdateMotivationAssessmentDto } from 'src/modules/assessments/dto/update-motivation-assessment.dto';
import { UpdateScoreDto } from 'src/modules/assessments/dto/update-score.dto';

@ApiTags('Assessments')
@Controller('assessments')
export class AssessmentsController {
  constructor(private assessmentService: AssessmentsService) {}

  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiCreatedResponse({ type: FetchAssessmentResponse })
  @Post()
  async addAssessment(@Body() assessmentName: AddAssessmentDto, @User() user: UserPlayloadResponse): Promise<FetchAssessmentResponse> {
    return await this.assessmentService.addAssessment(assessmentName, user);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: FetchAllAssessmentResponse, isArray: true })
  @Get('/all')
  @Serialize(FetchAllAssessmentResponse)
  async fetchAssessments(@User() user: UserPlayloadResponse): Promise<FetchAllAssessmentResponse[]> {
    return await this.assessmentService.fetchAssessments(user);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: CurrentAssessmentResponse })
  @Get('current-assessment')
  @UseInterceptors(HttpTokenCacheInterceptor, BigIntTransformerInterceptor)
  async fetchCurrentAssessmentDetail(@User() user: UserPlayloadResponse): Promise<CurrentAssessmentResponse> {
    return await this.assessmentService.fetchCurrentAssessmentDetail(user);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: FetchAssessmentResponse })
  @Get(':id')
  async fetchAssessment(@Param('id', ParseIntPipe) id: number, @User() user: UserPlayloadResponse): Promise<FetchAssessmentResponse> {
    return await this.assessmentService.fetchAssessment(id, user);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: PersonalInformationResponse })
  @ApiResponse({ status: 404, description: 'User assessment not found!' })
  @Put(':id/personal-informations')
  async updatePersonalInformation(
    @Body() createPersonalInfoAssessmentDto: CreatePersonalInfoAssessmentDto,
    @User() user: UserPlayloadResponse,
    @Param('id', ParseIntPipe) id: number,
  ): Promise<PersonalInformationResponse> {
    return await this.assessmentService.updatePersonalInformation(createPersonalInfoAssessmentDto, id, user);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: HabitInformationResponse })
  @ApiResponse({ status: 404, description: 'User assessment not found!' })
  @Put(':id/habit-informations')
  async updateHabit(
    @Body() createHabitInfoAssessmentDto: CreateHabitInfoAssessmentDto,
    @User() user: UserPlayloadResponse,
    @Param('id', ParseIntPipe) id: number,
  ): Promise<HabitInformationResponse> {
    return await this.assessmentService.updateHabit(createHabitInfoAssessmentDto, id, user);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: TobaccoResponse })
  @Put(':id/tobaccos')
  async updateTobacco(
    @Body() addTobaccoAssessmentDto: AddTobaccoAssessmentDto,
    @Param('id', ParseIntPipe) id: number,
    @User() user: UserPlayloadResponse,
  ): Promise<TobaccoResponse> {
    return await this.assessmentService.updateTobacco(addTobaccoAssessmentDto, id, user);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: AllergyResponse })
  @Put(':id/allergies')
  async updateAllergy(
    @Body() addAllergyAssessmentDto: AddAllergyAssessmentDto,
    @Param('id', ParseIntPipe) id: number,
    @User() user: UserPlayloadResponse,
  ): Promise<AllergyResponse> {
    return await this.assessmentService.updateAllergy(addAllergyAssessmentDto, id, user);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: Boolean })
  @ApiResponse({ status: 404, description: 'User assessment not found!' })
  @ApiBody({ type: IdDto, isArray: true })
  @Put(':id/fagerstroms')
  async updateFagerstromReport(@Body() IdsDto: IdDto[], @Param('id', ParseIntPipe) id: number, @User() user: UserPlayloadResponse): Promise<boolean> {
    return await this.assessmentService.updateFagerstromReport(IdsDto, id, user);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: Boolean })
  @ApiResponse({ status: 404, description: 'User assessment not found!' })
  @Put(':id/fagerstrom-score')
  async updateFagerstromScore(@Body() updateScoreDto: UpdateScoreDto, @Param('id', ParseIntPipe) id: number, @User() user: UserPlayloadResponse): Promise<number> {
    return await this.assessmentService.updateFagerstromScore(updateScoreDto, user, id);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: RiskFactorResponse })
  @Put(':id/risk-factors')
  async updateRiskFactor(
    @Body() updateRiskFactorAssessmentDto: UpdateRiskFactorAssessmentDto,
    @Param('id', ParseIntPipe) id: number,
    @User() user: UserPlayloadResponse,
  ): Promise<RiskFactorResponse> {
    return await this.assessmentService.updateRiskFactors(updateRiskFactorAssessmentDto, id, user);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: SmokingTriggerResponse })
  @Put(':id/smoking-triggers')
  async updateSmokingTrigger(
    @Body() updateSmokingTriggerAssessmentDto: UpdateSmokingTriggerAssessmentDto,
    @Param('id', ParseIntPipe) id: number,
    @User() user: UserPlayloadResponse,
  ): Promise<SmokingTriggerResponse> {
    return await this.assessmentService.updateSmokingTrigger(updateSmokingTriggerAssessmentDto, id, user);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: MotivationResponse })
  @Put(':id/motivations')
  async updateMotivation(
    @Body() updateMotivationAssessmentDto: UpdateMotivationAssessmentDto,
    @Param('id', ParseIntPipe) id: number,
    @User() user: UserPlayloadResponse,
  ): Promise<MotivationResponse> {
    return await this.assessmentService.updateMotivations(updateMotivationAssessmentDto, id, user);
  }
}
