import { Body, Controller, Get, Param, ParseIntPipe, Post } from '@nestjs/common';
import { ApiTags, ApiOperation, ApiResponse, ApiBody } from '@nestjs/swagger';
import { JwtAuth } from 'src/common/decorators/jwt-auth.decorator';
import { RoleDecoratorEnum } from 'src/common/enums/role-decorator.enum';
import { AddAvailabilitiesDto, AvailabilityPerDayDto } from './dto/add-availability-per-day';
import { AvailabilitiesService } from './availabilities.service';
import { User } from 'src/common/decorators/auth.decorator';
import { UserPlayloadResponse } from '../patients/response/user-playload.response';

@ApiTags('availabilities')
@Controller('availabilities')
export class AvailabilitiesController {
  constructor(private availabilitieService: AvailabilitiesService) {}

  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST])
  @Post('')
  @ApiOperation({ summary: 'Add availability for health specialist' })
  @ApiResponse({ type: AvailabilityPerDayDto, status: 201, description: 'Availability added successfully' })
  @ApiBody({ type: AddAvailabilitiesDto, description: 'Availability details' })
  async createAvailability(@Body() availabilityDto: AddAvailabilitiesDto, @User() user: UserPlayloadResponse): Promise<AvailabilityPerDayDto[]> {
    return await this.availabilitieService.addAvailability(user, availabilityDto);
  }

  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST])
  @Get('')
  @ApiOperation({ summary: 'Get availabilities for health specialist' })
  @ApiResponse({ status: 200, type: [AvailabilityPerDayDto], description: 'Availabilities retrieved successfully' })
  async getHealthSpecialistAvailabilities(@User() user: UserPlayloadResponse): Promise<AvailabilityPerDayDto[]> {
    return await this.availabilitieService.getHealthSpecialistAvailabilities(user);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @Get('patients/:patientId')
  @ApiOperation({ summary: 'Get availabilities for a specific patient' })
  @ApiResponse({ status: 200, type: [AvailabilityPerDayDto], description: 'Availabilities retrieved successfully' })
  async getHealthSpecialistAvailabilitiesByPatient(@Param('patientId', ParseIntPipe) id: number): Promise<AvailabilityPerDayDto[]> {
    return await this.availabilitieService.getCurrentDayAvailability(id);
  }
}
