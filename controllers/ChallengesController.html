import { Body, Controller, Get, Param, Put, UseInterceptors } from '@nestjs/common';
import { ApiBody, ApiOkResponse, ApiTags } from '@nestjs/swagger';
import { User } from 'src/common/decorators/auth.decorator';
import { JwtAuth } from 'src/common/decorators/jwt-auth.decorator';
import { RoleDecoratorEnum } from 'src/common/enums/role-decorator.enum';
import { BigIntTransformerInterceptor } from 'src/common/interceptors/bigInt-transformer.interceptor';
import { HttpTokenCacheInterceptor } from 'src/common/interceptors/http-token-cache.interceptor';
import { ChallengesService } from 'src/modules/challenges/challenges.service';
import { ChallengeWithProgress } from 'src/modules/challenges/response/challenge-with-progress.response';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';
import { ChallengeNameDto } from './dto/challenge.name.dto';

@Controller('challenges')
@ApiTags('Challenges')
export class ChallengesController {
  constructor(private readonly challengesService: ChallengesService) {}

  @Put('/reset')
  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: Boolean })
  @ApiBody({ type: ChallengeNameDto })
  async resetChallenge(@User() user: UserPlayloadResponse, @Body() challenge: ChallengeNameDto): Promise<boolean> {
    return await this.challengesService.createChallenge(challenge.type, user);
  }
  @Put(':challengeId')
  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: Boolean })
  async markDoneChallenge(@Param('challengeId') challengeId: number): Promise<boolean> {
    return await this.challengesService.markChallengeAsCompleted(challengeId);
  }

  @Get('/my-challenges')
  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: ChallengeWithProgress })
  @UseInterceptors(HttpTokenCacheInterceptor, BigIntTransformerInterceptor)
  async myChallenges(@User() user: UserPlayloadResponse): Promise<ChallengeWithProgress> {
    return await this.challengesService.myChallenges(user);
  }
}
