import { Body, Controller, Delete, Get, Param, ParseIntPipe, Post, Put, UseInterceptors } from '@nestjs/common';
import { ApiOkResponse, ApiTags } from '@nestjs/swagger';
import { LanguageEnum, PermissionEnum } from '@prisma/client';
import { HeaderLanguage } from 'src/common/decorators/header-language';
import { JwtAuth } from 'src/common/decorators/jwt-auth.decorator';
import { Permissions } from 'src/common/decorators/permissions.decorator';
import { RoleDecoratorEnum } from 'src/common/enums/role-decorator.enum';
import { BigIntTransformerInterceptor } from 'src/common/interceptors/bigInt-transformer.interceptor';
import { HttpLanguageCacheInterceptor } from 'src/common/interceptors/http-language-cache.interceptor';
import { ArticlePackagesService } from 'src/modules/article-packages/article-packages.service';
import { AddArticlePackageDto } from 'src/modules/article-packages/dto/add-article-package.dto';
import { AddRecommendedArticleDto } from 'src/modules/article-packages/dto/add-recommended-article.Dto';
import { UpdateArticlePackage } from 'src/modules/article-packages/dto/update-article-package.dto';
import { UpdateRecommendedArticleDto } from 'src/modules/article-packages/dto/update-recommended-article.Dto';
import { ArticlePackageWithCountResponse } from 'src/modules/article-packages/response/article-package.response';
import { FetchArticleById } from 'src/modules/article-packages/response/fetch-article-by-id.response';

@Controller('article-packages')
@ApiTags('Article packages')
export class ArticlePackagesController {
  constructor(private articlePackagesService: ArticlePackagesService) {}

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_ARTICLE_PACKAGES])
  @ApiOkResponse({ type: [ArticlePackageWithCountResponse] })
  @Get('')
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  async ArticlePackages(): Promise<ArticlePackageWithCountResponse[]> {
    return await this.articlePackagesService.articlePackages();
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.ADD_ARTICLE_PACKAGES])
  @ApiOkResponse({ type: ArticlePackageWithCountResponse })
  @Post('')
  async addArticlePackage(@Body() addArticlePackageDto: AddArticlePackageDto): Promise<ArticlePackageWithCountResponse> {
    return await this.articlePackagesService.addArticlePackage(addArticlePackageDto);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_ARTICLE_PACKAGES])
  @ApiOkResponse({ type: FetchArticleById })
  @Get(':id')
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  async articlePackageById(@HeaderLanguage() language: LanguageEnum, @Param('id', ParseIntPipe) id: number): Promise<FetchArticleById> {
    return await this.articlePackagesService.articlePackageById(language, id);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.UPDATE_ARTICLE_PACKAGES])
  @ApiOkResponse({ type: ArticlePackageWithCountResponse })
  @Put(':id')
  async updateArticlePackage(@Body() updateArticlePackage: UpdateArticlePackage, @Param('id', ParseIntPipe) id: number): Promise<ArticlePackageWithCountResponse> {
    return await this.articlePackagesService.updateArticlePackage(updateArticlePackage, id);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DELETE_ARTICLE_PACKAGES])
  @ApiOkResponse({ type: ArticlePackageWithCountResponse })
  @Delete(':id')
  async deleteArticlePackage(@Param('id', ParseIntPipe) id: number): Promise<ArticlePackageWithCountResponse> {
    return await this.articlePackagesService.deletePackage(id);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.ADD_ARTICLE_PACKAGES])
  @ApiOkResponse({ type: ArticlePackageWithCountResponse })
  @Post(':id/articles')
  async addArticleToArticlePackage(
    @Body() addRecommendedArticleDto: AddRecommendedArticleDto,
    @Param('id', ParseIntPipe) id: number,
  ): Promise<ArticlePackageWithCountResponse> {
    return await this.articlePackagesService.addArticleToArticlePackage(addRecommendedArticleDto, id);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.UPDATE_ARTICLE_PACKAGES])
  @ApiOkResponse({ type: ArticlePackageWithCountResponse })
  @Put(':id/articles/:articleId')
  async updateArticleInArticlePackage(
    @Body() updateRecommendedArticleDto: UpdateRecommendedArticleDto,
    @Param('id', ParseIntPipe) id: number,
    @Param('articleId', ParseIntPipe) articleId: number,
  ): Promise<ArticlePackageWithCountResponse> {
    return await this.articlePackagesService.updateArticleInArticlePackage(updateRecommendedArticleDto, id, articleId);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DELETE_ARTICLE_PACKAGES])
  @ApiOkResponse({ type: ArticlePackageWithCountResponse })
  @Delete(':id/articles/:articleId')
  async deleteArticleFromArticlePackage(
    @Param('id', ParseIntPipe) id: number,
    @Param('articleId', ParseIntPipe) articleId: number,
  ): Promise<ArticlePackageWithCountResponse> {
    return await this.articlePackagesService.deleteArticleFromArticlePackage(id, articleId);
  }
}
