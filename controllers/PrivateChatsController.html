import { ApiCreatedResponse, ApiOkResponse, ApiTags } from '@nestjs/swagger';
import { Body, Controller, Delete, Get, Param, ParseIntPipe, Post, Put, Query } from '@nestjs/common';
import { PrivateChatsService } from 'src/modules/private-chats/private-chats.service';
import { JwtAuth } from 'src/common/decorators/jwt-auth.decorator';
import { User } from 'src/common/decorators/auth.decorator';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';
import { RoleDecoratorEnum } from 'src/common/enums/role-decorator.enum';
import { PaginationFlatPrivateChatListResponse, PaginationPrivateChatMessageResponse } from 'src/common/responses/Paggination-data.response';
import { CreatePrivateMessageDto } from 'src/modules/private-chats/dto/create-private-message.dto';
import { PrivateChatSimpleResponse } from 'src/modules/private-chats/response/private-chat-simple.response';
import { PaginationCursorPrivateChatMessageResponse } from 'src/common/responses/paggination-cursor.response';
import { PrivateChatByIdResponse } from 'src/modules/private-chats/response/private-chat-by-is.response';
import { SearchWithCursorPaginationDto } from 'src/common/dto/search-with-cursor-pagination.dto';
import { SearchWithPaginationDto } from 'src/common/dto/search-with-pagination.dto';

@Controller('private-chats')
@ApiTags('Private chats')
export class PrivateChatsController {
  constructor(private privateChatsService: PrivateChatsService) {}

  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST])
  @Post('')
  @ApiCreatedResponse({ type: PrivateChatSimpleResponse })
  async createPrivateChat(@User() user: UserPlayloadResponse, @Body() createPrivateMessageDto: CreatePrivateMessageDto): Promise<PrivateChatSimpleResponse> {
    return await this.privateChatsService.createPrivateChat(user, createPrivateMessageDto);
  }

  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST, RoleDecoratorEnum.USER])
  @Get('my-private-chats')
  @ApiOkResponse({ type: PaginationFlatPrivateChatListResponse })
  async fetchPrivateChats(@User() user: UserPlayloadResponse, @Query() searchWithPaginationDto: SearchWithPaginationDto): Promise<PaginationFlatPrivateChatListResponse> {
    return await this.privateChatsService.fetchPrivateChats(user, searchWithPaginationDto);
  }

  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST])
  @Get(':id')
  @ApiOkResponse({ type: PrivateChatByIdResponse })
  async fetchPrivateChat(@Param('id') id: number): Promise<PrivateChatByIdResponse> {
    return await this.privateChatsService.fetchPrivateChatById(id);
  }

  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST])
  @Delete(':id')
  @ApiOkResponse({ type: PrivateChatByIdResponse })
  async deletePrivateChat(@User() user: UserPlayloadResponse, @Param('id') id: number): Promise<PrivateChatByIdResponse> {
    return await this.privateChatsService.deletePrivateChatById(user, id);
  }

  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST])
  @Get(':id/messages')
  @ApiOkResponse({ type: PaginationCursorPrivateChatMessageResponse })
  async healthSpecialistFetchPrivateChatMessages(
    @User() user: UserPlayloadResponse,
    @Query() searchWithCursorPaginationDto: SearchWithCursorPaginationDto,
    @Param('id', ParseIntPipe) id: number,
  ): Promise<PaginationCursorPrivateChatMessageResponse> {
    return await this.privateChatsService.healthSpecialistFetchPrivateChatMessages(user, searchWithCursorPaginationDto, id);
  }

  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST])
  @Put(':id/disable')
  @ApiOkResponse({ type: PrivateChatSimpleResponse })
  async disablePrivateChat(@User() user: UserPlayloadResponse, @Param('id', ParseIntPipe) id: number): Promise<PrivateChatSimpleResponse> {
    return await this.privateChatsService.disablePrivateChat(user, id);
  }

  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST])
  @Put(':id/enable')
  @ApiOkResponse({ type: PrivateChatSimpleResponse })
  async enablePrivateChat(@User() user: UserPlayloadResponse, @Param('id', ParseIntPipe) id: number): Promise<PrivateChatSimpleResponse> {
    return await this.privateChatsService.enablePrivateChat(user, id);
  }

  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST])
  @Get(':id/messages/offset-pagination')
  @ApiOkResponse({ type: PaginationPrivateChatMessageResponse })
  async healthSpecialistFetchPrivateChatMessagesOffsetPagination(
    @User() user: UserPlayloadResponse,
    @Query() searchWithPaginationDto: SearchWithPaginationDto,
    @Param('id', ParseIntPipe) id: number,
  ): Promise<PaginationPrivateChatMessageResponse> {
    return await this.privateChatsService.healthSpecialistFetchPrivateChatMessagesOffsetPagination(user, searchWithPaginationDto, id);
  }
}
