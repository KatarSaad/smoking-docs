@ApiTags('Articles')
@Controller('articles')
export class ArticlesController {
  constructor(private articlesService: ArticlesService) {}

  @Get('/published')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_CONTENTS, PermissionEnum.GENERATE_CONTENT])
  @ApiOkResponse({ type: PaginationArticles })
  @UseInterceptors(HttpTokenCacheInterceptor, BigIntTransformerInterceptor)
  @CacheTTL(Number(process.env.REDIS_LONG_TTL))
  async fetchArticles(
    @HeaderLanguage() language: LanguageEnum,
    @User() user: UserPlayloadResponse,
    @Query() fetchArticlesDto: FetchArticlesDto,
  ): Promise<PaginationArticles> {
    return await this.articlesService.fetchPublishedArticles(language, user, fetchArticlesDto);
  }

  ///////////////////////////////////// Fetch Favorite Articles of the user//////////////////////////////////

  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  @CacheTTL(Number(process.env.REDIS_TTL))
  @ApiOkResponse({ type: ArticleBookmedResponse })
  @Get('book-marks')
  @JwtAuth([RoleDecoratorEnum.USER])
  async getFavoriteArticles(
    @User() user: UserPlayloadResponse,
    @HeaderLanguage() language: LanguageEnum,

    @Query() fetchPatienthArticlesDto: FetchPatienthArticlesDto,
  ): Promise<PaginationArticlesBookmarked> {
    return await this.articlesService.getFavoriteArticlesOfUser(language, user.id, fetchPatienthArticlesDto);
  }

  @Post('/published')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.GENERATE_CONTENT])
  @UseInterceptors(FileInterceptor('file'))
  @ApiConsumes('multipart/form-data')
  @ApiCreatedResponse({ type: ArticleResponse })
  async createArticle(
    @HeaderLanguage() language: LanguageEnum,
    @User() user: UserPlayloadResponse,
    @Body() addArticleDto: AddArticleDto,
    @UploadedFile(ImageUploadPipe) file?: Express.Multer.File,
  ): Promise<ArticleResponse> {
    return await this.articlesService.createArticle(language, user, addArticleDto, file);
  }

  @Get('patients')
  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: PaginationArticles })

  @Get('scheduled')

    @Body() addDraftArticleDto: AddDraftArticleDto,
    @Param('id', ParseIntPipe) id: number,

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])

  async scheduleArticle(
    @HeaderLanguage() language: LanguageEnum,
    @User() user: UserPlayloadResponse,
    @Body() scheduleArticleDto: ScheduleArticleDto,
