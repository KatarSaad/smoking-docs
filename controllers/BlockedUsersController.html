import { CacheInterceptor } from '@nestjs/cache-manager';
import { Body, Controller, Get, Param, ParseIntPipe, Post, Put, Query, UseInterceptors } from '@nestjs/common';
import { ApiCreatedResponse, ApiOkResponse, ApiTags } from '@nestjs/swagger';
import { PermissionEnum } from '@prisma/client';
import { User } from 'src/common/decorators/auth.decorator';
import { JwtAuth } from 'src/common/decorators/jwt-auth.decorator';
import { Permissions } from 'src/common/decorators/permissions.decorator';
import { RoleDecoratorEnum } from 'src/common/enums/role-decorator.enum';
import { BigIntTransformerInterceptor } from 'src/common/interceptors/bigInt-transformer.interceptor';
import { PaginationBlockedUserResponseResponse } from 'src/common/responses/Paggination-data.response';
import { BlockedUsersService } from 'src/modules/blocked-users/blocked-users.service';
import { BlockUserDto } from 'src/modules/blocked-users/dto/block-user.dto';
import { FilterBlockedUsersDto } from 'src/modules/blocked-users/dto/filter-blocked-users.dto';
import { UpdateBlockDto } from 'src/modules/blocked-users/dto/update-block.dto';
import { BlockUserResponse } from 'src/modules/blocked-users/response/block-user.response';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';
import { BlockedUserHistoryResponse } from './response/blocked-user-history.response';
import { IsBlockedResponse } from './response/is-blocked.response';
import { BlockUserTypeEnum } from 'src/common/enums/block-user-type.enum';

@Controller('blocked-users')
@ApiTags('Blocked users')
export class BlockedUsersController {
  constructor(private blockedUsersService: BlockedUsersService) {}

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.CHAT_MODERATION])
  @ApiOkResponse({ type: PaginationBlockedUserResponseResponse })
  @Get()
  async blockedUsers(@Query() filterBlockedUsersDto: FilterBlockedUsersDto): Promise<PaginationBlockedUserResponseResponse> {
    return await this.blockedUsersService.blockedUsers(filterBlockedUsersDto);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: BlockUserResponse })
  @Get('one-to-one')
  async blockedUsersOneToOne(@User() user: UserPlayloadResponse): Promise<BlockUserResponse[]> {
    return await this.blockedUsersService.blockedUsersOneToOne(user);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.CHAT_MODERATION])
  @ApiOkResponse({ type: BlockedUserHistoryResponse })
  @Get(':userId/history')
  @UseInterceptors(CacheInterceptor, BigIntTransformerInterceptor)
  async blockedUserHistory(@Param('userId', ParseIntPipe) userId: number): Promise<BlockedUserHistoryResponse> {
    return await this.blockedUsersService.blockedUserHistory(userId, null, BlockUserTypeEnum.GLOBAL_BLOCK_COMMUNITY_CHAT);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN, RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: IsBlockedResponse })
  @Get(':userId/isBlocked')
  @UseInterceptors(CacheInterceptor, BigIntTransformerInterceptor)
  async isBlocked(@Param('userId', ParseIntPipe) userId: number): Promise<IsBlockedResponse> {
    return await this.blockedUsersService.isUserBlocked(userId, null, BlockUserTypeEnum.GLOBAL_BLOCK_COMMUNITY_CHAT);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.CHAT_MODERATION])
  @ApiCreatedResponse({ type: BlockUserResponse })
  @Post()
  async blockUser(@User() user: UserPlayloadResponse, @Body() blockUserDto: BlockUserDto): Promise<BlockUserResponse> {
    return await this.blockedUsersService.blockUser(user, blockUserDto, BlockUserTypeEnum.GLOBAL_BLOCK_COMMUNITY_CHAT);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.CHAT_MODERATION])
  @ApiOkResponse({ type: BlockUserResponse })
  @Put(':id')
  async updateBlock(@Param('id', ParseIntPipe) id: number, @Body() updateBlockDto: UpdateBlockDto): Promise<BlockUserResponse> {
    return await this.blockedUsersService.updateBlock(id, updateBlockDto);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.CHAT_MODERATION])
  @ApiOkResponse({ type: BlockUserResponse })
  @Put(':userId/unblock')
  async unblock(@User() user: UserPlayloadResponse, @Param('userId', ParseIntPipe) userId: number): Promise<BlockUserResponse> {
    return await this.blockedUsersService.unblock(user, userId, BlockUserTypeEnum.GLOBAL_BLOCK_COMMUNITY_CHAT);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: BlockUserResponse })
  @Put(':userId/one-to-one-unblock')
  async oneToOneUnblock(@User() user: UserPlayloadResponse, @Param('userId', ParseIntPipe) userId: number): Promise<BlockUserResponse> {
    return await this.blockedUsersService.unblock(user, userId, BlockUserTypeEnum.ONE_TO_ONE_BLOCK_COMMUNITY_CHAT);
  }
}
