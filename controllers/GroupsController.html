import { Body, Controller, Delete, Get, Param, Post, Put, Query, ParseIntPipe, UseInterceptors } from '@nestjs/common';
import { ApiExtraModels, ApiOkResponse, ApiQuery, ApiResponse, ApiTags, getSchemaPath } from '@nestjs/swagger';
import { PermissionEnum } from '@prisma/client';
import { User } from 'src/common/decorators/auth.decorator';
import { JwtAuth } from 'src/common/decorators/jwt-auth.decorator';
import { Permissions } from 'src/common/decorators/permissions.decorator';
import { RoleDecoratorEnum } from 'src/common/enums/role-decorator.enum';
import { Serialize } from 'src/common/interceptors/response-serialization.interceptor';
import { PaginationGroupNamesOnlyResponse, PaginationGroups } from 'src/common/responses/Paggination-data.response';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';
import { AddGroupDto } from './dto/add-group.dto';
import { AssignPatientToGroupDto } from './dto/assign-patient-group.dto';
import { FetchGroupFilterDto } from './dto/fetch-group-filter.dto';
import { GroupsService } from './groups.service';
import { GroupResponse } from './response/group.response';
import { RangeDto } from 'src/common/dto/range.dto';
import { PaginationOptionalDto } from 'src/common/dto/pagination-optional.dto';
import { HttpLanguageCacheInterceptor } from 'src/common/interceptors/http-language-cache.interceptor';
import { BigIntTransformerInterceptor } from 'src/common/interceptors/bigInt-transformer.interceptor';
import { GroupHasNrtResponse } from 'src/modules/groups/response/group-has-nrt.response';
import { GroupHasArticlePackageResponse } from 'src/modules/groups/response/group-has-article-package.response';

@ApiTags('Groups')
@Controller('groups')
export class GroupsController {
  constructor(private groupsService: GroupsService) {}

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_GROUPS])
  @Get()
  @ApiOkResponse({ type: PaginationGroups })
  @Serialize(PaginationGroups)
  @ApiExtraModels(RangeDto)
  @ApiQuery({
    name: 'relapsesRange',
    required: false,
    schema: {
      type: 'array',
      items: {
        type: 'object',
        $ref: getSchemaPath(RangeDto),
      },
    },
  })
  @ApiQuery({
    name: 'agesRange',
    required: false,
    schema: {
      type: 'array',
      items: {
        type: 'object',
        $ref: getSchemaPath(RangeDto),
      },
    },
  })
  @ApiQuery({
    name: 'cigarretesRange',
    required: false,
    schema: {
      type: 'array',
      items: {
        type: 'object',
        $ref: getSchemaPath(RangeDto),
      },
    },
  })
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  async fetchGroups(@Query() fetchGroupFilterDto: FetchGroupFilterDto): Promise<PaginationGroups> {
    return this.groupsService.getAllGroups(fetchGroupFilterDto);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.ADD_GROUPS])
  @Post()
  @ApiOkResponse({ type: GroupResponse })
  async addGroup(@Body() addGroupDto: AddGroupDto, @User() user: UserPlayloadResponse) {
    return this.groupsService.addGroup(addGroupDto, user);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_NRT])
  @ApiOkResponse({ type: [GroupHasArticlePackageResponse] })
  @Get('article-packages')
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  async groupArticlePackageList(): Promise<GroupHasArticlePackageResponse[]> {
    return await this.groupsService.groupArticlePackageList();
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_NRT])
  @ApiOkResponse({ type: [GroupHasNrtResponse] })
  @Get('nrts')
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  async groupNrtList(): Promise<GroupHasNrtResponse[]> {
    return await this.groupsService.groupNrtList();
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_GROUPS, PermissionEnum.DISPLAY_CONTENTS, PermissionEnum.GENERATE_CONTENT, PermissionEnum.DISPLAY_GROUPS_NAMES]) //DISPLAY_CONTENTS & GENERATE_CONTENT  for the dropdown list in articles
  @Get('/names-only')
  @ApiOkResponse({ type: PaginationGroupNamesOnlyResponse })
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  async fetchGroupsNamesOnly(@Query() paginationOptionalDto: PaginationOptionalDto): Promise<PaginationGroupNamesOnlyResponse> {
    return this.groupsService.fetchGroupsNamesOnly(paginationOptionalDto);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.ASSIGN_GROUPS])
  @Put('/assign-patient')
  @ApiResponse({ status: 403, description: 'This is for patients only!' })
  @ApiResponse({ status: 404, description: 'User not found!' })
  @ApiOkResponse({ type: Boolean })
  async assignPatientTOSCJourney(@Body() assignPatient: AssignPatientToGroupDto): Promise<boolean> {
    return await this.groupsService.assignPatientToGroup(assignPatient);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @Put('/reset-patient-groups')
  @ApiOkResponse({ type: Boolean })
  async assignAuthedPatientToGroups(@User() user: UserPlayloadResponse): Promise<boolean> {
    return await this.groupsService.assignAuthedPatientToGroups(user);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DELETE_GROUPS])
  @Delete(':id')
  @ApiOkResponse({ type: GroupResponse })
  async deleteGroup(@Param('id', ParseIntPipe) id: number) {
    return this.groupsService.deleteGroup(id);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_GROUPS])
  @Get(':id')
  @ApiOkResponse({ type: GroupResponse })
  @Serialize(GroupResponse)
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  async fetchGroup(@Param('id', ParseIntPipe) id: number): Promise<GroupResponse> {
    return this.groupsService.fetchGroupById(id);
  }
}
