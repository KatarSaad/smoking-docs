import { Body, Controller, Delete, Get, Param, ParseIntPipe, Patch, Post, Query, UseInterceptors } from '@nestjs/common';
import { ApiCreatedResponse, ApiOkResponse, ApiTags } from '@nestjs/swagger';
import { LanguageEnum, PermissionEnum } from '@prisma/client';
import { HeaderLanguage } from 'src/common/decorators/header-language';
import { JwtAuth } from 'src/common/decorators/jwt-auth.decorator';
import { Permissions } from 'src/common/decorators/permissions.decorator';
import { PaginationOptionalDto } from 'src/common/dto/pagination-optional.dto';
import { RoleDecoratorEnum } from 'src/common/enums/role-decorator.enum';
import { PaginationFaqCategories } from 'src/common/responses/Paggination-data.response';
import { FetchCategoriesWithFaqDto } from 'src/modules/faq-categories/dto/fetch-categories-with-faq.dto';
import { PaginationMultiLanguageFaqCategories } from './../../common/responses/Paggination-data.response';
import { AddFaqCategoryDto } from './dto/add-faq-category.dto';
import { FaqCategoriesService } from './faq-categories.service';
import { FaqCategoryResponse } from './response/faq-category.response';
import { HttpLanguageCacheInterceptor } from 'src/common/interceptors/http-language-cache.interceptor';
import { BigIntTransformerInterceptor } from 'src/common/interceptors/bigInt-transformer.interceptor';

@ApiTags('FAQ Categories')
@Controller('faq-categories')
export class FaqCategoriesController {
  constructor(private readonly faqCategoriesService: FaqCategoriesService) {}
  @Get('')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN, RoleDecoratorEnum.USER])
  @Permissions([PermissionEnum.DISPLAY_FAQ])
  @ApiOkResponse({ type: PaginationFaqCategories })
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  async fetchArticleLabels(@HeaderLanguage() language: LanguageEnum, @Query() paginationOptionalDto: PaginationOptionalDto): Promise<PaginationFaqCategories> {
    return await this.faqCategoriesService.fetchFaqCategories(language, paginationOptionalDto);
  }

  @Post()
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.ADD_FAQ])
  @ApiCreatedResponse({ type: FaqCategoryResponse })
  async createFaqCategory(@Body() addFaqCategoryDto: AddFaqCategoryDto): Promise<FaqCategoryResponse> {
    return await this.faqCategoriesService.createFaqCategory(addFaqCategoryDto);
  }
  @Get('all-languages')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_FAQ])
  @ApiOkResponse({ type: PaginationMultiLanguageFaqCategories })
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  async fetchMultiLanguageFaqCategories(@Query() fetchCategoriesWithFaqDto: FetchCategoriesWithFaqDto): Promise<PaginationMultiLanguageFaqCategories> {
    return await this.faqCategoriesService.fetchMultiLanguageFaqCategories(fetchCategoriesWithFaqDto);
  }

  @Patch(':id')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.UPDATE_FAQ])
  @ApiOkResponse({ type: FaqCategoryResponse })
  async updateFaqCategory(@Param('id', ParseIntPipe) id: number, @Body() addFaqCategoryDto: AddFaqCategoryDto): Promise<FaqCategoryResponse> {
    return await this.faqCategoriesService.updateFaqCategory(id, addFaqCategoryDto);
  }

  @Delete(':id')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DELETE_FAQ])
  @ApiOkResponse({ type: FaqCategoryResponse })
  async deleteFaqCategory(@Param('id', ParseIntPipe) id: number): Promise<FaqCategoryResponse> {
    return await this.faqCategoriesService.deleteFaqCategory(id);
  }
}
