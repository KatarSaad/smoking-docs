import { Body, Controller, Delete, Get, Param, ParseIntPipe, Post, Put, UseInterceptors } from '@nestjs/common';
import { ApiCreatedResponse, ApiOkResponse, ApiTags } from '@nestjs/swagger';
import { LanguageEnum, PermissionEnum } from '@prisma/client';
import { HeaderLanguage } from 'src/common/decorators/header-language';
import { JwtAuth } from 'src/common/decorators/jwt-auth.decorator';
import { Permissions } from 'src/common/decorators/permissions.decorator';
import { RoleDecoratorEnum } from 'src/common/enums/role-decorator.enum';
import { BigIntTransformerInterceptor } from 'src/common/interceptors/bigInt-transformer.interceptor';
import { HttpLanguageCacheInterceptor } from 'src/common/interceptors/http-language-cache.interceptor';
import { NrtDetailResponse } from 'src/modules/nrts-detail/response/nrt-detail.response';
import { AddNrtDto } from 'src/modules/nrts/dto/add-nrt.dto';
import { UpdateNrtDto } from 'src/modules/nrts/dto/update-nrt.dto';
import { UpdatePriorityDto } from 'src/modules/nrts/dto/update-priority.dto';
import { NrtsService } from 'src/modules/nrts/nrts.service';
import { NrtResponse } from 'src/modules/nrts/response/nrt.response';

@Controller('nrts')
@ApiTags('NRTS')
export class NrtsController {
  constructor(private nrtsService: NrtsService) {}

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_NRT])
  @ApiOkResponse({ type: [NrtResponse] })
  @Get('')
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  async nrtList(@HeaderLanguage() language: LanguageEnum): Promise<NrtResponse[]> {
    return await this.nrtsService.nrtList(language);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.ADD_NRT])
  @ApiCreatedResponse({ type: NrtResponse })
  @Post('')
  async addNrt(@HeaderLanguage() language: LanguageEnum, @Body() addNrtDto: AddNrtDto): Promise<NrtResponse> {
    return await this.nrtsService.addNrt(language, addNrtDto);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.UPDATE_NRT])
  @Put(':id')
  @ApiOkResponse({ type: NrtResponse })
  async updateNrt(@HeaderLanguage() language: LanguageEnum, @Body() updateNrtDto: UpdateNrtDto, @Param('id', ParseIntPipe) id: number): Promise<NrtResponse> {
    return await this.nrtsService.updateNrt(language, updateNrtDto, id);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DELETE_NRT])
  @Delete(':id')
  @ApiOkResponse({ type: NrtResponse })
  async deleteNrt(@HeaderLanguage() language: LanguageEnum, @Param('id', ParseIntPipe) id: number): Promise<NrtResponse> {
    return await this.nrtsService.deleteNrt(language, id);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_NRT, PermissionEnum.ASSIGN_NRTS_OTHERS])
  @ApiOkResponse({ type: NrtDetailResponse })
  @Get('/patients/:id')
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  async patientNrt(@HeaderLanguage() language: LanguageEnum, @Param('id', ParseIntPipe) id: number): Promise<NrtDetailResponse> {
    return await this.nrtsService.userNrt(language, id);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.UPDATE_NRT])
  @Put(':id/priority')
  @ApiOkResponse({ type: NrtResponse })
  async updateNrtPriority(
    @HeaderLanguage() language: LanguageEnum,
    @Body() updatePriorityDto: UpdatePriorityDto,
    @Param('id', ParseIntPipe) id: number,
  ): Promise<NrtResponse> {
    return await this.nrtsService.updateNrtPriority(language, updatePriorityDto, id);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.ASSIGN_NRT, PermissionEnum.ASSIGN_NRTS_OTHERS])
  @ApiOkResponse({ type: NrtDetailResponse })
  @Put(':id/users/:userId')
  async updatePatientNrt(
    @HeaderLanguage() language: LanguageEnum,
    @Param('id', ParseIntPipe) id: number,
    @Param('userId', ParseIntPipe) userId: number,
  ): Promise<NrtDetailResponse> {
    return await this.nrtsService.updatePatientNrt(language, id, userId);
  }
}
