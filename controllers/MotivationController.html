import { Body, Controller, Get, Patch } from '@nestjs/common';
import { ApiOkResponse, ApiTags } from '@nestjs/swagger';
import { User } from 'src/common/decorators/auth.decorator';
import { JwtAuth } from 'src/common/decorators/jwt-auth.decorator';
import { RoleDecoratorEnum } from 'src/common/enums/role-decorator.enum';
import { UserPlayloadResponse } from '../patients/response/user-playload.response';
import { MotivationService } from './motivation.service';
import { MotivationResponse } from './response/motivation.response';
import { UpdateMotivationDto } from 'src/modules/motivations/dto/update-motivation.dto';

@ApiTags('Motivation')
@Controller('motivations')
export class MotivationController {
  constructor(private readonly motivationService: MotivationService) {}

  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: MotivationResponse })
  @Get()
  async findPatientMotives(@User() user: UserPlayloadResponse): Promise<MotivationResponse> {
    return await this.motivationService.findPatientMotives(user);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @Patch('')
  @ApiOkResponse({ type: MotivationResponse })
  updatePatientMotivation(@User() user: UserPlayloadResponse, @Body() updateMotivationDto: UpdateMotivationDto): Promise<MotivationResponse> {
    return this.motivationService.updatePatientMotivation(user, updateMotivationDto);
  }
}
