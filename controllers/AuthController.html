import { Body, Controller, Get, Post, Query, Request } from '@nestjs/common';
import { ApiCreatedResponse, ApiOkResponse, ApiResponse, ApiTags } from '@nestjs/swagger';
import { User } from 'src/common/decorators/auth.decorator';
import { JwtAuth } from 'src/common/decorators/jwt-auth.decorator';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';
import { AuthService } from './auth.service';
import { LoginUserDto } from './dto/login-user.dto';
import { RegisterUserDto } from './dto/register-user.dto';
import { LoginPatientResponse } from './response/login-mobile.reponse';
import { RegisterResponse } from './response/register-response.response';
import { RoleDecoratorEnum } from 'src/common/enums/role-decorator.enum';
import { Request as RequestType } from 'express';
import { EmailDto } from 'src/modules/auth/dto/email.dto';
import { UpdatePasswordWithCode } from 'src/modules/auth/dto/update-password-with-code.dto';
import { TokenDto } from 'src/common/dto/token.dto';
import { TokenResponse } from 'src/common/responses/token.response';
import { AssessmentDetailResponse } from 'src/modules/assessments/resposnes/assessment-detail.response';

@ApiTags('Auth')
@Controller('auth')
export class AuthController {
  constructor(private readonly authService: AuthService) {}

  @Post('login/dashboard')
  @ApiCreatedResponse({ type: TokenResponse })
  @ApiResponse({ status: 403, description: 'Forbidden resources!' })
  @ApiResponse({ status: 404, description: "User doesn't exist!" })
  @ApiResponse({ status: 500, description: 'Wrong password' })
  async login(@Body() loginUserDto: LoginUserDto): Promise<TokenResponse> {
    return await this.authService.loginAdmin(loginUserDto);
  }

  @Post('register')
  @ApiCreatedResponse({ type: RegisterResponse })
  @ApiResponse({ status: 500, description: 'Something went wrong' })
  async register(@Body() registerUserDto: RegisterUserDto) {
    return await this.authService.register(registerUserDto);
  }

  @JwtAuth([RoleDecoratorEnum.USER, RoleDecoratorEnum.SYSADMIN])
  @Post('forgot-password')
  async forgotPassword(@Query('email') email: string) {
    return await this.authService.forgetPassword(email);
  }

  @Post('reset-password')
  async resetPassword(@Body('newPassword') newPassword: string, @Query('qid') qid: string) {
    return await this.authService.resetPassword(newPassword, qid);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @Get('test')
  @ApiOkResponse({ type: AssessmentDetailResponse })
  async test(@User() user: UserPlayloadResponse): Promise<AssessmentDetailResponse> {
    return await this.authService.tokenChecker(user);
  }

  @JwtAuth()
  @Post('logout')
  @ApiCreatedResponse({ type: Boolean })
  @ApiResponse({ status: 200, description: 'success !' })
  @ApiResponse({ status: 500, description: 'Failed !' })
  async logout(@Request() request: RequestType, @User() user: UserPlayloadResponse): Promise<boolean> {
    const access_token = request.headers.authorization.split(' ').pop();
    const device_id = request.headers.deviceid;

    return await this.authService.logout(user, { access_token, refresh_token: access_token }, device_id as string);
  }

  @Post('login/patient')
  @ApiCreatedResponse({ type: LoginPatientResponse })
  @ApiResponse({ status: 401, description: 'Only patients can login' })
  @ApiResponse({ status: 404, description: "User doesn't exist!" })
  @ApiResponse({ status: 500, description: 'Wrong password' })
  async loginPatient(@Body() loginUserDto: LoginUserDto): Promise<LoginPatientResponse> {
    return await this.authService.loginPatient(loginUserDto);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Post('send-login-code')
  @ApiCreatedResponse({ type: Boolean })
  async sendLoginCode(@Body() emailDto: EmailDto): Promise<boolean> {
    return await this.authService.sendLoginCode(emailDto);
  }

  @Post('update-password-by-code')
  @ApiCreatedResponse({ type: Boolean })
  async updatePasswordByCode(@Body() updatePasswordWithCode: UpdatePasswordWithCode, @Query() token: TokenDto): Promise<boolean> {
    return await this.authService.updatePasswordByCode(updatePasswordWithCode, token);
  }
}
