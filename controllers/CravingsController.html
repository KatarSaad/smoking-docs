import { Body, Controller, Delete, Get, Param, ParseIntPipe, Patch, Post } from '@nestjs/common';
import { ApiCreatedResponse, ApiOkResponse, ApiTags } from '@nestjs/swagger';
import { User } from 'src/common/decorators/auth.decorator';
import { JwtAuth } from 'src/common/decorators/jwt-auth.decorator';
import { RoleDecoratorEnum } from 'src/common/enums/role-decorator.enum';
import { CravingsService } from 'src/modules/cravings/cravings.service';
import { AddCravingDto } from 'src/modules/cravings/dto/add-carving.dto';
import { UpdateCravingDto } from 'src/modules/cravings/dto/update-carving.dto';
import { CravingResponse } from 'src/modules/cravings/response/craving.response';
import { IsCravingResponse } from 'src/modules/cravings/response/is-craving.response';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';

@Controller('cravings')
@ApiTags('Cravings')
export class CravingsController {
  constructor(private cravingsService: CravingsService) {}

  @Post('')
  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiCreatedResponse({ type: CravingResponse })
  async addCraving(@User() user: UserPlayloadResponse, @Body() addCravingDto: AddCravingDto): Promise<CravingResponse> {
    return await this.cravingsService.addCraving(user, addCravingDto);
  }

  @Patch(':id')
  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiCreatedResponse({ type: CravingResponse })
  async updateCraving(@User() user: UserPlayloadResponse, @Body() updateCravingDto: UpdateCravingDto, @Param('id', ParseIntPipe) id: number): Promise<CravingResponse> {
    return await this.cravingsService.updateCraving(user, updateCravingDto, id);
  }

  @Delete(':id')
  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiCreatedResponse({ type: CravingResponse })
  async deleteCraving(@User() user: UserPlayloadResponse, @Param('id', ParseIntPipe) id: number): Promise<CravingResponse> {
    return await this.cravingsService.deleteCraving(user, id);
  }

  @Get('authed-user/is-craving')
  @JwtAuth([RoleDecoratorEnum.USER])
  @ApiOkResponse({ type: IsCravingResponse })
  async fetchIsCraving(@User() user: UserPlayloadResponse): Promise<IsCravingResponse> {
    return await this.cravingsService.fetchIsCraving(user);
  }

  @Get('private-chats/:id/is-craving')
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @ApiOkResponse({ type: IsCravingResponse })
  async fetchIsCravingById(@Param('id', ParseIntPipe) id: number): Promise<IsCravingResponse> {
    return await this.cravingsService.fetchIsCravingIsById(id);
  }
}
