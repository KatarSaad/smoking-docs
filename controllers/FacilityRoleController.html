import { Body, Controller, Delete, Get, Param, ParseIntPipe, Patch, Post, Query, UseInterceptors } from '@nestjs/common';
import { ApiCreatedResponse, ApiOkResponse, ApiResponse, ApiTags } from '@nestjs/swagger';
import { JwtAuth } from 'src/common/decorators/jwt-auth.decorator';
import { Permissions } from 'src/common/decorators/permissions.decorator';
import { Serialize } from 'src/common/interceptors/response-serialization.interceptor';
import { CreateFacilityRoleDto } from './dto/create-facility-role.dto';
import { FacilityRoleResponse } from './response/created-facility-role.response';
import { FullFacilityRoleResponse } from './response/full-facility-role.response';
import { FacilityRoleService } from './facility-role.service';
import { UpdateFacilityRoleDto } from './dto/update-role.dto';
import { FetchFaciltiyRoleDto } from './dto/fetch-facility-role.dto';
import { PaginationRoles } from 'src/common/responses/Paggination-data.response';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';
import { User } from 'src/common/decorators/auth.decorator';
import { RoleDecoratorEnum } from 'src/common/enums/role-decorator.enum';
import { LanguageEnum, PermissionEnum } from '@prisma/client';
import { HeaderLanguage } from 'src/common/decorators/header-language';
import { GetUsersByRoleResponse } from './response/get-users-by-role.response';
import { AssignFacilityRoleDto } from './dto/assign-facility-role.dto';
import { GetUnassignedUsersResponse } from './response/get-unassigned-users.response';
import { UnassignFacilityRoleDto } from './dto/unassign-facility-role.dto';
import { HttpLanguageCacheInterceptor } from 'src/common/interceptors/http-language-cache.interceptor';
import { BigIntTransformerInterceptor } from 'src/common/interceptors/bigInt-transformer.interceptor';
import { NameAndIdResponse } from 'src/common/responses/id-and-name.response';
import { CacheInterceptor } from '@nestjs/cache-manager';

@ApiTags('Facility role')
@Controller('facility-roles')
export class FacilityRoleController {
  constructor(private readonly facilityRoleService: FacilityRoleService) {}

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_ROLES])
  @ApiOkResponse({ type: PaginationRoles })
  @Get()
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  findAll(@Query() fetchFaciltiyRoleDto: FetchFaciltiyRoleDto): Promise<PaginationRoles> {
    return this.facilityRoleService.SearchRoleByName(fetchFaciltiyRoleDto);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Serialize(FacilityRoleResponse)
  @ApiCreatedResponse({ type: FacilityRoleResponse, description: 'isActive default is true' })
  @Post()
  create(@Body() createFacilityRoleDto: CreateFacilityRoleDto) {
    return this.facilityRoleService.create(createFacilityRoleDto);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.DISPLAY_ROLES])
  @ApiOkResponse({ type: [NameAndIdResponse] })
  @Get('names-only')
  @UseInterceptors(CacheInterceptor, BigIntTransformerInterceptor)
  fetchRolesNames(): Promise<NameAndIdResponse[]> {
    return this.facilityRoleService.fetchRolesNames();
  }

  //assign role to worker You can not assign roles to patient
  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.ASSIGN_ROLES])
  @ApiResponse({ status: 409, description: 'User not found!' })
  @ApiResponse({ status: 404, description: 'You can not assign roles to patient!' })
  @Patch('assign/')
  assignRole(@User() user: UserPlayloadResponse, @Body() assignFacilityRoleDto: AssignFacilityRoleDto) {
    return this.facilityRoleService.assignRole(user, assignFacilityRoleDto);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.UNASSIGN_ROLES])
  @ApiResponse({ status: 403, description: 'you can only unassign custom roles!' })
  @ApiResponse({ status: 409, description: 'User does not have that role!' })
  @Patch('unassign/')
  unassignRole(@Body() unassignFacilityRoleDto: UnassignFacilityRoleDto) {
    return this.facilityRoleService.unassignRole(unassignFacilityRoleDto);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.ASSIGN_ROLES])
  @ApiResponse({ status: 200, type: GetUnassignedUsersResponse, isArray: true })
  @ApiResponse({ status: 404, description: 'no users found!' })
  @Get('users/unassigned/')
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  getUnAssignedUsers() {
    return this.facilityRoleService.getUnAssignedUsers();
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Permissions([PermissionEnum.ASSIGN_ROLES])
  @ApiResponse({ status: 200, type: GetUsersByRoleResponse, isArray: true })
  @ApiResponse({ status: 404, description: 'no users with that role found!' })
  @Get('users/:id')
  @UseInterceptors(HttpLanguageCacheInterceptor, BigIntTransformerInterceptor)
  getUsersByRole(@Param('id', ParseIntPipe) id: number) {
    return this.facilityRoleService.getUsersByRole(id);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @ApiOkResponse({ type: FullFacilityRoleResponse })
  @Get(':id')
  findOne(@HeaderLanguage() language: LanguageEnum, @Param('id', ParseIntPipe) id: number): Promise<FullFacilityRoleResponse> {
    return this.facilityRoleService.fetchFacilityRole(language, id);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Patch(':id')
  @ApiOkResponse({ type: FullFacilityRoleResponse })
  updateRolePermissions(
    @HeaderLanguage() language: LanguageEnum,
    @Param('id', ParseIntPipe) id: number,
    @Body() updateFacilityRoleDto: UpdateFacilityRoleDto,
    @User() user: UserPlayloadResponse,
  ): Promise<FullFacilityRoleResponse> {
    return this.facilityRoleService.updateRole(language, id, updateFacilityRoleDto, user);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @Delete(':id')
  @ApiOkResponse({ type: FacilityRoleResponse })
  @ApiResponse({ status: 404, description: 'Facility role not found!' })
  remove(@Param('id', ParseIntPipe) id: number, @User() user: UserPlayloadResponse) {
    return this.facilityRoleService.removeRoleById(+id, user);
  }
