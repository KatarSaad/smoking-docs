import { PipeTransform, Injectable, ArgumentMetadata } from '@nestjs/common';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';

@Injectable()
export class ImageUploadPipe implements PipeTransform {
  // eslint-disable-next-line @typescript-eslint/no-unused-vars
  transform(file: Express.Multer.File, _: ArgumentMetadata): Express.Multer.File | boolean {
    if (!file) return false; //the pipe shouldnt run when we dont have file

    const imageSizePerKb = 500;
    const imageSizePerMb = imageSizePerKb * 1000;
    const imageSize = imageSizePerMb * 20;
    const isFileSizeAcceptable = file.size < imageSize;
    const allowedExtensions = ['image/png', 'image/jpeg', 'image/jpg', 'image/gif', 'image/webp'];

    //validate file extension
    this.isFileAnImage(allowedExtensions, file.mimetype);

    if (!isFileSizeAcceptable) throw new MultiLanguageException(LanguageErrorEnum.LARGE_IMAGE);
    return file;
  }

  isFileAnImage(allowedExtensions: string[], uploadedImageExtension: string): boolean {
    const fileType = allowedExtensions.find((extension) => extension === uploadedImageExtension);
    if (!fileType) throw new MultiLanguageException(LanguageErrorEnum.UPLOAD_IMAGE_ONLY);
    return true;
  }
}
