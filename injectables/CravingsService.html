import { Inject, Injectable, forwardRef } from '@nestjs/common';
import { Craving, DistractionEnum, StressFactorEnum } from '@prisma/client';
import { plainToInstance } from 'class-transformer';
import { DateDto } from 'src/common/dto/date.dto';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';
import { PrismaService } from 'src/common/services/prisma.service';
import { PrismaTransactionClient } from 'src/common/types/transaction.type';
import { AddCravingDto } from 'src/modules/cravings/dto/add-carving.dto';
import { UpdateCravingDto } from 'src/modules/cravings/dto/update-carving.dto';
import { CravingResponse } from 'src/modules/cravings/response/craving.response';
import { IsCravingResponse } from 'src/modules/cravings/response/is-craving.response';
import { cravingSelect } from 'src/modules/cravings/const/craving-select';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';
import { PrivateChatsService } from 'src/modules/private-chats/private-chats.service';

@Injectable()
export class CravingsService {
  constructor(
    private prismaService: PrismaService,
    @Inject(forwardRef(() => PrivateChatsService))
    private privateChatsService: PrivateChatsService,
  ) {}

  async fetchCravings(patientId: number, dateDto: DateDto, isLastThreeRow: boolean): Promise<CravingResponse[]> {
    const { dateEndTime, dateStartTime } = dateDto;
    const take = isLastThreeRow ? 3 : undefined;

    const cravings = await this.prismaService.craving.findMany({
      where: {
        patientId,
        ...(!isLastThreeRow && {
          cravingDate: {
            gt: dateStartTime,
            lt: dateEndTime,
          },
        }),
      },
      select: cravingSelect,
      orderBy: { cravingDate: 'desc' },
      take,
    });

    return plainToInstance(CravingResponse, cravings, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });
  }

  async addCraving(user: UserPlayloadResponse, addCravingDto: AddCravingDto): Promise<CravingResponse> {
    const { distractions, stressFactors, ...rest } = addCravingDto;
    const isDistraction = distractions.length > 0;
    const reformatedDistractions = isDistraction
      ? distractions.map((distraction: DistractionEnum) => {
          return { distraction };
        })
      : [];

    const isStressFactor = stressFactors.length > 0;
    const reformatedStressFactors = isStressFactor
      ? stressFactors.map((stressFactor: StressFactorEnum) => {
          return { stressFactor };
        })
      : [];

    const addedCraving = await this.prismaService.craving.create({
      data: {
        ...rest,
        distractionOptions: {
          connect: reformatedDistractions,
        },
        stressFactorOptions: {
          connect: reformatedStressFactors,
        },
        patientId: user.patientId,
      },
      select: cravingSelect,
    });

    return plainToInstance(CravingResponse, addedCraving, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });
  }

  async fetchFirstCraving(user: UserPlayloadResponse): Promise<Craving> {
    return await this.prismaService.craving.findFirst({
      where: {
        patientId: user.patientId,
      },
    });
  }

  async updateCraving(user: UserPlayloadResponse, updateCravingDto: UpdateCravingDto, id: number): Promise<CravingResponse> {
    const { stressFactors, distractions, ...rest } = updateCravingDto;
    await this.isCravingOwner(user, id);

    const reformatedDistractions = distractions?.map((distraction: DistractionEnum) => {
      return { distraction };
    });

    const reformatedStressFactors = stressFactors?.map((stressFactor: StressFactorEnum) => {
      return { stressFactor };
    });

    const updatedCraving = await this.prismaService.craving.update({
      where: {
        id,
      },
      data: {
        ...(distractions && {
          distractionOptions: {
            set: reformatedDistractions,
          },
        }),
        ...(stressFactors && {
          stressFactorOptions: {
            set: reformatedStressFactors,
          },
        }),
        ...rest,
      },
      select: cravingSelect,
    });

    return plainToInstance(CravingResponse, updatedCraving, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });
  }

  async deleteCraving(user: UserPlayloadResponse, id: number): Promise<CravingResponse> {
    await this.isCravingOwner(user, id);

    const deletedCraving = await this.prismaService.craving.delete({
      where: {
        id,
      },
      select: cravingSelect,
    });

    return plainToInstance(CravingResponse, deletedCraving, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });
  }

  async fetchIsCraving(user: UserPlayloadResponse): Promise<IsCravingResponse> {
    const { isCraving } = await this.privateChatsService.privateChatByUserId(user.id);

    return { isCraving };
  }

  async fetchIsCravingIsById(id: number): Promise<IsCravingResponse> {
    const { isCraving } = await this.privateChatsService.fetchPrivateChatById(id);

    return { isCraving };
  }

  async addSosCraving(user: UserPlayloadResponse, transaction: PrismaTransactionClient): Promise<Craving> {
    const transactionOperation = transaction ?? this.prismaService;
    return await transactionOperation.craving.create({
      data: {
        cravingDate: new Date(),
        patientId: user.patientId,
      },
    });
  }

  async isCravingByIdExists(id: number): Promise<Craving> {
    const cravingByIdById = await this.prismaService.craving.findUnique({
      where: {
        id,
      },
    });

    if (!cravingByIdById) throw new MultiLanguageException(LanguageErrorEnum.CRAVING_NOT_FOUND);
    return cravingByIdById;
  }

  async isCravingOwner(user: UserPlayloadResponse, id: number): Promise<boolean> {
    const cravingById = await this.isCravingByIdExists(id);

    if (user.patientId !== Number(cravingById.patientId)) throw new MultiLanguageException(LanguageErrorEnum.CRAVING_NOT_YOURS);

    return true;
  }
}
