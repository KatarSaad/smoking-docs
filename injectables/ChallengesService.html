import { Inject, Injectable, forwardRef } from '@nestjs/common';
import { ChallengeEnum, DailyProgressTypeEnum } from '@prisma/client';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';
import { PrismaService } from 'src/common/services/prisma.service';
import { ChallengeWithProgress } from 'src/modules/challenges/response/challenge-with-progress.response';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';
import { ChallengePointsEnum } from './enums/challenges.points.enum';
import { UserChallengeDairyResponse } from './response/user.challenge.response';
import { DateDto } from 'src/common/dto/date.dto';
import { TodayProgressesService } from 'src/modules/today-progresses/today-progresses.service';
import { plainToInstance } from 'class-transformer';

@Injectable()
export class ChallengesService {
  constructor(
    private prismaService: PrismaService,
    @Inject(forwardRef(() => TodayProgressesService))
    private todayProgressesService: TodayProgressesService,
  ) {}

  async myChallenges(user: UserPlayloadResponse): Promise<ChallengeWithProgress> {
    // Step 1: Fetch all challenges for the user
    const challenges = await this.prismaService.userChallenge.findMany({
      where: { patientId: user.patientId },
      select: {
        isDone: true,
        id: true,
        patientId: true,
        type: true,
        startDate: true,
        points: true,
      },
    });

    await this.todayProgressesService.addCheckToTodaysProgress(user.id, DailyProgressTypeEnum.CHALLENGE);

    // Step 2: Group and sort challenges by startDate for each challenge type
    const groupedChallenges = challenges.reduce((acc, challenge) => {
      if (!acc[challenge.type]) {
        acc[challenge.type] = [];
      }
      acc[challenge.type].push(challenge);
      return acc;
    }, {});

    // Sort each group by startDate
    Object.keys(groupedChallenges).forEach((type) => {
      groupedChallenges[type].sort((a, b) => new Date(b.startDate).getTime() - new Date(a.startDate).getTime());
    });

    // Step 3: Merge missing challenge types with a default object
    const allChallengeTypes = Object.values(ChallengeEnum).map((type) => ({ type, challenges: {} }));
    const mergedChallenges = allChallengeTypes.map(({ type }) => {
      const existingChallenges = groupedChallenges[type] ? groupedChallenges[type][0] || {} : {};
      return { type, challenge: existingChallenges };
    });

    // Prepare the final result
    const data = mergedChallenges;

    // Calculate progress (assuming you have a way to calculate this)

    const totalChallenges = data.length;
    const completedChallenges = data.reduce((acc, item) => {
      // Access the challenge object correctly
      const challenge = item.challenge;
      // Check if the challenge object exists and isDone is true
      if (challenge && challenge.isDone) {
        return acc + 1;
      }
      return acc;
    }, 0);

    const challengesProgress = (completedChallenges / totalChallenges) * 100;
    return { data, challengesProgress };
  }

  async isChallengeByIdExists(id: number): Promise<boolean> {
    const challenge = await this.prismaService.userChallenge.findUnique({ where: { id }, select: { id: true } });

    if (!challenge) throw new MultiLanguageException(LanguageErrorEnum.CHALLENGE_NOT_FOUND);

    return true;
  }
  async isChallengeByNameExists(type: ChallengeEnum): Promise<boolean> {
    const challenge = await this.prismaService.userChallenge.findFirst({ where: { type }, select: { id: true, isDone: true } });

    if (challenge && challenge.isDone) throw new MultiLanguageException(LanguageErrorEnum.CHALLENGE_ALLREADY_DONE);

    return true;
  }
  async createChallenge(challengeType: ChallengeEnum, user: UserPlayloadResponse): Promise<boolean> {
    await this.prismaService.userChallenge.create({
      data: {
        type: challengeType,
        patient: { connect: { id: user.patientId } },
        isDone: false,
        // Add other necessary fields here
        points: ChallengePointsEnum[challengeType],
      },
    });
    return true;
  }

  async markChallengeAsCompleted(challengeId: number): Promise<boolean> {
    await this.isChallengeByIdExists(challengeId);
    await this.prismaService.userChallenge.update({
      where: { id: challengeId },
      data: {
        isDone: true,
        endDate: new Date(),
      },
    });
    return true;
  }
  async seedChallengesForPatient(id: bigint): Promise<boolean> {
    const user = await this.prismaService.user.findUnique({
      where: {
        id: id,
      },
      select: { patient: true },
    });
    for (const challengeType of Object.values(ChallengeEnum)) {
      await this.prismaService.userChallenge.create({
        data: {
          type: challengeType,
          patient: { connect: { id: user.patient.id } },
          isDone: false,
          // Add other necessary fields here
          points: ChallengePointsEnum[challengeType],
        },
      });
    }
    return true;
  }
  async getCommpletedChallenges(patientId: number, dateDto: DateDto, isLastThreeRow: boolean): Promise<UserChallengeDairyResponse[]> {
    const { dateEndTime, dateStartTime } = dateDto;
    const take = isLastThreeRow ? 3 : undefined;

    const challenges = await this.prismaService.userChallenge.findMany({
      where: {
        patientId,
        ...(!isLastThreeRow && {
          endDate: {
            //recheck it later
            gt: dateStartTime,
            lt: dateEndTime,
          },
        }),
      },
      take,
      select: { endDate: true, points: true, type: true },
    });

    return plainToInstance(UserChallengeDairyResponse, challenges, {
      excludeExtraneousValues: false,
      enableImplicitConversion: true,
    });
  }
}
