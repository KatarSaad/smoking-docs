import { Injectable } from '@nestjs/common';
import { Faq, LanguageEnum, Prisma } from '@prisma/client';
import { plainToInstance } from 'class-transformer';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';
import { CommonService } from 'src/common/services/common.service';
import { PrismaService } from 'src/common/services/prisma.service';
import { FaqCategoriesService } from 'src/modules/faq-categories/faq-categories.service';
import { AddDraftFaqDto } from 'src/modules/faqs/dto/add-faq-draft.dto';
import { AddFaqDto } from 'src/modules/faqs/dto/add-faq.dto';
import { FetchFaqsWithCategoriesDto } from 'src/modules/faqs/dto/fetch-faqs-with-categories.dto copy';
import { FetchFaqsDto } from 'src/modules/faqs/dto/fetch-faqs.dto';
import { FaqWithCategoryResponse } from 'src/modules/faqs/response/faq-with-category.response';
import { FaqWithMultiLanguage } from 'src/modules/faqs/response/faq-with-multi-language.response';
import { FaqResponse } from 'src/modules/faqs/response/faq.response';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';

@Injectable()
export class FaqsService {
  constructor(
    private prismaService: PrismaService,
    private faqCategoriesService: FaqCategoriesService,
    private commonService: CommonService,
  ) {}

  async fetchFaqs(language: LanguageEnum, fetchFaqsDto: FetchFaqsDto): Promise<FaqResponse[]> {
    const { faqCategoryId, search } = fetchFaqsDto;
    const fullTextSearch = search ? this.commonService.addLeadingPlusAndTrailingWildcard(search) : undefined;

    const where: Prisma.FaqWhereInput = {
      faqCategoryId,
      isPublished: true,

      faqsTranslation: {
        some: {
          question: {
            search: fullTextSearch,
          },
        },
      },
    };

    const faqs = await this.prismaService.faq.findMany({
      where,
      select: this.getFaqType(language),
      orderBy: {
        updatedAt: 'desc',
      },
    });

    const serialized = plainToInstance(FaqResponse, faqs, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    return serialized;
  }

  async createFaq(language: LanguageEnum, user: UserPlayloadResponse, addFaqDto: AddFaqDto): Promise<FaqResponse> {
    const { answerAr, answerEn, questionAr, questionEn, categoryId } = addFaqDto;

    //check if category id exists
    await this.faqCategoriesService.isCategoryIdExists(categoryId);

    const faq = await this.prismaService.faq.create({
      data: {
        publisher: {
          connect: {
            id: user.adminId,
          },
        },
        faqCategory: {
          connect: {
            id: categoryId,
          },
        },
        faqsTranslation: {
          create: [
            {
              question: questionEn,
              answer: answerEn,
              language: {
                connect: {
                  code: LanguageEnum.ENGLISH,
                },
              },
            },
            {
              question: questionAr,
              answer: answerAr,
              language: {
                connect: {
                  code: LanguageEnum.ARABIC,
                },
              },
            },
          ],
        },
      },
      select: this.getFaqType(language),
    });

    const serialized = plainToInstance(FaqResponse, faq, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    return serialized;
  }

  async fetchfaqsByCategory(language: LanguageEnum, fetchFaqsWithCategoriesDto: FetchFaqsWithCategoriesDto): Promise<FaqWithCategoryResponse[]> {
    const { search } = fetchFaqsWithCategoriesDto;
    const fullTextSearch = search ? this.commonService.addLeadingPlusAndTrailingWildcard(search) : undefined;

    const faqsWithCategories = await this.prismaService.faqCategory.findMany({
      select: {
        id: true,
        faqCategoriesTranslation: {
          where: {
            language: {
              code: language,
            },
          },
          select: {
            name: true,
          },
        },
        faqs: {
          where: {
            isPublished: true,
          },
          // take: 4,
          select: {
            id: true,
            faqsTranslation: {
              where: {
                question: {
                  search: fullTextSearch,
                },
                language: {
                  code: language,
                },
              },
              select: {
                answer: true,
                question: true,
              },
            },
          },
        },
      },
    });

    const serialized = plainToInstance(FaqWithCategoryResponse, faqsWithCategories, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    return serialized;
  }

  async fetchDraftFaqs(language: LanguageEnum, user: UserPlayloadResponse, fetchFaqsDto: FetchFaqsDto): Promise<FaqResponse[]> {
    const { faqCategoryId, search } = fetchFaqsDto;
    const fullTextSearch = search ? this.commonService.addLeadingPlusAndTrailingWildcard(search) : undefined;

    const where: Prisma.FaqWhereInput = {
      faqCategoryId,
      isPublished: false,
      publisherId: user.adminId,
      faqsTranslation: {
        some: {
          question: {
            search: fullTextSearch,
          },
        },
      },
    };

    const faqs = await this.prismaService.faq.findMany({
      where,
      select: this.getFaqType(language),
      orderBy: {
        updatedAt: 'desc',
      },
    });

    const serialized = plainToInstance(FaqResponse, faqs, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    return serialized;
  }

  async createDraftFaq(language: LanguageEnum, user: UserPlayloadResponse, addDraftFaqDto: AddDraftFaqDto): Promise<FaqResponse> {
    const { answerAr, answerEn, questionAr, questionEn, categoryId } = addDraftFaqDto;

    //check if category id exists
    await this.faqCategoriesService.isCategoryIdExists(categoryId);

    const faq = await this.prismaService.faq.create({
      data: {
        isPublished: false,
        publisher: {
          connect: {
            id: user.adminId,
          },
        },
        ...(categoryId && {
          faqCategory: {
            connect: {
              id: categoryId,
            },
          },
        }),
        faqsTranslation: {
          create: [
            {
              question: questionEn,
              answer: answerEn,
              language: {
                connect: {
                  code: LanguageEnum.ENGLISH,
                },
              },
            },
            {
              question: questionAr,
              answer: answerAr,
              language: {
                connect: {
                  code: LanguageEnum.ARABIC,
                },
              },
            },
          ],
        },
      },
      select: this.getFaqType(language),
    });

    const serialized = plainToInstance(FaqResponse, faq, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    return serialized;
  }

  async updateFaq(language: LanguageEnum, user: UserPlayloadResponse, addFaqDto: AddFaqDto, id: number): Promise<FaqResponse> {
    const { answerAr, answerEn, questionAr, questionEn, categoryId } = addFaqDto;

    //check if faq id exists
    await this.isFaqOwnedByAuthedUser(id, user.adminId);

    //check if category id exists
    if (categoryId) await this.faqCategoriesService.isCategoryIdExists(categoryId);

    const faq = await this.prismaService.faq.update({
      where: { id },
      data: {
        ...(categoryId && {
          faqCategory: {
            connect: {
              id: categoryId,
            },
          },
        }),
        faqsTranslation: {
          updateMany: [
            {
              where: {
                faqId: id,
                languageId: 1,
              },
              data: {
                question: questionEn,
                answer: answerEn,
              },
            },
            {
              where: {
                faqId: id,
                languageId: 2,
              },
              data: {
                question: questionAr,
                answer: answerAr,
              },
            },
          ],
        },
      },
      select: this.getFaqType(language),
    });

    const serialized = plainToInstance(FaqResponse, faq, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    return serialized;
  }

  async deleteFaq(language: LanguageEnum, user: UserPlayloadResponse, id: number): Promise<FaqResponse> {
    //check if faq id exists
    await this.isFaqOwnedByAuthedUser(id, user.adminId);

    const faq = await this.prismaService.faq.delete({
      where: { id },
      select: this.getFaqType(language),
    });

    const serialized = plainToInstance(FaqResponse, faq, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    return serialized;
  }

  async fetchFaqsWithMultiLanguage(fetchFaqsDto: FetchFaqsDto): Promise<FaqWithMultiLanguage[]> {
    const { search, faqCategoryId } = fetchFaqsDto;
    const fullTextSearch = search ? this.commonService.addLeadingPlusAndTrailingWildcard(search) : undefined;

    const faqsWithMultiLanguage = await this.prismaService.faq.findMany({
      where: {
        isPublished: true,
        faqCategoryId: faqCategoryId,
        faqsTranslation: {
          some: {
            question: {
              search: fullTextSearch,
            },
          },
        },
      },
      select: {
        id: true,
        isPublished: true,
        faqCategory: {
          select: {
            id: true,
            faqCategoriesTranslation: {
              select: {
                name: true,
                language: {
                  select: {
                    id: true,
                    code: true,
                    name: true,
                  },
                },
              },
            },
          },
        },
        faqsTranslation: {
          select: {
            question: true,
            answer: true,
            language: {
              select: {
                id: true,
                code: true,
                name: true,
              },
            },
          },
        },
      },
    });

    return faqsWithMultiLanguage;
  }

  async publishFaq(language: LanguageEnum, id: number): Promise<FaqResponse> {
    await this.isFaqIdExists(id);
    const publishFaq = await this.prismaService.faq.update({
      where: { id },
      data: {
        isPublished: true,
      },
      select: this.getFaqType(language),
    });

    const serializedPublishedFaq = plainToInstance(FaqResponse, publishFaq, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    return serializedPublishedFaq;
  }

  async unpublishFaq(language: LanguageEnum, id: number): Promise<FaqResponse> {
    await this.isFaqIdExists(id);
    const unpublishFaq = await this.prismaService.faq.update({
      where: { id },
      data: {
        isPublished: false,
      },
      select: this.getFaqType(language),
    });

    const serializedUnpublishedFaq = plainToInstance(FaqResponse, unpublishFaq, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    return serializedUnpublishedFaq;
  }

  async updateDraftFaq(language: LanguageEnum, user: UserPlayloadResponse, addDraftFaqDto: AddDraftFaqDto, id: number): Promise<FaqResponse> {
    const { answerAr, answerEn, questionAr, questionEn, categoryId } = addDraftFaqDto;

    //check if faq id exists
    await this.isFaqOwnedByAuthedUser(id, user.adminId);

    //check if category id exists
    if (categoryId) await this.faqCategoriesService.isCategoryIdExists(categoryId);

    const faq = await this.prismaService.faq.update({
      where: { id },
      data: {
        faqCategoryId: categoryId,

        faqsTranslation: {
          update: [
            {
              where: {
                faqId_languageId: {
                  faqId: id,
                  languageId: 1,
                },
              },
              data: {
                question: questionEn,
                answer: answerEn,
              },
            },
            {
              where: {
                faqId_languageId: {
                  faqId: id,
                  languageId: 2,
                },
              },
              data: {
                question: questionAr,
                answer: answerAr,
              },
            },
          ],
        },
      },
      select: this.getFaqType(language),
    });

    const serialized = plainToInstance(FaqResponse, faq, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    return serialized;
  }

  async isFaqIdExists(id: number): Promise<Faq> {
    const isArticlelabelExists = await this.prismaService.faq.findFirst({
      where: {
        id,
      },
    });
    if (!isArticlelabelExists) throw new MultiLanguageException(LanguageErrorEnum.FAQS_CATEGORY_NOT_FOUND);
    return isArticlelabelExists;
  }

  async isFaqOwnedByAuthedUser(faqId: number, authedAdminId: number): Promise<Faq> {
    const faqById = await this.isFaqIdExists(faqId);

    if (faqById.publisherId !== BigInt(authedAdminId)) throw new MultiLanguageException(LanguageErrorEnum.FAQ_NOT_YOURS);
    return faqById;
  }

  getFaqType(language: LanguageEnum): Prisma.FaqSelect {
    return {
      id: true,
      isPublished: true,
      updatedAt: true,
      faqCategory: {
        select: {
          id: true,
          faqCategoriesTranslation: {
            where: {
              language: { code: language },
            },
            select: {
              name: true,
            },
          },
        },
      },
      faqsTranslation: {
        where: {
          language: {
            code: language,
          },
        },
        select: {
          question: true,
          answer: true,
        },
      },
    };
  }
}
