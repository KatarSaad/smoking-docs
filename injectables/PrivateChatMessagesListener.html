import { Server, Socket } from 'socket.io';
import { UserPlayloadResponse } from '../patients/response/user-playload.response';
import { Injectable } from '@nestjs/common';
import { CommonService } from 'src/common/services/common.service';
import { PrivateChatMessagesService } from 'src/modules/private-chat-messages/private-chat-messages.service';
import { PrivateChatsService } from 'src/modules/private-chats/private-chats.service';
import { JoinHealthSpecialistPrivateChatRoomDto } from 'src/modules/private-chat-messages/dto/health-specialist-join-private-chat-room.dto';
import { ParticipantJoinPrivateChatRoomDto } from 'src/modules/private-chat-messages/dto/participant-join-private-chat-room.dto';
import { DeleteMessageDto } from 'src/modules/private-chat-messages/dto/delete-message.dto';
import { SuccessResponse } from 'src/common/responses/socket.response';
import { PrivateChatMessageSocketResponse } from 'src/modules/private-chat-messages/response/private-chat-message-socket.response';
import { SocketEventEnum } from 'src/common/enums/socket-event.enum';
import { SocketMessagePrivateDto } from 'src/modules/private-chat-messages/dto/socket-message-private.dto';
import { SettingOptionNameEnum } from '@prisma/client';

@Injectable()
export class PrivateChatMessagesListener {
  constructor(
    private privateChatMessagesService: PrivateChatMessagesService,
    private commonService: CommonService,
    private privateChatsService: PrivateChatsService,
  ) {}

  async sendPrivateChatMessage(payload: SocketMessagePrivateDto, client: Socket, user: UserPlayloadResponse, server: Server): Promise<PrivateChatMessageSocketResponse> {
    const { privateChatId, message } = payload;

    const room = String(privateChatId);

    const addMessageResponse = await this.privateChatMessagesService.addMessageToPrivateChat(user, { ...payload, eventType: SocketEventEnum.MESSAGE_PRIVATE_CHAT }); //save message in db

    server.to(`private_chats_${room}`).emit(SocketEventEnum.MESSAGE_PRIVATE_CHATS, this.commonService.objectParser({ privateChatId, message, senderId: user.id }));

    // broadcast message to sockets connected to the group room(except the sender)
    client.broadcast.to(`private_chat_${room}`).emit(SocketEventEnum.MESSAGE_PRIVATE_CHAT, this.commonService.objectParser(addMessageResponse));

    //sending notif
    const senderName = `${addMessageResponse.sender.personalInformation.firstName} ${addMessageResponse.sender.personalInformation.lastName}`;
    this.privateChatMessagesService.privateChatNotification(
      user,
      addMessageResponse.receiverId,
      privateChatId,
      SettingOptionNameEnum.ALL_MESSAGES_IN_PRIVATE_CHAT,
      senderName,
    );
    return { success: true, messageId: Number(addMessageResponse.id) };
  }

  async userJoinPrivateChatRoom(payload: ParticipantJoinPrivateChatRoomDto, user: UserPlayloadResponse, server: Server): Promise<SuccessResponse> {
    const { socketId, privateChatId } = payload;
    const roomName = String(privateChatId);

    await this.privateChatsService.isPatientInPrivateChat(user.id, privateChatId, null);

    server.in(socketId).socketsJoin(`private_chat_${roomName}`);

    return { success: true };
  }

  async healthSpecialistJoinPrivateChatRoom(payload: ParticipantJoinPrivateChatRoomDto, user: UserPlayloadResponse, server: Server): Promise<SuccessResponse> {
    const { socketId, privateChatId } = payload;

    await this.privateChatsService.isHealthSpecialistInPrivateChat(privateChatId, user.id);

    server.in(socketId).socketsJoin(`private_chat_${privateChatId}`);

    return { success: true };
  }

  async healthSpecialistJoinPrivateChatRooms(payload: JoinHealthSpecialistPrivateChatRoomDto, user: UserPlayloadResponse, server: Server): Promise<SuccessResponse> {
    const { socketId, privateChatIds } = payload;
    const privateChatIdsStrings = privateChatIds.map((id) => `private_chats_${id}`);

    await this.privateChatsService.isHealthSpecialistInPrivateChats(privateChatIds, user.id);
    server.in(socketId).socketsJoin(privateChatIdsStrings);

    return { success: true };
  }

  async deletePrivateChatMessage(deleteMessageDto: DeleteMessageDto, client: Socket, user: UserPlayloadResponse): Promise<PrivateChatMessageSocketResponse> {
    const { id } = deleteMessageDto;

    const deletedMessage = await this.privateChatMessagesService.deletePrivateMessage(user, id);

    const deletedMessageId = Number(deletedMessage.id);

    client.broadcast.timeout(5000).emit(SocketEventEnum.DELETE_MESSAGE_PRIVATE_CHAT, { messageId: deletedMessageId });

    return { success: true, messageId: deletedMessageId };
  }
}
