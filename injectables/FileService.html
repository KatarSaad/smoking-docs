import { Injectable } from '@nestjs/common';
import * as sharp from 'sharp';
import * as path from 'path';
import { EnvironmentService } from './environment.service';
import { FileResponse } from 'src/modules/private-chat-messages/response/file.response';
import { BlobServiceClient, StorageSharedKeyCredential } from '@azure/storage-blob';
import * as fs from 'fs-extra';

@Injectable()
export class FileService {
  constructor(private environmentService: EnvironmentService) {}

  private accountName = this.environmentService.azureBlobConfig.blobAccountName;
  private accountKey = this.environmentService.azureBlobConfig.blobKey;
  private containerName = this.environmentService.azureBlobConfig.blobContainerName;

  private sharedKeyCredential = new StorageSharedKeyCredential(this.accountName, this.accountKey);
  private blobServiceClient = new BlobServiceClient(`https://${this.accountName}.blob.core.windows.net`, this.sharedKeyCredential);
  test = 'test';

  private containerClient = this.blobServiceClient.getContainerClient(this.containerName);

  async uploadImage(file: Express.Multer.File): Promise<string> {
    const blobName = this.imageName(path.parse(file.originalname).name);

    const blockBlobClient = this.containerClient.getBlockBlobClient(blobName);

    const compressedImage = await this.optimizingImage(file.buffer);

    await blockBlobClient.upload(compressedImage.data, compressedImage.info.size);

    return blobName;
  }

  async uploadFile(file: Express.Multer.File): Promise<FileResponse> {
    const blobName = this.fileName(file.originalname);
    const { ext } = path.parse(file.originalname);
    const blockBlobClient = this.containerClient.getBlockBlobClient(blobName);

    await blockBlobClient.upload(file.buffer, file.buffer.length);

    return { blobName, name: file.originalname, extension: ext };
  }

  deleteFile(fileName: string): boolean {
    const blockBlobClient = this.containerClient.getBlockBlobClient(fileName);

    blockBlobClient.deleteIfExists();

    return true;
  }

  imageName(originalName: string): string {
    //3 random digits for optimizing the fetch performance, read the below docs
    //https://learn.microsoft.com/en-us/azure/storage/blobs/storage-performance-checklist#partitioning
    const randomThreeDigits = Math.floor(Math.random() * (999 - 100 + 1) + 100);

    const imageName = randomThreeDigits.toString() + '-' + Date.now() + '-' + originalName + '.webp';
    if (imageName.length > 50) {
      const extractImagesPart = imageName.substring(0, 50);
      return randomThreeDigits.toString() + '-' + Date.now() + '-' + extractImagesPart + '.webp';
    }
    return imageName;
  }

  fileName(originalName: string): string {
    //3 random digits for optimizing the fetch performance, read the below docs
    //https://learn.microsoft.com/en-us/azure/storage/blobs/storage-performance-checklist#partitioning
    const randomThreeDigits = Math.floor(Math.random() * (999 - 100 + 1) + 100);

    const fileName = randomThreeDigits.toString() + '-' + Date.now() + '-' + originalName;

    if (fileName.length > 50) {
      const extractImagesPart = fileName.substring(0, 50);
      return randomThreeDigits.toString() + '-' + Date.now() + '-' + extractImagesPart;
    }
    return fileName;
  }

  async optimizingImage(buffer: Buffer): Promise<{
    data: Buffer;
    info: sharp.OutputInfo;
  }> {
    const compressedImage = await sharp(buffer).webp({ effort: 5, quality: 100 }).toBuffer({ resolveWithObject: true }); //resize(800) work parallel even if u specify width only
    return compressedImage;
  }
  async uploadImages(files: Express.Multer.File[]): Promise<string[]> {
    if (!files) return [];
    const uploadPromises = files.map(async (file) => {
      const blobName = this.imageName(path.parse(file.originalname).name);
      const blockBlobClient = this.containerClient.getBlockBlobClient(blobName);

      const compressedImage = await this.optimizingImage(file.buffer);

      await blockBlobClient.upload(compressedImage.data, compressedImage.info.size);

      return blobName;
    });

    const uploadedFileNames = await Promise.all(uploadPromises);
    return uploadedFileNames;
  }
  readFilesAsMulterFiles = async (filePaths: string[]): Promise<Express.Multer.File[]> => {
    const files: Express.Multer.File[] = [];
    for (const filePath of filePaths) {
      const fileBuffer = await fs.readFile(filePath);
      const fileName = path.basename(filePath);
      files.push({
        fieldname: 'file',
        originalname: fileName,
        encoding: '7bit',
        mimetype: 'image/png',
        size: fileBuffer.length,
        buffer: fileBuffer,
        destination: '',
        filename: fileName,
        path: filePath,
        stream: fs.createReadStream(filePath),
      } as Express.Multer.File);
    }
    return files;
  };
}
