import { Injectable } from '@nestjs/common';
import { Symptom, SymptomEnum } from '@prisma/client';
import { plainToInstance } from 'class-transformer';
import { DateDto } from 'src/common/dto/date.dto';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';
import { PrismaService } from 'src/common/services/prisma.service';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';
import { symptomSelect } from 'src/modules/symptoms/const/symptom-select';
import { AddSymptomDto } from 'src/modules/symptoms/dto/add-symptom.dto';
import { UpdateSymptomDto } from 'src/modules/symptoms/dto/update-symptom.dto';
import { SymptomResponse } from 'src/modules/symptoms/response/symptom.response';

@Injectable()
export class SymptomsService {
  constructor(private prismaService: PrismaService) {}

  async fetchSymptoms(patientId: number, dateDto: DateDto, isLastThreeRow: boolean): Promise<SymptomResponse[]> {
    const { dateEndTime, dateStartTime } = dateDto;
    const take = isLastThreeRow ? 3 : undefined;

    const symptoms = await this.prismaService.symptom.findMany({
      where: {
        ...(!isLastThreeRow && {
          createdAt: {
            gt: dateStartTime,
            lt: dateEndTime,
          },
        }),
        patientId,
      },
      select: symptomSelect,
      orderBy: {
        createdAt: 'desc',
      },
      take,
    });

    return plainToInstance(SymptomResponse, symptoms, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });
  }

  async addSymptom(user: UserPlayloadResponse, addSymptomDto: AddSymptomDto): Promise<SymptomResponse> {
    const { comment, other, symptoms } = addSymptomDto;
    const isSymptoms = symptoms.length > 0;
    const reformatedSymptoms = isSymptoms
      ? symptoms.map((symptom: SymptomEnum) => {
          return { symptom };
        })
      : [];

    const addedSymptom = await this.prismaService.symptom.create({
      data: {
        comment,
        other,
        symptomOptions: {
          connect: reformatedSymptoms,
        },
        patientId: user.patientId,
      },
      select: symptomSelect,
    });

    return plainToInstance(SymptomResponse, addedSymptom, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });
  }

  async fetchFirstSymptom(user: UserPlayloadResponse): Promise<Symptom> {
    return await this.prismaService.symptom.findFirst({
      where: {
        patientId: user.patientId,
      },
    });
  }

  async updateSymptom(user: UserPlayloadResponse, updateSymptomDto: UpdateSymptomDto, id: number): Promise<SymptomResponse> {
    const { comment, other, symptoms } = updateSymptomDto;

    const reformatedSymptoms = symptoms?.map((symptom) => {
      return { symptom };
    });

    await this.isSymptomOwner(user, id);

    const updatedSymptom = await this.prismaService.symptom.update({
      where: { id },
      data: {
        comment,
        other,
        ...(symptoms && {
          symptomOptions: {
            set: reformatedSymptoms,
          },
        }),
      },
      select: symptomSelect,
    });

    return plainToInstance(SymptomResponse, updatedSymptom, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });
  }

  async deleteSymptom(user: UserPlayloadResponse, id: number): Promise<SymptomResponse> {
    await this.isSymptomOwner(user, id);

    const deletedSymptom = await this.prismaService.symptom.delete({
      where: { id },
      select: symptomSelect,
    });

    return plainToInstance(SymptomResponse, deletedSymptom, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });
  }

  async isSymptomByIdExists(id: number): Promise<Symptom> {
    const moodById = await this.prismaService.symptom.findUnique({
      where: {
        id,
      },
    });

    if (!moodById) throw new MultiLanguageException(LanguageErrorEnum.MOOD_NOT_FOUND);
    return moodById;
  }

  async isSymptomOwner(user: UserPlayloadResponse, id: number): Promise<boolean> {
    const moodById = await this.isSymptomByIdExists(id);

    if (user.patientId !== Number(moodById.patientId)) throw new MultiLanguageException(LanguageErrorEnum.MOOD_NOT_YOURS);

    return true;
  }
}
