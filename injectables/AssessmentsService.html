@Injectable()

  constructor(
    private prismaService: PrismaService,

  async addAssessment(assessmentDto: AddAssessmentDto, user: UserPlayloadResponse): Promise<FetchAssessmentResponse> {
    const { name } = assessmentDto;

    return await this.upsertAssessment(user, name);
  }

  async fetchAssessments(user: UserPlayloadResponse): Promise<FetchAllAssessmentResponse[]> {
    const assessmentUser = await this.prismaService.assessment.findMany({
      where: {
        patientId: user.patientId,
      },
      select: {
        id: true,
        fagerstromScore: true,
        name: true,
        createdAt: true,
        habitInformation: {
          select: {
            numberCigaretesPerDay: true,
            numberTriesQuittingSmoking: true,
            numberYearsQuittingSmoking: true,
            periodeSmoking: true,
            quitDate: true,
            quitSmoking: true,
          },
        },
        tobacco: { select: tobaccoSelect },
        allergy: { select: allergySelect },
        riskFactor: { select: riskFactorSelect },
        smokingTrigger: { select: smokingTriggerSelect },
        motivation: {
          select: motivationSelect,
        },
      },
    });

    return assessmentUser;
  }

  async fetchCurrentAssessmentDetail(user: UserPlayloadResponse): Promise<CurrentAssessmentResponse> {
    let durationForNextAssessment = null;
    const assessmentDetails = await this.fetchUserAssessmentsByUserId(user);
    const { nextAssessment, welcomeAssessmentAt, finalAssesmentDone } = assessmentDetails;

    //Extract the duration of the next assessments from assessments json
    const monthsNumberAfterWelcomeAssessment = nextAssessment && JSON_ASSESSMENTS[nextAssessment].MonthsAfterWelcomeAssessment;

    // Transform the duration to milli sec
    const actualDurationForNextAssessment = monthsNumberAfterWelcomeAssessment && monthsNumberAfterWelcomeAssessment * 30 * 24 * 60 * 60 * 1000;

    //Duration for next assessment it will be welcomeAssessmentAt
    // if user assessments got reseted we will have to use lastAssessmentResetAt
    const assessmentLastRestDate = await this.assessmentLastRestDate(user.patientId);

    durationForNextAssessment = this.commonService.remainingTime(assessmentLastRestDate);
    if (!assessmentLastRestDate) durationForNextAssessment = this.commonService.remainingTime(welcomeAssessmentAt);
    //fetching the current assessment details
    const assessmentSection = await this.getAssessmentSection(nextAssessment);

    // if the time of the next assessment has came we send it in the current assessment
    if (durationForNextAssessment > actualDurationForNextAssessment) {
      const { id } = await this.upsertAssessment(user, nextAssessment);
      // current assessment time has came
      return { ...assessmentDetails, currentAssessment: finalAssesmentDone ? null : nextAssessment, currentAssessmentId: Number(id), assessmentSection };
    }

    return { ...assessmentDetails, currentAssessment: null, currentAssessmentId: null, assessmentSection: null };
  }

  async fetchAssessment(id: number, user: UserPlayloadResponse): Promise<FetchAssessmentResponse> {
    await this.isAssessmentOwner(user, id);

    const assessmentUser = await this.prismaService.assessment.findFirst({
      where: {
        id,
      },
      select: {
        id: true,
        fagerstromScore: true,
        name: true,
        habitInformation: {
          select: {
            numberCigaretesPerDay: true,
            numberTriesQuittingSmoking: true,
            numberYearsQuittingSmoking: true,
            periodeSmoking: true,
            quitDate: true,
            quitSmoking: true,
          },
        },
        tobacco: { select: tobaccoSelect },
        allergy: { select: allergySelect },
        riskFactor: { select: riskFactorSelect },
        smokingTrigger: { select: smokingTriggerSelect },
        motivation: {
          select: motivationSelect,
        },
        createdAt: true,
      },
    });
    return assessmentUser;
  }

  async upsertAssessment(user: UserPlayloadResponse, name: AssessmentNameEnum): Promise<FetchAssessmentResponse> {
    const assessmentUser = await this.fetchCurrentAssessment(user, name);

    if (assessmentUser) return assessmentUser;

    return await this.prismaService.assessment.create({
      data: {
        patient: {
          connect: {
            id: user.patientId,
          },
        },
        name,
      },
      select: {
        id: true,
        name: true,
        fagerstromScore: true,
        createdAt: true,
        habitInformation: true,
        tobacco: { select: tobaccoSelect },
        allergy: { select: allergySelect },
        riskFactor: { select: riskFactorSelect },
        smokingTrigger: { select: smokingTriggerSelect },
        motivation: {
          select: motivationSelect,
        },
      },
    });
  }

  async updatePersonalInformation(
    createPersonalInfoAssessmentDto: CreatePersonalInfoAssessmentDto,
    id: number,
    user: UserPlayloadResponse,
  ): Promise<PersonalInformationResponse> {
    await this.findAssessmentUser(id, user);

    return this.prismaService.$transaction(async (tx) => {
      await this.updateStep(id, personalInfoStep, tx);

      return await this.onboardingService.updatePersonalInfo(user, createPersonalInfoAssessmentDto, tx);
    });
  }

  async updateHabit(createHabitInfoAssessmentDto: CreateHabitInfoAssessmentDto, id: number, user: UserPlayloadResponse): Promise<HabitInformationResponse> {
    const assessmentUser = await this.findAssessmentUser(id, user);
    console.log('dsvc');
    return this.prismaService.$transaction(async (tx) => {
      await tx.assessment.update({
        where: {
          id: assessmentUser.id,
        },
        data: {
          habitInformation: {
            upsert: {
              where: {
                id: assessmentUser.habitInformation?.id,
              },
              update: {
                ...createHabitInfoAssessmentDto,
              },
              create: {
                ...createHabitInfoAssessmentDto,
              },
            },
          },
        },
        include: {
          habitInformation: true,
        },
      });
      await this.updateStep(id, habitInfoStep, tx);
      return await this.onboardingService.updateHabitInfo(user.patientId, createHabitInfoAssessmentDto);
    });
  }

  async updateFagerstromReport(IdsDto: IdDto[], id: number, user: UserPlayloadResponse): Promise<boolean> {
    const assessmentUser = await this.findAssessmentUser(id, user);

    const reportResult = this.fagerstromService.calculateFagerstromScore(IdsDto);
    const addictionLevel = reportResult > 7 ? AddictionLevelEnum.HIGH : AddictionLevelEnum.LOW;

    await this.prismaService.assessment.update({
      where: {
        id: assessmentUser.id,
      },
      data: {
        patient: {
          update: {
            addictionLevel,
          },
        },
        fagerstromScore: reportResult,
      },
    });

    this.groupsService.assignAuthedPatientToGroups(user); //this function :automatic join to groups that matches the user characteristics

    return true;
  }

  async findAssessmentUser(id: number, user: UserPlayloadResponse) {
    const userAssessment = await this.prismaService.assessment.findFirst({
      where: {
        id,
      },
      include: {
        habitInformation: {
          select: {
            id: true,
          },
        },
      },
    });

    if (!userAssessment) throw new MultiLanguageException(LanguageErrorEnum.ASSESSMENT_NOT_FOUND);
    if (Number(userAssessment.patientId) !== user.patientId) throw new MultiLanguageException(LanguageErrorEnum.ASSESSMENT_NOT_YOURS);

    return userAssessment;
  }

  assesmentDetail(assessments: Assessment[]): AssessmentDetailResponse {
    const finishedAssessments = assessments.filter((assessment) => assessment.isDone && !assessment.isReseted);
    // console.log('Assessment finished', finishedAssessments);
    const finalAssesmentDone = finishedAssessments.filter((assessment) => assessment.name === AssessmentNameEnum.FINAL);
    const assessmentsLength = finishedAssessments.length;
    const assessmentsName = Object.values(AssessmentNameEnum);
    // console.log(finalAssesmentDone);
    return {
      welcomeAssessmentAt: finishedAssessments[0] ? finishedAssessments[0].createdAt : null,
      nextAssessment: assessmentsName[assessmentsLength] ? assessmentsName[assessmentsLength] : AssessmentNameEnum.FINAL,
      step: null,
      finalAssesmentDone: finalAssesmentDone.length === 0 ? false : true,
    };
  }
  async fetchUserAssessmentsByUserId(user: UserPlayloadResponse): Promise<AssessmentDetailResponse> {
    const userAssessments = await this.prismaService.user.findUnique({
      where: {
        id: user.id,
      },
      include: {
        patient: {
          include: {
            assessments: {
              orderBy: {
                createdAt: 'asc',
              },
            },
          },
        },
      },
    });

    if (!userAssessments.patient) throw new MultiLanguageException(LanguageErrorEnum.PATIENTS_ONLY);

    const assessmentDetail = this.assesmentDetail(userAssessments.patient.assessments);
    // console.log('assessmentDetail', assessmentDetail);
    const fetchLastPatientAssessment = await this.fetchLastPatientAssessment(user.patientId);

    return { ...assessmentDetail, step: fetchLastPatientAssessment?.step ?? null };
  }

  async updateAssessmentAsDone(id: number, transaction?: PrismaTransactionClient): Promise<boolean> {
    const transactionOperation = transaction ?? this.prismaService;

    await transactionOperation.assessment.update({
      where: { id },
      data: {
        isDone: true,
      },
    });
    return true;
  }

  async getAssessmentSection(assessment: AssessmentNameEnum): Promise<AssessmentInfoResponse> {
    return assessment ? JSON_ASSESSMENTS[assessment] : null;
  }

  async assessmentLastRestDate(patientId: number): Promise<Date> {
    const { lastAssessmentResetAt } = await this.prismaService.patient.findUnique({ where: { id: patientId } });
    return lastAssessmentResetAt;
  }

  async fetchCurrentAssessment(user: UserPlayloadResponse, name: AssessmentNameEnum, transaction?: PrismaTransactionClient): Promise<FetchAssessmentResponse> {
    const prismaOperation = transaction ?? this.prismaService;

    return await prismaOperation.assessment.findFirst({
      where: {
        name,
        patientId: user.patientId,
        isReseted: false,
      },
      select: {
        id: true,
        name: true,
        fagerstromScore: true,
        createdAt: true,
        habitInformation: {
          select: {
            numberCigaretesPerDay: true,
            numberTriesQuittingSmoking: true,
            numberYearsQuittingSmoking: true,
            periodeSmoking: true,
            quitDate: true,
            quitSmoking: true,
          },
        },
        tobacco: { select: tobaccoSelect },
        allergy: { select: allergySelect },
        riskFactor: { select: riskFactorSelect },
        smokingTrigger: { select: smokingTriggerSelect },
        motivation: {
          select: motivationSelect,
        },
      },
    });
  }

  async resetAssessments(patientId: number): Promise<boolean> {
    const relapsesNumberToRestAssessments = 20;
    const assessmentLastRestDate = await this.assessmentLastRestDate(patientId);

    const countRelapses = await this.prismaService.relapse.count({
      where: {
        patientId,
        ...(assessmentLastRestDate && {
          createdAt: {
            gt: assessmentLastRestDate,
          },
        }),
      },
    });

    if (countRelapses >= relapsesNumberToRestAssessments) {
      await this.prismaService.$transaction([
        this.prismaService.assessment.updateMany({
          where: {
            name: {
              not: AssessmentNameEnum.WELCOME,
            },
            patientId,
          },
          data: {
            isReseted: true,
          },
        }),

        this.prismaService.patient.update({
          where: { id: patientId },
          data: {
            lastAssessmentResetAt: new Date(),
          },
        }),
      ]);
    }

    return true;
  }

  async fetchLastPatientAssessment(patientId: number): Promise<Assessment> {
    const assessments = await this.prismaService.assessment.findMany({
      where: {
        patientId,
        isReseted: false,
      },
      orderBy: {
        createdAt: 'desc',
      },
      take: 1,
    });
    return assessments[0];
  }

  async fetchStep(assesmentId: number, transaction?: PrismaTransactionClient): Promise<Assessment> {
    const transactionOperation = transaction ?? this.prismaService;

    return await transactionOperation.assessment.findUnique({
      where: {
        id: assesmentId,
      },
    });
  }

  async updateStep(assesmentId: number, step: number, transaction?: PrismaTransactionClient): Promise<boolean> {
    const transactionOperation = transaction ?? this.prismaService;
    const assessment = await this.fetchStep(assesmentId, transaction);

    //if statement to make sure step stores last step reached by the user
    if (!assessment.step || assessment.step < step)
      await transactionOperation.assessment.update({
        where: {
          id: assesmentId,
        },
        data: {
          step,
        },
      });

    return true;
  }

  async isAssessmentByIdExists(id: number): Promise<Assessment> {
    const assessment = await this.prismaService.assessment.findUnique({ where: { id } });
    if (!assessment) throw new MultiLanguageException(LanguageErrorEnum.ASSESSMENT_NOT_FOUND);
    return assessment;
  }

  async isAssessmentOwner(user: UserPlayloadResponse, id: number): Promise<boolean> {
    const assessment = await this.isAssessmentByIdExists(id);

    if (Number(assessment.patientId) !== user.patientId) throw new MultiLanguageException(LanguageErrorEnum.ASSESSMENT_NOT_YOURS);
    return true;
  }
