import { Injectable } from '@nestjs/common';
import { PrismaService } from 'src/common/services/prisma.service';

import { UserPlayloadResponse } from '../patients/response/user-playload.response';
import { CreatePatientAchievementDto } from './dtos/patient.achievement.dto';
import { AchievementDairyResponse, PatientAchievementResponse } from './responses/patient.achievement.response';

import { AchievementResponse } from './responses/achievement.response';
import { DateDto } from 'src/common/dto/date.dto';
import { plainToInstance } from 'class-transformer';
import { LanguageEnum } from '@prisma/client';

@Injectable()
export class AchievementsService {
  constructor(private readonly prisma: PrismaService) {}

  async getAchievements(language: LanguageEnum): Promise<AchievementResponse[]> {
    const achievements = await this.prisma.achievement.findMany({ include: { translation: true } });
    const achievementResponses = achievements.map((achievement) => ({
      type: achievement.type,
      title: achievement.translation.find((t) => t.language === language)?.title ?? null,
      description: achievement.translation.find((t) => t.language === language)?.description ?? null,
      imageName: achievement.imageName,
      points: achievement.points,
    }));

    return achievementResponses;
  }

  async createPatientAchievement(patient: UserPlayloadResponse, dto: CreatePatientAchievementDto) {
    const patientAchievement = await this.prisma.patientAchievement.create({
      data: {
        patient: { connect: { id: patient.patientId } }, // Assuming 'patient' object contains 'id'
        achievement: { connect: { type: dto.type } }, // Correctly connect achievement by type
        isCompleted: dto.isCompleted,
        endDate: new Date(),
      },
    });
    return patientAchievement;
  }
  async getPatientAchievements(patient: UserPlayloadResponse, language: LanguageEnum): Promise<PatientAchievementResponse[]> {
    // Retrieve all achievements
    const allAchievements = await this.prisma.achievement.findMany({ include: { translation: true } });

    // Retrieve patient's completed achievements
    const patientAchievements = await this.prisma.patientAchievement.findMany({
      where: {
        patient: {
          id: patient.patientId,
        },
      },
      include: { achievement: { include: { translation: true } } },
    });

    // Create a map of completed achievement types for the patient
    const completedAchievementTypes = patientAchievements.map((pa) => pa.type);

    // Filter out completed achievements
    const missingAchievements = allAchievements.filter((achievement) => !completedAchievementTypes.includes(achievement.type));
    // Create patient achievement responses for missing achievements
    const missingAchievementResponses = missingAchievements.map((achievement) => ({
      id: null, // Set to null for missing achievements
      isCompleted: false,
      type: achievement.type,
      title: achievement.translation.find((t) => t.language === language)?.title,
      description: achievement.translation.find((t) => t.language === language)?.description,
      imageName: achievement.imageName,
      endDate: null, // Set to null for missing achievements
      points: achievement.points,
      startDate: null, // Set to null for missing achievements
    }));

    // Create patient achievement responses for completed achievements
    const completedAchievementResponses = patientAchievements.map((patientAchievement) => {
      return {
        id: patientAchievement.id,
        isCompleted: true,
        type: patientAchievement.achievement.type,
        title: patientAchievement.achievement.translation.find((t) => t.language === language)?.title,
        description: patientAchievement.achievement.translation.find((t) => t.language === language)?.description,
        imageName: patientAchievement.achievement.imageName,
        endDate: patientAchievement.endDate,
        points: patientAchievement.achievement.points,
        startDate: patientAchievement.startDate,
      };
    });

    // Combine completed and missing achievements
    const allAchievementResponses = [...completedAchievementResponses, ...missingAchievementResponses];

    return allAchievementResponses;
  }
  async getPatientAchievement(patientId: number, dateDto: DateDto, isLastThreeRow: boolean, language: LanguageEnum): Promise<AchievementDairyResponse[]> {
    const { dateEndTime, dateStartTime } = dateDto;
    const take = isLastThreeRow ? 3 : undefined;

    const patientAchievements = await this.prisma.patientAchievement.findMany({
      where: {
        ...(!isLastThreeRow && {
          endDate: {
            //recheck it later
            gt: dateStartTime,
            lt: dateEndTime,
          },
        }),
        patient: {
          id: patientId,
        },
      },
      take,
      include: {
        achievement: { include: { translation: true } },
      },
    });

    const completedAchievementResponses = patientAchievements.map((patientAchievement) => {
      return {
        id: patientAchievement.id,
        type: patientAchievement.achievement.type,
        title: patientAchievement.achievement.translation.find((t) => t.language === language)?.title,
        description: patientAchievement.achievement.translation.find((t) => t.language === language)?.description,
        endDate: patientAchievement.endDate,
        points: patientAchievement.achievement.points,
      };
    });
    return plainToInstance(AchievementDairyResponse, completedAchievementResponses, {
      excludeExtraneousValues: false,
      enableImplicitConversion: true,
    });
  }
}
