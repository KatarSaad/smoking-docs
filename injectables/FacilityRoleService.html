import { Injectable } from '@nestjs/common';
import { LanguageEnum, Prisma, RoleProtectionEnum, RoleRestrictionEnum } from '@prisma/client';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { RoleEnum } from 'src/common/enums/role.enum';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';
import { PaginationRoles } from 'src/common/responses/Paggination-data.response';
import { PrismaService } from 'src/common/services/prisma.service';
import { AuthService } from 'src/modules/auth/auth.service';
import { FacilityGroupService } from 'src/modules/facility-group/facility-group.service';
import { FetchRoleFacilitiesResponse } from 'src/modules/facility-role/response/fetch-role-facilities.response';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';
import { PermissionsService } from 'src/modules/permissions/permissions.service';
import { HealthFacilityService } from '../facilities/health-facility.service';
import { AssignFacilityRoleDto } from './dto/assign-facility-role.dto';
import { CreateFacilityRoleDto } from './dto/create-facility-role.dto';
import { FetchFaciltiyRoleDto } from './dto/fetch-facility-role.dto';
import { UnassignFacilityRoleDto } from './dto/unassign-facility-role.dto';
import { UpdateFacilityRoleDto } from './dto/update-role.dto';
import { FacilityRoleResponse } from './response/created-facility-role.response';
import { FullFacilityRoleResponse } from './response/full-facility-role.response';
import { GetUnassignedUsersResponse } from './response/get-unassigned-users.response';
import { GetUsersByRoleResponse } from './response/get-users-by-role.response';
import { NameAndIdResponse } from 'src/common/responses/id-and-name.response';

@Injectable()
export class FacilityRoleService {
  constructor(
    private readonly prismaService: PrismaService,
    private facilityGroupService: FacilityGroupService,
    private healthFacilityService: HealthFacilityService,
    private authService: AuthService,
    private permissionService: PermissionsService,
  ) {}

  async SearchRoleByName(fetchFaciltiyRoleDto: FetchFaciltiyRoleDto): Promise<PaginationRoles> {
    const { search, currentPage, sizePerPage, sortColumn, isSorted } = fetchFaciltiyRoleDto;
    const incrementedPage = currentPage - 1;
    const searchWord = `${search}%`; //this is a trick to pass the search value
    const tableArg = Prisma.raw(sortColumn); //to be able to use the column name as variable
    const systemRoleRestriction = RoleRestrictionEnum.CUSTOM;
    //i m using sql query because i cant filter on totalUsers using prisma

    const result: [FetchRoleFacilitiesResponse] = await this.prismaService.$queryRaw`
    SELECT id,name,isActive,
    (SELECT COUNT(*) FROM admin where facilityRole.id=admin.roleId) as totalUsers
    FROM FacilityRole facilityRole
    ${search ? Prisma.sql`WHERE name LIKE ${searchWord} AND restriction = ${systemRoleRestriction}` : Prisma.sql`WHERE restriction = ${systemRoleRestriction}`}
    ORDER BY isActive DESC, ${tableArg} ${isSorted ? Prisma.sql` DESC` : Prisma.sql` ASC`}

    LIMIT ${sizePerPage} OFFSET ${incrementedPage * sizePerPage};`;

    //cant seriaize with serializer decorator cause it will produce problems with our pagination
    const serializing = result.map((journey) => {
      return { isActive: !!journey.isActive, id: journey.id, name: journey.name, totalUsers: journey.totalUsers };
    });

    const totalItems = await this.prismaService.facilityRole.aggregate({
      _count: {
        _all: true,
      },
      where: {
        name: {
          startsWith: search,
        },
        restriction: systemRoleRestriction,
      },
    });

    return { data: serializing, totalItems: totalItems._count._all };
  }

  async create(createFacilityRoleDto: CreateFacilityRoleDto): Promise<FacilityRoleResponse> {
    const { name, facilityGroups, facilities, isActive, permissions } = createFacilityRoleDto;

    if (facilities && facilityGroups) throw new MultiLanguageException(LanguageErrorEnum.CANNOT_PROVIDE_GROUP_AND_FACILITYGROUP);

    await this.isFacilitiyRoleNameExists(null, name);

    const permissionsNb = await this.permissionService.fetchNumberPermissionsMatched(permissions, true);
    if (permissionsNb < permissions.length) throw new MultiLanguageException(LanguageErrorEnum.PERMISSION_NOT_FOUND);

    if (facilityGroups) {
      const facilityGroupsNb = await this.facilityGroupService.fetchFacilityGroups(facilityGroups);
      if (facilityGroupsNb < facilityGroups.length) throw new MultiLanguageException(LanguageErrorEnum.GROUP_FACILITY_NOT_FOUND);
    }

    if (facilities) {
      const facilitiesNumber = await this.healthFacilityService.fetchNumberFacilitiesMatched(facilities);
      if (facilitiesNumber < facilities.length) throw new MultiLanguageException(LanguageErrorEnum.FACILITY_NOT_FOUND);
      await this.healthFacilityService.isFacilitiesSameGroup(facilities);
    }

    const faciltiyRole = await this.prismaService.facilityRole.create({
      data: {
        name,
        isActive,
        protection: RoleProtectionEnum.NORMAL,
        permissions: {
          connect: permissions,
        },
        facilities: {
          connect: facilities,
        },
        facilityGroups: {
          connect: facilityGroups,
        },
      },
      select: {
        id: true,
        name: true,
        isActive: true,
        restriction: true,
        facilityGroups: { select: { id: true, name: true } },
      },
    });
    return faciltiyRole;
  }

  async fetchRolesNames(): Promise<NameAndIdResponse[]> {
    return await this.prismaService.facilityRole.findMany({
      select: {
        id: true,
        name: true,
      },
      where: {
        restriction: RoleRestrictionEnum.CUSTOM,
      },
    });
  }

  async assignRole(user: UserPlayloadResponse, assignFacilityRoleDto: AssignFacilityRoleDto) {
    assignFacilityRoleDto.managerId.forEach(async (id) => {
      await this.authService.isNotPatient(id);
    });
    await this.isRoleMutable(assignFacilityRoleDto.roleId, user);

    return await this.prismaService.admin.updateMany({
      where: { userId: { in: assignFacilityRoleDto.managerId } },
      data: {
        roleId: {
          set: assignFacilityRoleDto.roleId,
        },
      },
    });
  }

  async unassignRole(unassignFacilityRoleDto: UnassignFacilityRoleDto) {
    const role = await this.prismaService.facilityRole.findUnique({
      where: {
        id: unassignFacilityRoleDto.roleId,
      },
    });

    if (role.protection !== RoleProtectionEnum.NORMAL) {
      throw new MultiLanguageException(LanguageErrorEnum.ONLY_UNASSIGN_CUSTOM_ROLES);
    }
    try {
      return await this.prismaService.admin.update({
        where: {
          roleId: unassignFacilityRoleDto.roleId,
          userId: unassignFacilityRoleDto.managerId,
        },
        data: {
          role: {
            disconnect: true,
          },
        },
      });
    } catch (error) {
      throw new MultiLanguageException(LanguageErrorEnum.USER_DOES_NOT_HAVE_THAT_ROLE);
    }
  }
  async getUsersByRole(id: number): Promise<GetUsersByRoleResponse[]> {
    const admins = await this.prismaService.user.findMany({
      where: {
        admin: {
          roleId: id,
        },
      },
      select: {
        id: true,
        qid: true,
        personalInformation: {
          select: {
            firstName: true,
            lastName: true,
          },
        },
        admin: {
          select: {
            role: {
              select: {
                id: true,
                name: true,
                facilities: {
                  select: {
                    name: true,
                    id: true,
                  },
                },
                facilityGroups: {
                  include: {
                    facilities: {
                      select: {
                        name: true,
                        id: true,
                      },
                    },
                  },
                },
              },
            },
          },
        },
      },
    });

    return admins.map((admin) => {
      const groupFacilities = [];
      if (admin.admin.role.facilityGroups.length) {
        admin.admin.role.facilityGroups.forEach((fg) => {
          groupFacilities.push(...fg.facilities);
        });
      }
      return {
        id: admin.id,
        firstName: admin.personalInformation.firstName,
        lastName: admin.personalInformation.lastName,
        qid: admin.qid,
        facilities: admin.admin.role.facilities.length ? admin.admin.role.facilities : groupFacilities,
        roleId: admin.admin.role.id,
        roleName: admin.admin.role.name,
      };
    });
  }

  async getUnAssignedUsers(): Promise<GetUnassignedUsersResponse[]> {
    const users = await this.prismaService.user.findMany({
      where: {
        admin: {
          roleId: null,
        },
      },
      select: {
        id: true,
        qid: true,
        personalInformation: {
          select: {
            firstName: true,
            lastName: true,
          },
        },
      },
    });

    return users.map((user) => ({
      id: user.id,
      firstName: user.personalInformation.firstName,
      lastName: user.personalInformation.lastName,
      qid: user.qid,
    }));
  }

  async fetchFacilityRole(language: LanguageEnum, id: number): Promise<FullFacilityRoleResponse> {
    const role = await this.prismaService.facilityRole.findUnique({
      where: { id },
      select: {
        id: true,
        isActive: true,
        name: true,
        restriction: true,
        permissions: {
          select: {
            id: true,
            permissionsTranslation: {
              where: {
                language: {
                  code: language,
                },
              },
              select: {
                name: true,
              },
            },
          },
        },
        facilities: {
          select: {
            id: true,
            name: true,
          },
        },
        facilityGroups: {
          select: {
            id: true,
            name: true,
          },
        },
      },
    });

    if (!role) throw new MultiLanguageException(LanguageErrorEnum.ROLE_NOT_FOUND);
    return {
      ...role,
      permissions: role.permissions.map((permission) => ({
        id: permission.id,
        permission: {
          name: permission.permissionsTranslation[0].name,
        },
      })),
    };
  }

  async updateRole(language: LanguageEnum, roleId: number, updateFacilityRoleDto: UpdateFacilityRoleDto, user: UserPlayloadResponse): Promise<FullFacilityRoleResponse> {
    const { permissions, name, facilities, facilityGroups, isActive } = updateFacilityRoleDto;
    if (facilities && facilityGroups) throw new MultiLanguageException(LanguageErrorEnum.CANNOT_PROVIDE_GROUP_AND_FACILITYGROUP);
    if (facilities?.length === 0 && facilityGroups?.length === 0) throw new MultiLanguageException(LanguageErrorEnum.PROVIDE_FACILITIES_OR_GROUPFACILITY);

    await this.isFacilitiyRoleNameExists(roleId, name);

    await this.isRoleMutable(roleId, user);

    if (permissions) {
      const permissionsNb = await this.permissionService.fetchNumberPermissionsMatched(permissions, true);
      if (permissionsNb < permissions.length) throw new MultiLanguageException(LanguageErrorEnum.PERMISSION_NOT_FOUND);
    }

    if (facilityGroups) {
      const groupFacilities = await this.facilityGroupService.fetchFacilityGroups(facilityGroups);

      if (groupFacilities < facilityGroups.length) throw new MultiLanguageException(LanguageErrorEnum.GROUP_FACILITY_NOT_FOUND);
    }

    if (facilities) {
      const facilitiesNumber = await this.healthFacilityService.fetchNumberFacilitiesMatched(facilities);

      if (facilitiesNumber < facilities.length) throw new MultiLanguageException(LanguageErrorEnum.FACILITY_NOT_FOUND);
      await this.healthFacilityService.isFacilitiesSameGroup(facilities);
    }

    const result = await this.prismaService.facilityRole.update({
      where: { id: roleId },
      data: {
        name,
        isActive,
        permissions: {
          set: permissions,
        },
        facilities: { set: facilities },
        facilityGroups: { set: facilityGroups },
      },
      select: {
        id: true,
        isActive: true,
        name: true,
        restriction: true,
        permissions: {
          select: {
            id: true,
            permissionsTranslation: {
              where: {
                language: {
                  code: language,
                },
              },
              select: {
                name: true,
              },
            },
          },
        },
        facilities: {
          select: {
            id: true,
            name: true,
          },
        },
        facilityGroups: {
          select: {
            id: true,
            name: true,
          },
        },
      },
    });
    return {
      ...result,
      permissions: result.permissions.map((permission) => ({
        id: permission.id,
        permission: {
          name: permission.permissionsTranslation[0].name,
        },
      })),
    };
  }

  async removeRoleById(id: number, user: UserPlayloadResponse) {
    await this.isRoleMutable(id, user);

    return await this.prismaService.facilityRole.delete({
      where: { id },
    });
  }

  async isFacilitiyRoleNameExists(roleId: number | null, name: string): Promise<boolean> {
    const faciltiyRoleName = await this.prismaService.facilityRole.findFirst({
      where: {
        ...(roleId && {
          id: {
            not: roleId,
          },
        }),
        name,
      },
      select: { id: true },
    });
    if (faciltiyRoleName) throw new MultiLanguageException(LanguageErrorEnum.ROLE_NAME_EXISTS);
    return true;
  }

  async isRoleMutable(id: number, user: UserPlayloadResponse): Promise<boolean> {
    const highRoles = [RoleEnum.SUPERADMIN.toString()];
    const role = await this.prismaService.facilityRole.findUnique({
      where: {
        id,
      },
    });

    if (!role) throw new MultiLanguageException(LanguageErrorEnum.ROLE_NOT_FOUND);
    if (role.protection === RoleProtectionEnum.SYSTEM) throw new MultiLanguageException(LanguageErrorEnum.SYSTEM_ROLE_IMMUTABLE);

    const currentAdmin = await this.prismaService.admin.findUnique({
      where: {
        id: user.adminId,
      },
      select: {
        role: {
          select: {
            name: true,
          },
        },
      },
    });

    if (role.protection === RoleProtectionEnum.PROTECTED && !highRoles.includes(currentAdmin.role.name))
      throw new MultiLanguageException(LanguageErrorEnum.PROTECTED_ROLE);

    return true;
  }
}
