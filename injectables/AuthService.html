import { Injectable } from '@nestjs/common';
import { JwtService } from '@nestjs/jwt';
import { plainToInstance } from 'class-transformer';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';
import { CommonService } from 'src/common/services/common.service';
import { EnvironmentService } from 'src/common/services/environment.service';
import { PrismaService } from 'src/common/services/prisma.service';
import { AssessmentsService } from 'src/modules/assessments/assessments.service';
import { UsersService } from 'src/modules/users/users.service';
import { PatientService } from '../patients/patient.service';
import { UserPlayloadResponse } from '../patients/response/user-playload.response';
import { BcryptService } from './bcrypt.service';
import { LoginUserDto } from './dto/login-user.dto';
import { LogoutUserDto } from './dto/logout-user.dto';
import { RegisterUserDto } from './dto/register-user.dto';
import { LoginPatientResponse } from './response/login-mobile.reponse';
import { InjectQueue } from '@nestjs/bullmq';
import { QueueEnum } from 'src/common/enums/queues.enum';
import { Queue } from 'bullmq';
import { parseExpression } from 'cron-parser';
import { NotificationHubService } from '../notification/notification-hub.service';
import { EmailDto } from 'src/modules/auth/dto/email.dto';
import { UpdatePasswordWithCode } from 'src/modules/auth/dto/update-password-with-code.dto';
import { PrismaTransactionClient } from 'src/common/types/transaction.type';
import { MailService } from 'src/common/services/mail.service';
import { TokenDto } from 'src/common/dto/token.dto';
import { TokenPayloadResponse } from 'src/common/responses/token-payload.response';
import { TokenResponse } from 'src/common/responses/token.response';
import { AssessmentDetailResponse } from 'src/modules/assessments/resposnes/assessment-detail.response';

@Injectable()
export class AuthService {
  constructor(
    private readonly patientService: PatientService,
    private readonly jwtService: JwtService,
    private readonly bcryptService: BcryptService,
    private environmentService: EnvironmentService,
    private commonService: CommonService,
    private prismaService: PrismaService,
    private usersService: UsersService,
    private assessmentService: AssessmentsService,
    private notificationHubService: NotificationHubService,
    private mailService: MailService,
    @InjectQueue(QueueEnum.AUTH_QUEUE) private authQueue: Queue,
  ) {}

  async onModuleInit() {
    const interval = parseExpression('0 0 0 1 * *');
    console.log(interval.next());

    await this.authQueue.drain();

    await this.authQueue.add(
      'logoutCleanup',
      {},
      {
        //repeat every 1st of each month
        repeat: {
          pattern: '0 0 0 1 * *',
        },
      },
    );
  }

  async loginAdmin(loginUserDto: LoginUserDto): Promise<TokenResponse> {
    //check if user exists

    const user = await this.validateUser(loginUserDto.qid, loginUserDto.password);

    if (user.patientId) throw new MultiLanguageException(LanguageErrorEnum.FORBIDDEN_RESOURCES);
    const bearerToken = await this.generateBearerToken(user.id);
    await this.updateLastLogin(user.id);
    const adminByUser = await this.usersService.findAdminByUserId(user.id);
    if (!adminByUser.admin?.role) throw new MultiLanguageException(LanguageErrorEnum.ADMIN_HAS_NO_ROLE);
    return { bearerToken };
  }

  async register(registerUserDto: RegisterUserDto): Promise<{ result: UserPlayloadResponse; bearerToken: string }> {
    const hashedPassword = await this.bcryptService.hash(registerUserDto.password);

    await this.usersService.isQidNotExist(registerUserDto.qid, null);

    const user = await this.patientService.createPatient({
      ...registerUserDto,
      password: hashedPassword,
    });

    const result = plainToInstance(UserPlayloadResponse, user, {
      excludeExtraneousValues: true,
    });

    const bearerToken = await this.generateBearerToken(Number(result.id));
    return { result, bearerToken };
  }

  async forgetPassword(qid: string): Promise<{ resetPasswordToken: string }> {
    const user = await this.patientService.findOneUser('qid', qid);
    if (!user) throw new MultiLanguageException(LanguageErrorEnum.WRONG_CREDENTIALS);
    // generate resest pasword token and store it in redis
    const resetPasswordToken = await this.generateResetPasswordToken(user.qid);
    // send email with reset password link
    return { resetPasswordToken };
  }

  async resetPassword(newPassword: string, qid: string): Promise<{ bearerToken: string }> {
    //hash the new password and update user password
    const hashedPassword = await this.bcryptService.hash(newPassword);
    const user = await this.patientService.updatePatientPassword({ qid }, hashedPassword);
    // generate access token and return it
    const bearerToken = await this.generateBearerToken(this.commonService.BigIntParser(user.id)); //Number.parseInt(user.id.toString())
    return { bearerToken };
  }

  async tokenChecker(user: UserPlayloadResponse): Promise<AssessmentDetailResponse> {
    const userAssessments = await this.assessmentService.fetchUserAssessmentsByUserId(user);

    return userAssessments;
  }

  async loginPatient(loginUserDto: LoginUserDto): Promise<LoginPatientResponse> {
    //check if user exists
    const user = await this.validateUser(loginUserDto.qid, loginUserDto.password);

    const bearerToken = await this.generateBearerToken(user.id);
    const userAssessments = await this.assessmentService.fetchUserAssessmentsByUserId(user);
    const { step } = userAssessments;
    //update last login
    await this.updateLastLogin(user.id);
    return { bearerToken, ...userAssessments, step };
  }

  async logout(user: UserPlayloadResponse, userLogoutDto: LogoutUserDto, device_id?: string): Promise<boolean> {
    await this.prismaService.userToken
      .create({
        data: {
          access_token: userLogoutDto.access_token,
          refresh_token: userLogoutDto.refresh_token,
          userId: user.id,
        },
      })
      .catch((error) => {
        console.error(error);
        return false;
      });
    if (device_id) {
      await this.notificationHubService.removeDeviceInstallation({ deviceId: device_id }, user);
    }
    return true;
  }

  async validateUser(qid: string, pass: string): Promise<UserPlayloadResponse> {
    const user = await this.patientService.findOneUser('qid', qid);
    const isValidPassword = await this.bcryptService.compare(pass, user.password);
    if (!isValidPassword) throw new MultiLanguageException(LanguageErrorEnum.WRONG_CREDENTIALS);
    const result = plainToInstance(UserPlayloadResponse, user, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });
    return result;
  }

  async generateBearerToken(userId: number, expiresIn?: string | number): Promise<string> {
    const payload: TokenPayloadResponse = { sub: userId };

    return await this.jwtService.signAsync(payload, {
      secret: this.environmentService.jasonWebTokenConfig.JWTSecretKey,
      expiresIn: expiresIn ?? '10 days',
    });
  }

  async isNotPatient(userId: number): Promise<boolean> {
    const user = await this.prismaService.user.findUnique({
      where: {
        id: userId,
      },
      select: {
        patient: { select: { id: true } },
      },
    });
    if (!user) throw new MultiLanguageException(LanguageErrorEnum.USER_NOT_FOUND);

    if (user.patient) throw new MultiLanguageException(LanguageErrorEnum.FORBIDDEN_RESOURCES);

    return true;
  }

  async generateResetPasswordToken(qid: string): Promise<string> {
    const token = await this.jwtService.signAsync(
      { sub: qid },
      {
        secret: this.environmentService.jasonWebTokenConfig.JWTSecretKey,
      },
    );
    return token;
  }

  async updateLastLogin(userId: number): Promise<void> {
    await this.prismaService.user.update({
      where: {
        id: userId,
      },
      data: {
        lastLogin: new Date(),
      },
    });
  }

  async sendLoginCode(emailDto: EmailDto): Promise<boolean> {
    const { email } = emailDto;
    const code = this.commonService.randomNumber();

    const { id } = await this.usersService.getUserPersonalInformationByEmail(email);

    const token = await this.generateBearerToken(Number(id), '1d');

    this.mailService.sendLoginCode(email, token, code);

    await this.prismaService.signInCode.upsert({
      where: {
        userId: id,
      },
      create: {
        userId: id,
        code: code.toString(),
      },
      update: { code: code.toString(), isUsed: false },
    });

    return true;
  }

  async updatePasswordByCode(updatePasswordWithCode: UpdatePasswordWithCode, tokenDto: TokenDto): Promise<boolean> {
    const { sub } = await this.verifyToken(tokenDto.token);

    await this.usersService.isCodeByUserExists(sub, updatePasswordWithCode.code);

    await this.prismaService.$transaction(async (tx) => {
      await this.updatePassword(sub, updatePasswordWithCode.password, tx);

      await tx.user.update({
        where: {
          id: sub,
        },
        data: {
          signInCode: {
            update: {
              isUsed: true,
            },
          },
        },
      });
    });

    return true;
  }

  async updatePassword(userId: number, password: string, transaction: PrismaTransactionClient): Promise<boolean> {
    const prismaOperation = transaction ?? this.prismaService;

    await this.usersService.isUserByIdExists(userId, transaction);

    const hashedPassword = await this.bcryptService.hash(password);

    await prismaOperation.user.update({
      where: { id: userId },
      data: {
        password: hashedPassword,
      },
    });

    return true;
  }

  async verifyToken(token: string): Promise<TokenPayloadResponse> {
    const payload: TokenPayloadResponse = await this.jwtService.verifyAsync(token, {
      secret: this.environmentService.jasonWebTokenConfig.JWTSecretKey,
    });
    return payload;
  }
}
