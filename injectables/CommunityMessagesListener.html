import { ConnectedSocket, MessageBody, OnGatewayConnection, OnGatewayDisconnect, SubscribeMessage, WebSocketServer } from '@nestjs/websockets';
import { Server, Socket } from 'socket.io';
import { JwtAuth } from 'src/common/decorators/jwt-auth.decorator';
import { UserPlayloadResponse } from '../patients/response/user-playload.response';
import { User } from 'src/common/decorators/auth.decorator';
import { Inject, Injectable, Logger, forwardRef } from '@nestjs/common';
import { CommonService } from 'src/common/services/common.service';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { CommunityMessagesService } from './community-messages.service';
import { CommunityMessageDto } from './dto/community-messages.dto';
import { PatientService } from '../patients/patient.service';
import { CommunityChatSocketResponse } from './response/community-chat-socket.response';
import { DeleteMessageDto } from '../private-chat-messages/dto/delete-message.dto';
import { SocketEventEnum } from 'src/common/enums/socket-event.enum';

@Injectable()
export class CommunityMessagesListener implements OnGatewayConnection, OnGatewayDisconnect {
  constructor(
    private communityMessagesService: CommunityMessagesService,
    @Inject(forwardRef(() => PatientService))
    private patientService: PatientService,
    private commonService: CommonService,
  ) {}
  @WebSocketServer() server: Server;

  private logger = new Logger('communityChatGateway');

  async sendCommunityChatMessage(payload: CommunityMessageDto, client: Socket, user: UserPlayloadResponse): Promise<CommunityChatSocketResponse> {
    const patientProfile = await this.patientService.patientProfile(user, 'ENGLISH');

    if (new Date().getFullYear() - new Date(patientProfile.personalInformation.dateOfBirth).getFullYear() < 18) {
      throw new MultiLanguageException(LanguageErrorEnum.FORBIDDEN_RESOURCES);
    }
    const senderId = Number(user.id);

    const addMessageResponse = await this.communityMessagesService.addMessageToCommunity(user, payload); //save message in db
    const { sender } = addMessageResponse;
    const isSender = senderId === user.id;

    client.broadcast.emit(SocketEventEnum.MESSAGE_COMMUNITY_CHAT, this.commonService.objectParser({ ...addMessageResponse, sender: { ...sender, isSender } }));

    return { success: true, messageId: Number(addMessageResponse.id) };
  }

  @JwtAuth()
  @SubscribeMessage(SocketEventEnum.DELETE_MESSAGE_COMMUNITY_CHAT)
  async handleChatDeleteEvent(
    @MessageBody()
    payload: DeleteMessageDto,
    @ConnectedSocket() client: Socket,
    @User() user: UserPlayloadResponse,
  ): Promise<CommunityChatSocketResponse> {
    const deleteMessageResponse = await this.communityMessagesService.deleteMessage(user, payload.id); //save message in db
    client.broadcast.emit(SocketEventEnum.DELETE_MESSAGE_COMMUNITY_CHAT, { messageId: Number(deleteMessageResponse.id) });

    return { success: true, messageId: Number(deleteMessageResponse.id) };
  }

  async handleConnection(socket: Socket): Promise<void> {
    this.logger.log(`Socket connected: ${socket.id}`);
  }

  async handleDisconnect(socket: Socket): Promise<void> {
    this.logger.log(`Socket disconnected: ${socket.id}`);
  }
}
