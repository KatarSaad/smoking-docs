import { Injectable } from '@nestjs/common';
import { ArticleStatusEnum, LanguageEnum } from '@prisma/client';
import { plainToInstance } from 'class-transformer';
import { SmokingCessationPlan } from 'src/common/enums/smoking-cessation-plan.enum';
import { CommonService } from 'src/common/services/common.service';
import { PrismaService } from 'src/common/services/prisma.service';
import { ArticleResponse } from 'src/modules/articles/response/article.response';
import { GroupsService } from 'src/modules/groups/groups.service';
import { NrtsService } from 'src/modules/nrts/nrts.service';
import { OnboardingService } from 'src/modules/onboarding/onboarding.service';
import { PatientScPlanResponse } from 'src/modules/sc-plans/response/patient-sc-plan.response';
import { ScPlansHomeResponse } from 'src/modules/sc-plans/response/sc-plans-home.response';
import { UsersService } from 'src/modules/users/users.service';
import { RecommendedNotificationResponse } from '../notification/response/recommended-notification.response';
import { recommendedArticleResponse } from './response/recommended.article.response';

@Injectable()
export class ScPlansService {
  constructor(
    private readonly prismaService: PrismaService,
    private readonly groupsService: GroupsService,
    private readonly nrtsService: NrtsService,
    private readonly onboardingService: OnboardingService,
    private readonly commonService: CommonService,
    private readonly usersService: UsersService,
  ) {}

  async fetchAuthedPatientSCJourneyPlan(language: LanguageEnum, userId: number): Promise<PatientScPlanResponse> {
    const articlesTest = [];
    const user = await this.usersService.findPatientByUserId(userId);
    const patientId = Number(user.patient.id);
    const { quitDate } = await this.onboardingService.getHabitInfo(patientId);
    const isQuittedDatePassed = quitDate < this.commonService.adjustDate();
    const daysAfterQuitting = this.commonService.datesDifferenceInDays(quitDate, this.commonService.adjustDate());

    const isChatBotPermission = true;
    const { dateOfBirth } = await this.usersService.getUserPersonalInformation(userId);
    const patientAge = this.commonService.dateOfBirthToAge(dateOfBirth);
    const isCommunityChatPermission = patientAge >= 18;

    //fetch nrts
    const currentPatientNrtDetail = await this.nrtsService.userNrt(language, userId);

    //fetch articles
    const patientGroups = await this.groupsService.fetchPatientGroups(patientId);
    const patientGroupsIds = patientGroups.map((patientGroup) => patientGroup.id);

    const patientArticlePackages = await this.prismaService.articlePackage.findMany({
      where: {
        OR: [
          {
            targetGroupId: { in: patientGroupsIds },
          },
          {
            isEditable: false,
          },
        ],
      },
      select: {
        recommendedArticles: {
          where: {
            article: {
              status: ArticleStatusEnum.APPROVED,
            },
            OR: [
              { displayDaysAfterQuitting: null },

              isQuittedDatePassed
                ? {
                    displayDaysAfterQuitting: {
                      lte: daysAfterQuitting,
                    },
                  }
                : {},
            ],
          },
          select: {
            article: {
              select: {
                id: true,
                image: true,
                status: true,
                scheduledAt: true,
                updatedAt: true,
                genderFilter: true,
                targetGroups: {
                  select: {
                    id: true,
                    name: true,
                  },
                },
                publisher: {
                  select: {
                    user: {
                      select: {
                        personalInformation: {
                          select: {
                            firstName: true,
                            lastName: true,
                          },
                        },
                      },
                    },
                  },
                },
                articleLabel: {
                  select: {
                    id: true,
                    articleLabelsTranslation: {
                      where: {
                        language: { code: language },
                      },
                      select: {
                        name: true,
                      },
                    },
                  },
                },
                articlesTranslation: {
                  where: {
                    language: {
                      code: language,
                    },
                  },
                  select: {
                    title: true,
                    body: true,
                  },
                },
              },
            },
          },
        },
      },
    });

    //we just need to get the actual articles from the packages
    patientArticlePackages.forEach(
      (articlePackage) => articlePackage?.recommendedArticles.forEach((recommendedArticles) => articlesTest.push(recommendedArticles.article)),
    );

    const serializedRecommendedArticles = plainToInstance(ArticleResponse, articlesTest, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    //fetch chat permission

    const chatPermissions = {
      isChatBotPermission,
      isCommunityChatPermission,
    };
    return { currentPatientNrtDetail, recommendedArticles: serializedRecommendedArticles || [], chatPermissions };
  }

  async fetchPatientSCJourneyPlanNotificationsByHealthProvider(language: LanguageEnum, userId: number): Promise<RecommendedNotificationResponse[]> {
    const notificationsResult = [];
    const user = await this.usersService.findPatientByUserId(userId);
    const patientId = Number(user.patient.id);

    //fetch notifications
    const patientGroups = await this.groupsService.fetchPatientGroups(patientId);
    const patientGroupsIds = patientGroups.map((patientGroup) => patientGroup.id);

    const patientNotificationPackages = await this.prismaService.notificationPackage.findMany({
      where: {
        OR: [
          {
            targetGroupId: { in: patientGroupsIds },
          },
        ],
      },
      select: {
        notifications: {
          select: {
            id: true,
            imageLink: true,
            isCritical: true,
            deliveryDate: true,
            notificationType: true,
            status: true,
            notificationTranslation: {
              where: {
                language: {
                  code: language,
                },
              },
              select: {
                title: true,
                body: true,
              },
            },
            targetGroupIds: {
              select: {
                id: true,
                name: true,
              },
            },
            recipientId: true,
            subjectId: true,
            NotificationSeen: {
              include: {
                seenBy: true,
              },
            },
            daysAfterQuitting: true,
          },
          orderBy: {
            createdAt: 'desc',
          },
        },
      },
    });

    patientNotificationPackages.forEach(
      (notificationPackage) => notificationPackage?.notifications.forEach((recommendedNotifications) => notificationsResult.push(recommendedNotifications)),
    );
    const serializedRecommendedNotifications = plainToInstance(RecommendedNotificationResponse, notificationsResult, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    return serializedRecommendedNotifications;
  }

  async fetchSCPlanHome(language: LanguageEnum): Promise<ScPlansHomeResponse> {
    const [NRTList, NRTDefault, articlePackageList, defaultArticlePackage, automaticNotificationPackage, defaultNotificationPackage] =
      await this.prismaService.$transaction([
        this.prismaService.nrt.findMany({
          select: {
            id: true,
            nrtDetail: {
              select: {
                id: true,
                nrtType: true,
                nrtDetailsTranslation: {
                  where: {
                    language: {
                      code: language,
                    },
                  },
                  select: {
                    gum: true,
                    patch: true,
                  },
                },
              },
            },
            priority: true,
            isEditable: true,
            targetGroup: {
              select: {
                id: true,
                name: true,
              },
            },
          },
          take: 2,
          orderBy: {
            priority: 'desc',
          },
        }),

        this.prismaService.nrt.findMany({
          select: {
            id: true,
            nrtDetail: {
              select: {
                id: true,
                nrtType: true,
                nrtDetailsTranslation: {
                  where: {
                    language: {
                      code: language,
                    },
                  },
                  select: {
                    gum: true,
                    patch: true,
                  },
                },
              },
            },
            priority: true,
            isEditable: true,
            targetGroup: {
              select: {
                id: true,
                name: true,
              },
            },
          },
          take: 1,
          orderBy: {
            priority: 'asc',
          },
        }),

        //we will split the default package and the custom packages to be able to filter
        this.prismaService.articlePackage.findMany({
          where: {
            isEditable: true,
          },
          select: {
            id: true,
            isEditable: true,
            targetGroup: {
              select: {
                id: true,
                name: true,
              },
            },
            _count: {
              select: {
                recommendedArticles: true,
              },
            },
          },
          take: 2,
          orderBy: {
            recommendedArticles: {
              _count: 'desc',
            },
          },
        }),

        this.prismaService.articlePackage.findMany({
          where: {
            isEditable: false,
          },
          select: {
            id: true,
            isEditable: true,
            targetGroup: {
              select: {
                id: true,
                name: true,
              },
            },
            _count: {
              select: {
                recommendedArticles: true,
              },
            },
          },
        }),
        //same thing for notification packages
        this.prismaService.notificationPackage.findMany({
          where: {
            id: {
              not: 1,
            },
          },
          select: {
            id: true,
            targetGroup: {
              select: {
                id: true,
                name: true,
              },
            },
            _count: {
              select: {
                notifications: true,
              },
            },
          },
          take: 2,
        }),
        this.prismaService.notificationPackage.findMany({
          where: {
            id: 1,
          },
          select: {
            id: true,
            targetGroup: {
              select: {
                id: true,
                name: true,
              },
            },
            _count: {
              select: {
                notifications: true,
              },
            },
          },
        }),
      ]);

    const NRTListMerge = [...NRTList, ...NRTDefault];

    const articlePackagesMerge = [...articlePackageList, ...defaultArticlePackage];

    const notificationPackagesMerge = [...automaticNotificationPackage, ...defaultNotificationPackage];

    const NRTWithType = {
      nrtHome: { data: NRTListMerge, somkingcessationPlanType: SmokingCessationPlan.NRT },
      articlePackageHome: { data: articlePackagesMerge, somkingcessationPlanType: SmokingCessationPlan.ARTICLES },
      automaticNotificationPackage: {
        data: notificationPackagesMerge.map((np) => ({
          ...np,
          targetGroup: {
            id: np.targetGroup ? np.targetGroup.id : 0,
            name: np.targetGroup ? np.targetGroup.name : 'Default',
          },
        })),
        somkingcessationPlanType: SmokingCessationPlan.AUTOMATIC_NOTIFICATIONS,
      },
    };

    const serializedScPlans = plainToInstance(ScPlansHomeResponse, NRTWithType, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    return serializedScPlans;
  }
