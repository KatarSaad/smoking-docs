import {
  // AppleInstallation,
  // FcmLegacyInstallation,
  Installation,
  NotificationHubsClient,
  // createAppleInstallation,
  createFcmLegacyInstallation,
  createFcmLegacyNotification,
} from '@azure/notification-hubs';
import { Injectable, InternalServerErrorException, Logger, NotFoundException } from '@nestjs/common';
import { EnvironmentService } from 'src/common/services/environment.service';
import { PrismaService } from 'src/common/services/prisma.service';
// import { DevicePlatformEnum } from '../enums/device-platform.enum';
import { v4 as uuidv4 } from 'uuid';
import { GetDeviceInstallationDto } from './dto/get-device-installation.dto';
import { DeviceInstallationDto } from './dto/device-installation.dto';
import { CreateDeviceInstallationDto } from './dto/create-device-installation.dto';
import { DeleteDeviceInstallationDto } from './dto/delete-device-installation.dto';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { UserPlayloadResponse } from '../patients/response/user-playload.response';

@Injectable()
export class NotificationHubService {
  client: NotificationHubsClient;
  private logger = new Logger(NotificationHubService.name);
  constructor(
    private appConfigService: EnvironmentService,
    private prismaService: PrismaService,
  ) {
    this.client = new NotificationHubsClient(this.appConfigService.notificationHubSettings.connectionString, this.appConfigService.notificationHubSettings.hubName);
  }
  // async onModuleInit() {
  //   await this.broadCastSendNotification();
  // }
  async getDeviceInstallation(getDeviceInstallationDto: GetDeviceInstallationDto): Promise<DeviceInstallationDto> {
    try {
      const userDeviceInstallation = await this.prismaService.deviceInstallation.findFirst({
        where: {
          deviceId: getDeviceInstallationDto.deviceId,
          userId: getDeviceInstallationDto.userId,
        },
        select: {
          deviceId: true,
          installationId: true,
          userId: true,
          tags: {
            select: { tagName: true },
          },
        },
      });
      if (!userDeviceInstallation) {
        throw new MultiLanguageException(LanguageErrorEnum.USER_NOT_FOUND);
      }
      const installation = await this.client.getInstallation(userDeviceInstallation.installationId);
      if (installation) await this.sendPushNotification([userDeviceInstallation]);
      return {
        userId: userDeviceInstallation.userId,
        deviceId: userDeviceInstallation.deviceId,
        installationId: userDeviceInstallation.installationId,
      };
    } catch (error) {
      this.logger.error(error);
      throw new NotFoundException('Installation doesnt exist!');
    }
  }

  async getListDeviceInstallation(user: { id: number }): Promise<DeviceInstallationDto[]> {
    try {
      const userDeviceInstallations = await this.prismaService.deviceInstallation.findMany({
        where: {
          userId: user.id,
        },
        select: {
          deviceId: true,
          installationId: true,
          userId: true,
          pushChannel: true,
          tags: {
            select: { tagName: true },
          },
        },
      });
      return userDeviceInstallations;
    } catch (error) {
      this.logger.error(error);
      throw new NotFoundException('Installation doesnt exist!');
    }
  }

  async createDeviceInstallation(createDeviceInstallationDto: CreateDeviceInstallationDto): Promise<DeviceInstallationDto> {
    try {
      const { userId, PNSdeviceToken, deviceId } = createDeviceInstallationDto;
      const installation: Installation = await createFcmLegacyInstallation({
        installationId: uuidv4(),
        pushChannel: PNSdeviceToken,
        userId: deviceId,
      });

      await this.client.createOrUpdateInstallation(installation);

      const userDeviceInstallation = await this.prismaService.deviceInstallation.findFirst({
        where: {
          deviceId: deviceId,
        },
        include: {
          tags: true,
        },
      });

      if (!userDeviceInstallation) {
        await this.prismaService.deviceInstallation.create({
          data: {
            userId,
            installationId: installation.installationId,
            deviceId,
            pushChannel: PNSdeviceToken,
            tags: {
              create: installation.tags
                ? installation.tags.map((v) => ({
                    tagName: v,
                  }))
                : [],
            },
          },
        });
      } else {
        await this.prismaService.deviceInstallation.update({
          where: {
            id: userDeviceInstallation.id,
          },
          data: {
            userId,
            installationId: installation.installationId,
            tags: {
              deleteMany: {
                DeviceInstallationId: userDeviceInstallation.id,
              },
              create: installation.tags
                ? installation.tags.map((v) => ({
                    tagName: v,
                  }))
                : [],
            },
            deviceId,
            pushChannel: PNSdeviceToken,
          },
        });
      }
      return {
        installationId: installation.installationId,
        deviceId: installation.userId,
      };
    } catch (error) {
      this.logger.error(error);
      throw new InternalServerErrorException('Cannot create Installation!');
    }
  }

  async broadCastSendNotification() {
    const messageBody = `{
      "notification": {
        "icon": "ic_stat_mazad",
        "title": "New Message",
        "body": "You've received a new message!"
      },
      "data": {
        "auctionId": "xx_ID",
        "receiptId": "yy_ID",
        "type": "PAYMENT"
      },
    }`;

    const message = createFcmLegacyNotification({
      body: messageBody,
      headers: {
        priority: 'high',
      },
    });

    const result = await this.client.sendNotification(message);

    this.logger.log(`result: ${JSON.stringify(result)}`);

    // Only available in Standard SKU and above
    if (result.notificationId) {
      this.logger.log(`Notification ID: ${result.notificationId}`);
    }
  }

  async sendPushNotification(deviceInstallations: DeviceInstallationDto[], pushNotificationMesage?: string) {
    const notificationBody = pushNotificationMesage ? `"${pushNotificationMesage}"` : '"You\'ve received a new message!"';

    try {
      const messageBody = `{
        "notification": {
          "icon":"ic_stat_mazad",
          "title": "New Message",
          "body": ${notificationBody},
        },
        "data": {
          "auctionId": "xx_ID",
          "receiptId": "yy_ID",
          "type": "PAYMENT"
        },
      }`;

      const message = createFcmLegacyNotification({
        body: messageBody,
        headers: {
          priority: 'high',
        },
      });

      const result = await this.client.sendNotification(message, {
        deviceHandle: deviceInstallations.map((install) => install.pushChannel),
      });

      this.logger.log(`result: ${JSON.stringify(result)}`);

      // Only available in Standard SKU and above
      if (result.notificationId) {
        this.logger.log(`Notification ID: ${result.notificationId}`);
      }
    } catch (error) {
      this.logger.error(error);
    }
  }

  async removeDeviceInstallation(deleteDeviceInstallationDto: DeleteDeviceInstallationDto, user: UserPlayloadResponse): Promise<boolean> {
    try {
      const userDeviceInstallation = await this.prismaService.deviceInstallation.findFirst({
        where: {
          deviceId: deleteDeviceInstallationDto.deviceId,
        },
        include: {
          tags: true,
        },
      });

      if (userDeviceInstallation) {
        if (Number(userDeviceInstallation.userId) != user.id) {
          throw new MultiLanguageException(LanguageErrorEnum.CANNOT_DELETE_THIS_INSTALLATION);
        }
        await this.client.deleteInstallation(userDeviceInstallation.installationId).catch((err) => {
          throw err;
        });
        await this.prismaService.deviceInstallation.delete({
          where: { id: userDeviceInstallation.id },
        });
      }
      return true;
    } catch (error) {
      this.logger.error(error);
      throw new InternalServerErrorException('Cannot remove Installation!');
    }
  }
}
