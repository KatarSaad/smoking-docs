import { Inject, Injectable, forwardRef } from '@nestjs/common';
import { Prisma } from '@prisma/client';
import { SearchWithOptionalPaginationDto } from 'src/common/dto/search-with-optional-pagination.dto';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { RoleEnum } from 'src/common/enums/role.enum';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';
import { PrismaService } from 'src/common/services/prisma.service';
import { sortColumnHSEnum } from 'src/modules/health-specialists/enum/sort-column-hs.enum';
import { FetchHealthSpecialistName } from 'src/modules/health-specialists/response/fetch-health-specialist-by-facility.response';
import { FilterHSDto } from './dto/Filter-hs.dto';
import { PaginatedHSResponseDto } from './response/pagginated-HS-admin.response';
import { CommonService } from 'src/common/services/common.service';
import { CreateHealthSpecialistDto } from 'src/modules/health-specialists/dto/create-health-specialist.dto';
import { BcryptService } from 'src/modules/auth/bcrypt.service';
import { UsersService } from 'src/modules/users/users.service';
import { OnboardingService } from 'src/modules/onboarding/onboarding.service';
import { UpdateHealthSpecialistDto } from 'src/modules/health-specialists/dto/update-health-specialist.dto';
import { FetchHealthSpecialistByIdResponse } from 'src/modules/users/response/fetch-health-specialist-by-id.response';
import { SettingService } from '../settings/settings.service';
import { SettingsData } from 'src/common/consts/settings-data';
import { UserAdminResponse } from 'src/modules/admins/response/fetch-admin.response';

@Injectable()
export class HealthSpecialistService {
  constructor(
    private readonly settingsService: SettingService,
    private readonly prismaService: PrismaService,
    private readonly commonService: CommonService,
    private readonly bcryptService: BcryptService,
    @Inject(forwardRef(() => UsersService)) private readonly usersService: UsersService,
    @Inject(forwardRef(() => OnboardingService)) private readonly onboardingService: OnboardingService,
  ) {}

  async findAll(facilities: number[], filterHSDto: FilterHSDto): Promise<PaginatedHSResponseDto> {
    const { sizePerPage, currentPage, name, speciality, facility, sortColumn, isSorted } = filterHSDto;
    const sortDirection = isSorted ? 'desc' : 'asc';

    const fullTextSearch = name ? this.commonService.addLeadingPlusAndTrailingWildcard(name) : undefined;

    const isHealthSpecialistSort = [sortColumnHSEnum.JOININGDATE, sortColumnHSEnum.LICENSENUMBER, sortColumnHSEnum.FACILITYNAME].includes(sortColumn);
    const where: Prisma.AdminWhereInput = {
      ...(name && {
        user: {
          OR: [
            {
              personalInformation: {
                firstName: {
                  search: fullTextSearch,
                },
                lastName: {
                  search: fullTextSearch,
                },
              },
            },
            {
              qid: {
                startsWith: name,
              },
            },
          ],
        },
      }),
      healthSpecialist: {
        speciality: speciality ? { equals: speciality } : undefined,
        facility: facility ? { name: facility } : undefined,
        ...(facilities && {
          facility: {
            id: {
              in: facilities,
            },
          },
        }),
      },
      role: {
        name: RoleEnum.HEALTH_SPECIALIST,
      },
    };
    const [healthSpecialists, paginationData] = await this.prismaService
      .extendPagination()
      .admin.paginate({
        select: {
          user: {
            select: {
              personalInformation: {
                select: {
                  firstName: true,
                  lastName: true,
                },
              },
              id: true,
              qid: true,
            },
          },
          healthSpecialist: {
            select: {
              joiningDate: true,
              licenseNumber: true,
              facility: {
                select: {
                  id: true,
                  name: true,
                  _count: {
                    select: {
                      patients: true,
                    },
                  },
                  systemAdmin: {
                    select: {
                      user: {
                        select: {
                          id: true,
                          personalInformation: {
                            select: {
                              firstName: true,
                              lastName: true,
                            },
                          },
                        },
                      },
                    },
                  },
                },
              },
            },
          },
        },

        where,
        orderBy: {
          ...(sortColumn === sortColumnHSEnum.QID && { user: { qid: sortDirection } }),
          ...(sortColumn === sortColumnHSEnum.FULLNAME && {
              user: {
                personalInformation: {
                  firstName: sortDirection,
                },
              },
            } && {
              user: {
                personalInformation: {
                  lastName: sortDirection,
                },
              },
            }),
          ...(isHealthSpecialistSort && {
            healthSpecialist: {
              ...(sortColumn === sortColumnHSEnum.JOININGDATE && { joiningDate: sortDirection }),
              ...(sortColumn === sortColumnHSEnum.LICENSENUMBER && { licenseNumber: sortDirection }),
              ...(sortColumn === sortColumnHSEnum.SUPERVISOR && {
                facility: {
                  name: sortDirection,
                },
              }),
              ...(sortColumn === sortColumnHSEnum.MANAGINPATIENTS && { managingPatients: { _count: sortDirection } }),
            },
          }),
        },
      })
      .withPages({
        includePageCount: true,
        page: currentPage,
        limit: sizePerPage,
      });

    return { data: healthSpecialists, totalItems: paginationData.totalCount };
  }

  async addHealthSpecialist(createHealthSpecialistDto: CreateHealthSpecialistDto): Promise<UserAdminResponse> {
    const {
      email,
      location,
      joiningDate,
      licenseNumber,
      speciality,
      personalInformation: { firstName, gender, lastName, dateOfBirth },
      phone,
      qid,
      facilityId,
    } = createHealthSpecialistDto;

    const HashedPassword = await this.bcryptService.randomPassword(5);

    await this.usersService.isQidNotExist(qid, null);
    await this.onboardingService.isEmailNotExist(email, null);
    await this.isLicenseNumberNotExist(licenseNumber, null);

    return await this.prismaService.$transaction(async (prisma) => {
      const user = await prisma.user.create({
        data: {
          qid,
          password: HashedPassword,
          location,
          personalInformation: {
            create: {
              firstName,
              lastName,
              email,
              gender,
              dateOfBirth,
              phoneNumber: phone,
            },
          },
          admin: {
            create: {
              role: {
                connect: {
                  name: RoleEnum.HEALTH_SPECIALIST,
                },
              },
              healthSpecialist: {
                create: {
                  joiningDate,
                  licenseNumber,
                  speciality,
                  facility: {
                    connect: { id: facilityId },
                  },
                },
              },
            },
          },
        },
      });

      // Fetch the user with specific select options
      const userWithSelect = await this.prismaService.user.findUnique({
        where: { id: user.id },
        select: {
          admin: {
            select: {
              id: true,
              department: true,
              role: {
                select: {
                  id: true,
                  isActive: true,
                  name: true,
                  restriction: true,
                },
              },
            },
          },
          personalInformation: {
            select: {
              firstName: true,
              lastName: true,
            },
          },
          qid: true,
          id: true,
        },
      });

      const settingsData = SettingsData.healthSpecialist;
      await this.settingsService.assignSettings(settingsData, user.id, prisma);

      return userWithSelect;
    });
  }

  async healthSpecialistIdByUserId(userId: number): Promise<bigint> {
    const user = await this.prismaService.user.findUnique({
      where: {
        id: userId,
      },
      select: {
        admin: {
          select: {
            healthSpecialist: {
              select: { id: true },
            },
          },
        },
      },
    });
    if (!user) throw new MultiLanguageException(LanguageErrorEnum.USER_NOT_FOUND);

    if (!user?.admin?.healthSpecialist) throw new MultiLanguageException(LanguageErrorEnum.NOT_HEALTH_SPECIALIST);

    return user.admin.healthSpecialist.id;
  }

  async healthSpecialistsNames(facilities: number[], searchWithOptionalPaginationDto: SearchWithOptionalPaginationDto): Promise<FetchHealthSpecialistName[]> {
    const { search, currentPage, sizePerPage } = searchWithOptionalPaginationDto;
    const isPagination = currentPage !== undefined && sizePerPage !== undefined;
    const take = isPagination ? sizePerPage : undefined;
    const skip = isPagination ? (currentPage || 0) * sizePerPage : undefined;

    return await this.prismaService.user.findMany({
      select: {
        id: true,
        personalInformation: {
          select: {
            firstName: true,
            lastName: true,
          },
        },
      },
      where: {
        admin: {
          healthSpecialist: {
            facilityId: {
              in: facilities,
            },
          },
        },

        ...(search && {
          OR: [
            {
              personalInformation: {
                OR: [
                  {
                    firstName: {
                      search,
                    },
                    lastName: {
                      search,
                    },
                  },
                  {
                    firstName: { startsWith: search },
                  },
                  {
                    lastName: { startsWith: search },
                  },
                ],
              },
            },
            { qid: search },
          ],
        }),
      },
      take,
      skip,
    });
  }

  async healthSpecialistsByFacility(facilityId: number, searchWithOptionalPaginationDto: SearchWithOptionalPaginationDto): Promise<FetchHealthSpecialistName[]> {
    const { search, currentPage, sizePerPage } = searchWithOptionalPaginationDto;
    const skip = currentPage * sizePerPage;

    return await this.prismaService.user.findMany({
      select: {
        id: true,
        personalInformation: {
          select: {
            firstName: true,
            lastName: true,
          },
        },
      },
      where: {
        admin: {
          healthSpecialist: {
            facilityId,
          },
        },
        ...(search && {
          OR: [
            {
              personalInformation: {
                OR: [
                  {
                    firstName: {
                      search,
                    },
                    lastName: {
                      search,
                    },
                  },
                  {
                    firstName: { startsWith: search },
                  },
                  {
                    lastName: { startsWith: search },
                  },
                ],
              },
            },
            { qid: search },
          ],
        }),
      },
      take: sizePerPage,
      skip: currentPage && sizePerPage && skip,
    });
  }

  async updateHealthSpecialistByUserId(userId: number, updateHealthSpecialistDto: UpdateHealthSpecialistDto): Promise<FetchHealthSpecialistByIdResponse> {
    const { email, facilityId, joiningDate, licenseNumber, location, personalInformation, phone, speciality } = updateHealthSpecialistDto;

    await this.healthSpecialistIdByUserId(userId);

    if (email) await this.onboardingService.isEmailNotExist(email, userId);

    await this.isLicenseNumberNotExist(licenseNumber, userId);

    return await this.prismaService.user.update({
      where: {
        id: userId,
      },
      data: {
        personalInformation: {
          update: {
            ...personalInformation,
            email,
            phoneNumber: phone,
          },
        },
        location,
        admin: {
          update: {
            data: {
              healthSpecialist: {
                update: {
                  facilityId,
                  joiningDate,
                  licenseNumber,
                  speciality,
                },
              },
            },
          },
        },
      },
      select: {
        id: true,
        qid: true,
        location: true,
        personalInformation: {
          select: {
            email: true,
            firstName: true,
            lastName: true,
            gender: true,
            dateOfBirth: true,
            phoneNumber: true,
          },
        },
        admin: {
          select: {
            healthSpecialist: {
              select: {
                speciality: true,
                licenseNumber: true,
                joiningDate: true,
                facility: {
                  select: { id: true, name: true },
                },
              },
            },
          },
        },
      },
    });
  }

  async isLicenseNumberNotExist(licenseNumber: string, userId: number): Promise<boolean> {
    const healthSpecialistlicenseNumber = await this.prismaService.healthSpecialist.findUnique({
      where: {
        licenseNumber,
        ...(userId && {
          admin: {
            user: {
              id: {
                not: userId,
              },
            },
          },
        }),
      },
      select: { id: true },
    });

    if (healthSpecialistlicenseNumber) throw new MultiLanguageException(LanguageErrorEnum.LICENSE_NUMBER_EXISTS);
    return true;
  }
}
