import { Server, Socket } from 'socket.io';
import { UserPlayloadResponse } from '../patients/response/user-playload.response';
import { GroupChatsService } from 'src/modules/group-chats/group-chats.service';
import { Injectable } from '@nestjs/common';
import { MessagesService } from 'src/modules/messages/messages.service';
import { HealthSpecialistJoinRoomDto } from 'src/modules/messages/dto/health-specialist-join-room.dto';
import { CommonService } from 'src/common/services/common.service';
import { PatientJoinRoomDto } from 'src/modules/messages/dto/patient-join-room.dto';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { MessageGroupChatSocketResponse } from 'src/modules/messages/response/message-group-chat.response';
import { SocketEventEnum } from 'src/common/enums/socket-event.enum';
import { DeleteMessageDto } from 'src/modules/private-chat-messages/dto/delete-message.dto';
import { SocketMessageGroupChatDto } from 'src/modules/messages/dto/socket-message-group-chat.dto';
import { JoinHealthSpecialistGroupChatRoomsDto } from 'src/modules/messages/response/health-specialist-join-group-chat-room.dto';
import { SuccessResponse } from 'src/common/responses/socket.response';

@Injectable()
export class MessagesListener {
  constructor(
    private groupChatsService: GroupChatsService,
    private messagesService: MessagesService,
    private commonService: CommonService,
  ) {}

  async sendGroupChatMessage(
    socketMessageGroupChatDto: SocketMessageGroupChatDto,
    client: Socket,
    user: UserPlayloadResponse,
    server: Server,
  ): Promise<MessageGroupChatSocketResponse> {
    const { groupChatId, message } = socketMessageGroupChatDto;
    const room = 'group_chat_' + String(groupChatId);
    const groupChatsRoom = 'group_chats_' + String(groupChatId);

    const addMessageResponse = await this.messagesService.addMessageToGroup(user, socketMessageGroupChatDto); //save message in db
    const { sender } = addMessageResponse;
    const isHealthSpecialist = user.isHS;

    server.to(groupChatsRoom).emit(SocketEventEnum.MESSAGE_GROUP_CHATS, this.commonService.objectParser({ groupChatId, message, senderId: user.id, hasFile: false }));

    // broadcast message to sockets connected to the group room(except the sender)
    client.broadcast
      .to(room)
      .timeout(5000)
      .emit(SocketEventEnum.MESSAGE_GROUP_CHAT, this.commonService.objectParser({ ...addMessageResponse, sender: { ...sender, isHealthSpecialist } }));

    return { success: true, messageId: Number(addMessageResponse.id) };
  }

  async participantJoinGroupChatRoom(payload: PatientJoinRoomDto, user: UserPlayloadResponse, server: Server) {
    const { socketId } = payload;

    const groupChatId = await this.groupChatsService.userGroupChatId(user.id);

    if (!groupChatId) throw new MultiLanguageException(LanguageErrorEnum.WITHOUT_GROUP_CHAT);

    server.in(socketId).socketsJoin('group_chat_' + String(groupChatId));

    return { success: true };
  }

  async adminJoinGroupChatRoom(payload: HealthSpecialistJoinRoomDto, user: UserPlayloadResponse, server: Server) {
    const { groupChatId, socketId } = payload;
    const room = 'group_chat_' + String(groupChatId);

    if (user.isHS) await this.groupChatsService.isHealthSpecialistInGroup(Number(groupChatId), user.healthSpecialistId);

    server.in(socketId).socketsJoin(room);

    return { success: true };
  }

  async adminJoinGroupChatRooms(payload: JoinHealthSpecialistGroupChatRoomsDto, user: UserPlayloadResponse, server: Server): Promise<SuccessResponse> {
    const { socketId, groupChatIds } = payload;
    const groupChatIdsStrings = groupChatIds.map((id) => `group_chats_${id}`);

    if (user.isHS) await this.messagesService.isHealthSpecialistInGroupChats(groupChatIds, user.healthSpecialistId);
    server.in(socketId).socketsJoin(groupChatIdsStrings);

    return { success: true };
  }

  async deleteGroupChatMessage(deleteMessageDto: DeleteMessageDto, client: Socket, user: UserPlayloadResponse): Promise<MessageGroupChatSocketResponse> {
    const { id } = deleteMessageDto;

    const deletedMessage = await this.messagesService.deleteMessage(user, id);

    const deletedMessageId = Number(deletedMessage.id);

    client.broadcast.timeout(5000).emit(SocketEventEnum.DELETE_MESSAGE_GROUP_CHAT, { messageId: deletedMessageId });

    return { success: true, messageId: deletedMessageId };
  }
}
