import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';
import { Injectable } from '@nestjs/common';
import { PrismaService } from 'src/common/services/prisma.service';
import { GroupChatsService } from 'src/modules/group-chats/group-chats.service';
import { MessageGroupChatDto } from 'src/modules/messages/dto/message-group-chat.dto';
import { MessageResponse } from 'src/modules/messages/response/message.response';
import { CommonService } from 'src/common/services/common.service';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { SimpleMessageResponse } from 'src/modules/messages/response/simple-message.response';
import { SocketService } from 'src/modules/socket/socket.service';
import { SocketEventEnum } from 'src/common/enums/socket-event.enum';
import { FileService } from 'src/common/services/file.service';
import { FileResponse } from 'src/modules/private-chat-messages/response/file.response';
import { SocketMessageGroupChatDto } from 'src/modules/messages/dto/socket-message-group-chat.dto';
import { UsersService } from 'src/modules/users/users.service';
import { NotificationService } from 'src/modules/notification/notification.service';
import { IdDto } from 'src/common/dto/id.dto';
import { SettingOptionNameEnum } from '@prisma/client';

@Injectable()
export class MessagesService {
  constructor(
    private readonly prismaService: PrismaService,
    private readonly groupChatsService: GroupChatsService,
    private commonService: CommonService,
    private socketService: SocketService,
    private fileService: FileService,
    private usersService: UsersService,
    private notificationService: NotificationService,
  ) {}

  async addMessageToGroupChatHttp(user: UserPlayloadResponse, messageGroupChatDto: MessageGroupChatDto, file: Express.Multer.File): Promise<MessageResponse> {
    const { groupChatId, message, socketId } = messageGroupChatDto;
    let fileData: FileResponse = null;
    const room = String(groupChatId);

    if (file) fileData = await this.fileService.uploadFile(file);

    const addMessageResponse = await this.addMessageToGroup(user, messageGroupChatDto, fileData); //save message in db
    const { sender } = addMessageResponse;

    this.socketService.socket
      .to(`group_chats_${room}`)
      .emit(SocketEventEnum.MESSAGE_GROUP_CHATS, this.commonService.objectParser({ groupChatId, message, hasFile: !!fileData, senderId: user.id }));

    this.socketService.socket
      .to(`group_chat_${room}`)
      .except(socketId) //make this dynamic
      .emit(SocketEventEnum.MESSAGE_GROUP_CHAT, this.commonService.objectParser({ ...addMessageResponse, sender: { ...sender, isHealthSpecialist: true } }));

    return addMessageResponse;
  }

  async deleteMessage(user: UserPlayloadResponse, id: number): Promise<SimpleMessageResponse> {
    await this.isMessageOwner(id, user.id);

    return await this.prismaService.message.delete({
      where: { id },
      select: {
        id: true,
        message: true,
        createdAt: true,
      },
    });
  }

  async addMessageToGroup(user: UserPlayloadResponse, socketMessageGroupChatDto: SocketMessageGroupChatDto, fileData?: FileResponse): Promise<MessageResponse> {
    const { groupChatId, message, replyId, mentionedUserIds } = socketMessageGroupChatDto;
    const isMentionedUsers = mentionedUserIds?.length > 0;
    const userIds = isMentionedUsers && mentionedUserIds.map((user: IdDto) => Number(user.id));
    const userIdsForNotification =
      isMentionedUsers &&
      mentionedUserIds.map((user: IdDto) => {
        return {
          id: user.id,
        };
      });

    if (user.isPatient && message) {
      const isUrlNotDetected = this.commonService.isUrlNotDetected(message);
      if (isUrlNotDetected) throw new MultiLanguageException(LanguageErrorEnum.URL_NOT_ALLOWED);
    }

    if (user.isHS) await this.groupChatsService.isHealthSpecialistInGroup(groupChatId, user.healthSpecialistId);
    else await this.groupChatsService.isParticipantExists(groupChatId, user.id);

    if (replyId) await this.findMessageById(replyId);

    if (isMentionedUsers) await this.usersService.isUsersByIdExist(userIds);

    const sentMessage = await this.prismaService.message.create({
      data: {
        message,
        groupChatId,
        senderId: user.id,
        replyId,
        ...(fileData && { file: { create: fileData } }),
        ...(isMentionedUsers && { mentionedUsers: { connect: mentionedUserIds } }),
      },
      select: {
        id: true,
        message: true,
        sender: {
          select: {
            id: true,
            personalInformation: { select: { firstName: true, lastName: true } },
          },
        },
        reply: {
          select: {
            id: true,
            file: {
              select: {
                blobName: true,
                extension: true,
                name: true,
              },
            },
            mentionedUsers: {
              select: {
                id: true,
                personalInformation: { select: { firstName: true, lastName: true } },
              },
            },
            sender: {
              select: {
                id: true,
                personalInformation: { select: { firstName: true, lastName: true } },
              },
            },
            message: true,
            createdAt: true,
          },
        },
        file: {
          select: {
            name: true,
            blobName: true,
            extension: true,
            size: true,
          },
        },
        mentionedUsers: {
          select: {
            id: true,
            personalInformation: {
              select: { id: true, firstName: true, lastName: true },
            },
          },
        },
        createdAt: true,
      },
    });
    const senderName = `${sentMessage.sender.personalInformation.firstName} ${sentMessage.sender.personalInformation.lastName}`;

    const pushNotificationMessage = `${sentMessage.sender.personalInformation.firstName} ${sentMessage.sender.personalInformation.lastName} sent you a message!`;
    if (isMentionedUsers) {
      this.groupChatMentionedNotification(user, groupChatId, SettingOptionNameEnum.MENTION_IN_PRIVATE_GROUP, senderName, userIdsForNotification);
      this.notificationService.sendPushNotif(userIdsForNotification, SettingOptionNameEnum.MENTION_IN_PRIVATE_GROUP, pushNotificationMessage);
    }

    return sentMessage;
  }

  async findMessageById(id: number): Promise<boolean> {
    const message = await this.prismaService.message.findUnique({ where: { id }, select: { id: true } });

    if (!message) throw new MultiLanguageException(LanguageErrorEnum.MESSAGE_NOT_FOUND);
    return true;
  }

  async isMessageOwner(messageId: number, userId: number): Promise<boolean> {
    await this.findMessageById(messageId);

    const message = await this.prismaService.message.findUnique({ where: { id: messageId, senderId: userId } });

    if (!message) throw new MultiLanguageException(LanguageErrorEnum.MESSAGE_NOT_YOURS);
    return true;
  }

  async isHealthSpecialistInGroupChats(groupChatIds: number[], healthSpecialistId: number): Promise<boolean> {
    const groupChat = await this.prismaService.groupChat.aggregate({
      where: {
        id: { in: groupChatIds },
        healthSpecialistId,
      },
      _count: {
        _all: true,
      },
    });

    if (groupChat._count._all < groupChatIds.length) throw new MultiLanguageException(LanguageErrorEnum.HEALTH_SPECIALIST_NOT_IN_GROUP_CHAT);

    return true;
  }

  async groupChatMentionedNotification(
    user: UserPlayloadResponse,
    groupChatId: number,
    settingOptionNameEnum: SettingOptionNameEnum,
    senderName: string,
    userIdsForNotification: IdDto[],
  ): Promise<boolean> {
    const notificationMessage = `${senderName} sent you a message!`;
    const notificationMessageAr = `${senderName} أرسل لك رسالة!`;

    const createNotificationDto = {
      messageAr: notificationMessageAr,
      messageEn: notificationMessage,
      titleAr: 'رسالة جديدة!',
      titleEn: 'New message!',
      deliveryDate: new Date(),
      notificationType: 'GroupChat',
      isCritical: false,
    };

    const healthSpecialistUserId = await this.groupChatsService.groupChatHealthSpecialistUserId(groupChatId);

    const isHealthSpecialistMentioned = userIdsForNotification.map((user) => Number(user.id)).includes(healthSpecialistUserId);

    if (isHealthSpecialistMentioned)
      this.notificationService.createAdminNotification(user, createNotificationDto, BigInt(healthSpecialistUserId), groupChatId, settingOptionNameEnum);
    return true;
  }
}
