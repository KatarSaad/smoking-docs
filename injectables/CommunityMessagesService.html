import { CursorDirectionEnum } from 'src/common/enums/cursor-direction.enum';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';
import { Inject, Injectable, forwardRef } from '@nestjs/common';
import { PrismaService } from 'src/common/services/prisma.service';
import { CommonService } from 'src/common/services/common.service';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { CommunityMessageDto } from './dto/community-messages.dto';
import { CommunityMessageResponse } from './response/community-message.response';
import { PaginationCursorCommunityMessages } from 'src/common/responses/paggination-cursor.response';
import { CommunityMessages, SettingOptionNameEnum } from '@prisma/client';
import { BlockedUsersService } from 'src/modules/blocked-users/blocked-users.service';
import { SearchWithCursorPaginationDto } from 'src/common/dto/search-with-cursor-pagination.dto';

@Injectable()
export class CommunityMessagesService {
  constructor(
    private readonly prismaService: PrismaService,
    private commonService: CommonService,
    @Inject(forwardRef(() => BlockedUsersService))
    private blockedUsersService: BlockedUsersService,
    @Inject(forwardRef(() => UsersService))
    private usersService: UsersService,
    @Inject(forwardRef(() => NotificationService))
    private notificationService: NotificationService,
    @Inject(forwardRef(() => PatientService))
    private patientService: PatientService,

  async addMessageToCommunity(user: UserPlayloadResponse, communityMessageDto: CommunityMessageDto): Promise<CommunityMessageResponse> {
    const { message, replyId, mentionedUserIds } = communityMessageDto;

    const isMentionedUsers = mentionedUserIds?.length > 0;
    const userIds = isMentionedUsers && mentionedUserIds.map((user) => Number(user.id));

    const userIdsForNotification =
      isMentionedUsers &&
      mentionedUserIds.map((user: IdDto) => {
        return {
          id: user.id,
        };
      });

    await this.patientService.isUserHasUsername(user.patientId);

    await this.blockedUsersService.isUserNotBlocked(user);

    const isUrlNotDetected = this.commonService.isUrlNotDetected(message);
    if (isUrlNotDetected) throw new MultiLanguageException(LanguageErrorEnum.URL_NOT_ALLOWED);

    if (replyId) await this.findMessageById(replyId);

    if (isMentionedUsers) await this.usersService.isUsersByIdExist(userIds);

    const savedMessage = await this.prismaService.communityMessages.create({
      data: {
        message,
        senderId: user.id,
        replyId,
        ...(isMentionedUsers && { mentionedUsers: { connect: mentionedUserIds } }),
      },
      select: {
        id: true,
        message: true,
        sender: {
          select: {
            id: true,
            patient: {
              select: {
                username: true,
              },
            },
          },
        },
        reply: {
          select: {
            id: true,
            message: true,
            mentionedUsers: {
              select: {
                id: true,
                patient: { select: { username: true } },
              },
            },
            createdAt: true,
            sender: {
              select: {
                id: true,
                patient: {
                  select: { username: true },
                },
              },
            },
          },
        },
        mentionedUsers: {
          select: {
            id: true,
            patient: { select: { username: true } },
          },
        },
        createdAt: true,
      },
    });

    const pushNotificationMessage = `${savedMessage.sender.patient.username} sent you a message!`;
    if (isMentionedUsers) this.notificationService.sendPushNotif(userIdsForNotification, SettingOptionNameEnum.MENTION_IN_COMMUNITY_CHAT, pushNotificationMessage);

    return {
      id: savedMessage.id,
      message: savedMessage.message,
      createdAt: savedMessage.createdAt,
      sender: {
        id: savedMessage.sender.id,
        username: savedMessage.sender.patient.username,
      },
      reply: replyId
        ? {
            id: savedMessage.reply.id,
            message: savedMessage.reply.message,
            createdAt: savedMessage.reply.createdAt,
            mentionedUsers: savedMessage.reply.mentionedUsers.map((mentionedUser) => {
              return { id: mentionedUser.id, username: mentionedUser.patient.username };
            }),
            sender: {
              id: savedMessage.reply.sender.id,
              username: savedMessage.reply.sender.patient.username,
            },
          }
        : null,
      mentionedUsers: savedMessage.mentionedUsers.map((mentionedUser) => {
        return { id: mentionedUser.id, username: mentionedUser.patient.username };
      }),
    };
  }

  async findMessageById(id: number): Promise<CommunityMessages> {
    const message = await this.prismaService.communityMessages.findUnique({ where: { id } });

    if (!message) throw new MultiLanguageException(LanguageErrorEnum.MESSAGE_NOT_FOUND);
    return message;
  }

  async isMessageOwner(messageId: number, userId: number): Promise<boolean> {
    await this.findMessageById(messageId);

    const message = await this.prismaService.communityMessages.findFirst({ where: { id: messageId, senderId: userId } });

    if (!message) throw new MultiLanguageException(LanguageErrorEnum.MESSAGE_NOT_YOURS);
    return true;
  }

  async fetchCommmunityChat(user: UserPlayloadResponse, searchWithCursorPaginationDto: SearchWithCursorPaginationDto): Promise<PaginationCursorCommunityMessages> {
    const { oldCursorId, sizePerPage, search, direction } = searchWithCursorPaginationDto;
    const fullTextSearch = search ? this.commonService.addLeadingPlusAndTrailingWildcard(search) : undefined;

    const dir = direction === CursorDirectionEnum.AFTER ? { after: oldCursorId ? `${oldCursorId}` : null } : { before: oldCursorId ? `${oldCursorId}` : null };

    const [messages, messagePagination] = await this.prismaService
      .extendPagination()
      .communityMessages.paginate({
        where: {
          message: { search: fullTextSearch },
          deletedAt: null,
        },
        select: {
          id: true,
          message: true,
          mentionedUsers: {
            select: {
              id: true,
              personalInformation: { select: { firstName: true, lastName: true } },
              patient: { select: { username: true } },
            },
          },
          createdAt: true,
          sender: {
            select: {
              id: true,
              personalInformation: { select: { firstName: true, lastName: true } },
              patient: {
                select: { username: true },
              },
            },
          },
          reply: {
            select: {
              id: true,
              message: true,
              createdAt: true,
              mentionedUsers: {
                select: {
                  id: true,
                  personalInformation: { select: { firstName: true, lastName: true } },
                  patient: { select: { username: true } },
                },
              },
              sender: {
                select: {
                  id: true,
                  personalInformation: { select: { firstName: true, lastName: true } },
                  patient: {
                    select: { username: true },
                  },
                },
              },
            },
          },
        },
        orderBy: {
          createdAt: 'desc',
        },
      })
      .withCursor({
        limit: sizePerPage,
        ...dir,
        getCursor({ id }) {
          return id.toString();
        },
      });

    //adding isSender field to the response as boolean =>isSender:true if the authed user is the sender
    const messagesWithIsSenderCheck: CommunityMessageResponse[] = messages.map((message) => {
      const senderId = Number(message.sender.id);
      const isSender = senderId === user.id;
      return {
        id: message.id,
        createdAt: message.createdAt,
        message: message.message,
        sender: {
          id: message.sender.id,
          username: message.sender.patient.username,
          firstName: message.sender.personalInformation.firstName,
          lastName: message.sender.personalInformation.lastName,
        },
        isSender,
        reply: message.reply
          ? {
              id: message.reply.id,
              createdAt: message.reply.createdAt,
              mentionedUsers: message.reply.mentionedUsers.map((mentionedUser) => {
                return {
                  id: mentionedUser.id,
                  username: mentionedUser.patient.username,
                  firstName: mentionedUser.personalInformation.firstName,
                  lastName: mentionedUser.personalInformation.lastName,
                };
              }),
              message: message.reply.message,
              sender: {
                id: message.reply.sender.id,
                username: message.reply.sender.patient.username,
                firstName: message.reply.sender.personalInformation.firstName,
                lastName: message.reply.sender.personalInformation.lastName,
              },
              isSender: Number(message.reply.sender.id) === user.id,
            }
          : null,
        mentionedUsers: message.mentionedUsers.map((mentionedUser) => {
          return {
            id: mentionedUser.id,
            username: mentionedUser.patient.username,
            firstName: mentionedUser.personalInformation.firstName,
            lastName: mentionedUser.personalInformation.lastName,
          };
        }),
      };
    });

    const { hasNextPage, hasPreviousPage, endCursor, startCursor } = messagePagination;

    return { data: messagesWithIsSenderCheck, hasNextPage, hasPreviousPage, endCursor: Number(endCursor), startCursor: Number(startCursor) };
  }

  async deleteMessage(user: UserPlayloadResponse, id: number): Promise<CommunityMessageResponse> {
    await this.isMessageOwner(id, user.id);
    return await this.softDeleteMessage(id);
  }

  async softDeleteMessage(id: number): Promise<CommunityMessageResponse> {
    return await this.prismaService.communityMessages.update({
      where: { id },
      data: { deletedAt: new Date() },
      select: {
        id: true,
        message: true,
        createdAt: true,
      },
    });
  }
