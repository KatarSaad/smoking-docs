import { Inject, Injectable, forwardRef } from '@nestjs/common';

  async getAllGroups(fetchGroupFilterDto: FetchGroupFilterDto): Promise<PaginationGroups> {
    const { name, currentPage, sizePerPage, addictionLevel, agesRange, cigarretesRange, gender, massBodies, relapsesRange } = fetchGroupFilterDto;

    const isRelapses = relapsesRange?.length > 0;
    const isAges = agesRange?.length > 0;
    const isCigarretes = cigarretesRange?.length > 0;

    //we r reformating the agesRange to use it inside the prisma where
    const agesRangeMapping =
      isAges &&
      agesRange.map((ageRange) => {
        return {
          ageMin: {
            lte: ageRange.max,
          },
          ageMax: {
            gte: ageRange.min,
          },
        };
      });

    //we r reformating the relapsesRange to use it inside the prisma where
    const relapsesRangeMapping =
      isRelapses &&
      relapsesRange.map((relapseRange) => {
        return {
          relapsesMin: {
            lte: relapseRange.max,
          },
          relapsesMax: {
            gte: relapseRange.min,
          },
        };
      });

    //we r reformating the cigarretesRange to use it inside the prisma where
    const cigarretesRangeMapping =
      isCigarretes &&
      cigarretesRange.map((cigarreteRange) => {
        return {
          cigarretesMin: {
            lte: cigarreteRange.max,
          },
          cigarretesMax: {
            gte: cigarreteRange.min,
          },
        };
      });

    const where: Prisma.GroupWhereInput = {
      name: {
        startsWith: name,
      },
      condition: {
        addictionLevel,
        massBody: {
          in: massBodies,
        },
        gender: gender,
        AND: [
          // filtring the age
          {
            OR: isAges ? agesRangeMapping : undefined,
          },
          //filtring the relapses
          {
            OR: isRelapses ? relapsesRangeMapping : undefined,
          },

          //filtring the cigarretes
          { OR: isCigarretes ? cigarretesRangeMapping : undefined },
        ],
      },
    };

    const [groups, paginationData] = await this.prismaService
      .extendPagination()
      .group.paginate({
        where,
        orderBy: {
          id: 'desc',
        },
        select: {
          id: true,
          name: true,
          condition: {
            include: {
              riskFactor: true,
            },
          },
          _count: {
            select: { patients: true },
          },
        },
      })
      .withPages({
        includePageCount: true,
        page: currentPage,
        limit: sizePerPage,
      });
    const totalPatientCount = groups.reduce((total, group) => total + group._count.patients, 0);

    // edit the response to add patients count

    return { data: groups, totalItems: paginationData.totalCount, totalPatient: totalPatientCount };
  }

  async addGroup(addGroupDto: AddGroupDto, user: UserPlayloadResponse) {
    const {
      name,
      condition: { addictionLevel, ageMax, ageMin, cigarretesMax, cigarretesMin, gender, massBody, relapsesMax, relapsesMin },
    } = addGroupDto;

    //check if name exists in the current facility
    const isGroupNameExists = !!(await this.prismaService.group.findFirst({
      where: {
        name: addGroupDto.name,
      },
    }));

    if (isGroupNameExists) throw new MultiLanguageException(LanguageErrorEnum.GROUP_NAME_EXISTS);

    return await this.prismaService.group.create({
      data: {
        name,
        condition: {
          create: {
            addictionLevel,
            gender,
            massBody,
            ageMax,
            ageMin,
            relapsesMax,
            relapsesMin,
            cigarretesMax,
            cigarretesMin,
          },
        },
        creator: {
          connect: {
            id: user.adminId,
          },
        },
      },
      select: {
        id: true,
        name: true,
        condition: {
          select: {
            addictionLevel: true,
            ageMax: true,
            ageMin: true,
            cigarretesMax: true,
            cigarretesMin: true,
            gender: true,
            massBody: true,
            relapsesMax: true,
            relapsesMin: true,
          },
        },
      },
    });
  }

  async fetchGroupsNamesOnly(paginationOptionalDto: PaginationOptionalDto): Promise<PaginationGroupNamesOnlyResponse> {
    const { currentPage, sizePerPage } = paginationOptionalDto;

    const isPagination = currentPage !== undefined && sizePerPage !== undefined;
    const take = isPagination ? sizePerPage : undefined;
    const skip = isPagination ? (currentPage || 0) * sizePerPage : undefined;

    const [groups, totalItems] = await this.prismaService.$transaction(async (tx) => {
      const result = await tx.group.findMany({
        select: {
          id: true,
          name: true,
        },
        take,
        skip,
      });
      let count = null;
      if (isPagination)
        count = await tx.group.aggregate({
          _count: {
            _all: true,
          },
        });
      return [result, count];
    });

    return { data: groups, totalItems: totalItems?._count?._all };
  }

  async deleteGroup(id: number) {
    const group = await this.prismaService.group.findUnique({
      where: { id },
    });

    if (!group) throw new MultiLanguageException(LanguageErrorEnum.GROUP_NOT_FOUND);

    return await this.prismaService.group.delete({
      where: {
        id,
      },
      select: {
        id: true,
        name: true,
        condition: {
          select: {
            addictionLevel: true,
            ageMax: true,
            ageMin: true,
            cigarretesMax: true,
            cigarretesMin: true,
            gender: true,
            massBody: true,
            relapsesMax: true,
            relapsesMin: true,
          },
        },
      },
    });
  }

  async fetchGroupById(id: number): Promise<GroupResponse> {
    const group = await this.prismaService.group.findFirst({
      where: { id },
      include: {
        condition: {
          include: {
            riskFactor: true,
          },
        },
      },
    });
    if (!group) throw new MultiLanguageException(LanguageErrorEnum.GROUP_NOT_FOUND);
    return group;
  }

  async fetchPatientGroups(patientId: number): Promise<GroupNameOnlyResponse[]> {
    const fetchPatientGroupsRedisKey = `fetchPatientGroups/${patientId}`;
    const cacheResults = await this.redisService.getResponseKey(fetchPatientGroupsRedisKey);
    if (cacheResults) return cacheResults;

    const groups = await this.prismaService.group.findMany({
      where: {
        patients: {
          some: {
            id: patientId,
          },
        },
      },
      select: {
        id: true,
        name: true,
      },
    });

    this.redisService.setResponseKey(fetchPatientGroupsRedisKey, groups, this.environmentService.redisConfig.longerTimeToLive);

    return groups;
  }

  async isGroupsByIdExist(ids: IdDto[]): Promise<boolean> {
    const idFlated = ids.map((idObject) => idObject.id);
    const groupFilterCount = await this.prismaService.group.aggregate({
      where: {
        id: {
          in: idFlated,
        },
      },
      _count: {
        _all: true,
      },
    });

    if (groupFilterCount._count._all < idFlated.length) throw new MultiLanguageException(LanguageErrorEnum.GROUP_NOT_FOUND);
    return true;
  }

  async assignPatientToGroup(assignPatient: AssignPatientToGroupDto): Promise<boolean> {
    //AssignGroupResponse
    const { groupId, userId } = assignPatient;
    await this.patientService.isUserPatient(this.commonService.BigIntParser(userId));

    await this.isGroupByIdExists(this.commonService.BigIntParser(groupId));

    await this.prismaService.user.update({
      where: {
        id: userId,
      },
      data: {
        patient: {
          update: {
            groups: {
              connect: {
                id: groupId,
              },
            },
          },
        },
      },
    });
    return true;
  }

  async assignAuthedPatientToGroups(user: UserPlayloadResponse): Promise<boolean> {
    let mapedRisks = [];
    const userMatchedGroups: IdDto[] = [];
    const authedUser = await this.prismaService.patient.findUnique({
      where: {
        id: user.patientId,
      },
      select: {
        user: {
          select: {
            patient: {
              select: {
                habitInformation: {
                  select: {
                    numberCigaretesPerDay: true,
                  },
                },
                relapses: {
                  select: { id: true },
                },
                addictionLevel: true,
                riskFactor: {
                  select: groupRiskFactorSelect,
                },
              },
            },
            personalInformation: {
              select: {
                dateOfBirth: true,
                gender: true,
                height: true,
                weight: true,
              },
            },
          },
        },
      },
    });

    const { height, weight, dateOfBirth } = authedUser.user.personalInformation;

    const patientAge = this.commonService.dateOfBirthToAge(dateOfBirth);
    const relapsesNumber = authedUser.user.patient.relapses.length;

    const cigarrestesNumber = authedUser.user.patient.habitInformation.numberCigaretesPerDay;
    const addictionLevel = authedUser.user.patient.addictionLevel;

    const massBody = this.commonService.BMI(height, weight);

    const patientRiskFactor = authedUser.user.patient.riskFactor;

    //mapping the riskFactors to use them in OR condition
    if (patientRiskFactor) mapedRisks = this.commonService.mapObject(patientRiskFactor);

    //we r splittig the queries based on the grp type because not all the grp types have same filters
    const [highRiskSCJourneys, customGroups, defaultGroups] = await Promise.all([
      this.prismaService.group.findMany({
        where: {
          type: GroupsTypeEnum.HIGH_RISK,
          condition: {
            riskFactor: {
              OR: mapedRisks,
            },
          },
        },
      }),

      //in this query we r using OR between the age and equals just because we cant filter number on null
      this.prismaService.group.findMany({
        select: {
          id: true,
        },
        where: {
          type: GroupsTypeEnum.CUSTOM,

          condition: {
            AND: [
              {
                OR: [
                  {
                    gender: authedUser.user.personalInformation.gender,
                  },
                  {
                    gender: null,
                  },
                ],
              },
              {
                OR: [
                  {
                    addictionLevel,
                  },
                  {
                    addictionLevel: null,
                  },
                ],
              },
              {
                OR: [
                  {
                    massBody,
                  },
                  {
                    massBody: null,
                  },
                ],
              },
            ],

            ageMax: {
              gte: patientAge,
            },
            ageMin: {
              lte: patientAge,
            },

            relapsesMax: {
              gte: relapsesNumber,
            },
            relapsesMin: {
              lte: relapsesNumber,
            },

            ...(cigarrestesNumber && {
              cigarretesMax: {
                gte: cigarrestesNumber,
              },
              cigarretesMin: {
                lte: cigarrestesNumber,
              },
            }),
          },
        },
      }),

      this.prismaService.group.findMany({
        select: {
          id: true,
        },
        where: {
          type: GroupsTypeEnum.DEFAULT,

          condition: {
            gender: authedUser.user.personalInformation.gender,
            addictionLevel,
            ageMax: {
              gte: patientAge,
            },
            ageMin: {
              lte: patientAge,
            },
          },
        },
      }),
    ]);

    //transforming the type of the results to match IdDto
    //so we can assign the ids easily to the pateint using prisma set operation
    const transformedhighRiskGroupIds: IdDto[] = highRiskSCJourneys.map((group) => {
      return { id: group.id };
    });

    const transformedCustomkGroupIds: IdDto[] = customGroups.map((group) => {
      return { id: group.id };
    });

    const transformedDefaultGroupIds: IdDto[] = defaultGroups.map((group) => {
      return { id: group.id };
    });

    //adding matched grps to the array so we can assign them all at the end
    userMatchedGroups.push(...transformedhighRiskGroupIds);

    userMatchedGroups.push(...transformedCustomkGroupIds);

    userMatchedGroups.push(...transformedDefaultGroupIds);

    this.resetPatientGroups(userMatchedGroups, user.patientId);

    return true;
  }

  async isGroupByIdExists(groupId: number): Promise<boolean> {
    const isGroupByIdExistsRedisKey = `isGroupByIdExists/${groupId}`;
    const cacheResults = await this.redisService.getResponseKey(isGroupByIdExistsRedisKey);
    if (cacheResults) return cacheResults;

    const grp = await this.prismaService.group.findFirst({
      where: {
        id: groupId,
      },
      select: { id: true },
    });

    if (!grp) throw new MultiLanguageException(LanguageErrorEnum.GROUP_NOT_FOUND);
    this.redisService.setResponseKey(isGroupByIdExistsRedisKey, 'true', this.environmentService.redisConfig.TTL);

    return true;
  }

  async resetPatientGroups(groupIds: IdDto[], patientId: number): Promise<boolean> {
    await this.prismaService.patient.update({
      where: { id: patientId },
      data: {
        groups: {
          set: groupIds,
        },
      },
    });
    return true;
  }

  async groupNrt(language: LanguageEnum, groupId: number, nrtId: number): Promise<GroupNrtResponse> {
    await this.isGroupByIdExists(Number(groupId));

    const groupNrt = await this.prismaService.group.findFirst({
      where: {
        id: groupId,
        ...(nrtId && {
          nrts: {
            id: { not: nrtId },
          },
        }),
      },
      select: {
        id: true,
        nrts: {
          select: {
            id: true,
            nrtDetail: {
              select: {
                id: true,
                nrtType: true,
                nrtDetailsTranslation: {
                  where: {
                    language: {
                      code: language,
                    },
                  },
                  select: {
                    gum: true,
                    patch: true,
                  },
                },
              },
            },
            priority: true,
            isEditable: true,
            targetGroup: {
              select: {
                id: true,
                name: true,
              },
            },
          },
        },
      },
    });

    return groupNrt;
  }

  async groupArticlePackage(groupId: number): Promise<ArticlePackageWithCountResponse> {
    await this.isGroupByIdExists(Number(groupId));

    const groupArticle = await this.prismaService.group.findFirst({
      where: {
        id: groupId,
      },

      select: {
        articlePackages: {
          select: {
            id: true,
            isEditable: true,
            targetGroup: {
              select: {
                id: true,
                name: true,
              },
            },
            _count: {
              select: {
                recommendedArticles: true,
              },
            },
          },
        },
      },
    });

    return groupArticle.articlePackages;
  }
