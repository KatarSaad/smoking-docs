import { Injectable } from '@nestjs/common';
import { LanguageEnum, SettingOptionNameEnum } from '@prisma/client';
import { PrismaService } from 'src/common/services/prisma.service';
import { SettingResponse } from './response/settings.response';
import { SettingOptionResponse } from './response/setting-option.response';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { RoleSettings } from 'src/common/consts/settings-data';
import { PrismaTransactionClient } from 'src/common/types/transaction.type';

@Injectable()
export class SettingService {
  constructor(private readonly prismaService: PrismaService) {}

  async getSettings(userId: number | bigint): Promise<SettingResponse | null> {
    //update this to return the translation of the header luanguage
    try {
      const setting = await this.prismaService.setting.findFirst({
        where: {
          userId,
        },
        select: {
          id: true,
          language: true,
          options: {
            select: {
              id: true,
              isOn: true,
              name: true,
              section: true,
              type: true,
              // Include any other fields you need here
            },
          },
          // Exclude userId by not including it here
        },
      });

      return setting;
    } catch (error) {
      throw new MultiLanguageException(LanguageErrorEnum.PROBLEM_WHEN_FETCHING_SETTING); //next time create Can not get settings Error
    }
  }

  async toggleSettingOption(userId: number | bigint, name: SettingOptionNameEnum): Promise<SettingOptionResponse | null> {
    try {
      // Find the setting option belonging to the specified user with the provided name
      const settingOption = await this.prismaService.settingOption.findFirst({
        where: {
          setting: {
            userId,
          },
          name,
        },
      });

      if (!settingOption) {
        // If the setting option does not exist, return null
        throw new MultiLanguageException(LanguageErrorEnum.SETTING_OPTION_NOT_FOUND);
      }

      // Toggle the 'isOn' field
      const updatedIsOn = !settingOption.isOn;

      // Update the setting option with the new 'isOn' value
      const updatedSettingOption = await this.prismaService.settingOption.update({
        where: {
          id: settingOption.id,
        },
        data: {
          isOn: updatedIsOn,
        },
      });

      return updatedSettingOption;
    } catch (error) {
      throw new MultiLanguageException(LanguageErrorEnum.PROBLEM_IN_TOGGLE_SETTING);
    }
  }

  async assignSettings(SettingsData: RoleSettings, userId: bigint, transaction?: PrismaTransactionClient) {
    const settingsData = SettingsData;
    const settingData = {
      type: settingsData.type,
      options: {
        create: settingsData.options.map((option) => ({
          name: option.name,
          section: option.section,
          type: option.type,
        })),
      },
      user: {
        connect: { id: BigInt(userId) }, // Ensure id is of type number or bigint
      },
    };

    // Use the transaction if available, otherwise use the default Prisma client
    const prismaClient = transaction ?? this.prismaService;

    const setting = await prismaClient.setting.upsert({
      where: { userId: userId }, // Identify the setting by userId
      update: settingData, // Update the existing setting or create a new one
      create: settingData, // Create a new setting if it doesn't exist
      include: { user: true }, // Include the associated user in the result
    });

    return setting;
  }

  async getSettingOptionOfUserByName(userId: number | bigint, name: SettingOptionNameEnum): Promise<SettingOptionResponse | null> {
    // Find the setting option belonging to the specified user with the provided name
    const settingOption = await this.prismaService.settingOption.findFirst({
      where: {
        setting: {
          userId,
        },
        name,
      },
      select: {
        id: true,
        isOn: true,
        name: true,
        section: true,
        type: true,
        // Include any other fields you need here
      },
    });

    // Explicitly check for null and throw the exception
    if (!settingOption) {
      return null;
    }

    return settingOption;
  }

  async changeLanguage(userId: bigint, language: LanguageEnum) {
    // Update the setting for the user with the new language preference
    const updatedSetting = await this.prismaService.setting.update({
      where: {
        userId: userId,
      },
      data: {
        language: language,
      },
    });

    return updatedSetting;
  }
}
