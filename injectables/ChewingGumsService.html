import { Injectable } from '@nestjs/common';
import { ChewingGum } from '@prisma/client';
import { plainToInstance } from 'class-transformer';
import { DateDto } from 'src/common/dto/date.dto';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';
import { PrismaService } from 'src/common/services/prisma.service';
import { AddChewingGumDto } from 'src/modules/chewing-gums/dto/add-chewing-gum.dto';
import { UpdateChewingGumDto } from 'src/modules/chewing-gums/dto/update-chewing-gum.dto';
import { ChewingGumResponse } from 'src/modules/chewing-gums/response/chewing-gum.response';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';

@Injectable()
export class ChewingGumsService {
  constructor(private prismaService: PrismaService) {}

  async fetchChewingGums(patientId: number, dateDto: DateDto, isLastThreeRow: boolean): Promise<ChewingGumResponse[]> {
    const { dateEndTime, dateStartTime } = dateDto;
    const take = isLastThreeRow ? 3 : undefined;

    const chewingGums = await this.prismaService.chewingGum.findMany({
      where: {
        patientId,
        ...(!isLastThreeRow && {
          takeDate: {
            gt: dateStartTime,
            lt: dateEndTime,
          },
        }),
      },
      select: {
        id: true,
        takeDate: true,
        number: true,
        nicotinePerPiece: true,
        chewedPeriod: true,
        cravingLevel: true,
        symptom: true,
        createdAt: true,
      },
      orderBy: { takeDate: 'desc' },
      take,
    });

    return plainToInstance(ChewingGumResponse, chewingGums, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });
  }

  async addChewingGum(user: UserPlayloadResponse, addChewingGumDto: AddChewingGumDto): Promise<ChewingGumResponse> {
    const addedChwingGum = await this.prismaService.chewingGum.create({
      data: {
        ...addChewingGumDto,
        patientId: user.patientId,
      },
      select: {
        id: true,
        takeDate: true,
        number: true,
        nicotinePerPiece: true,
        chewedPeriod: true,
        cravingLevel: true,
        symptom: true,
        createdAt: true,
      },
    });

    return plainToInstance(ChewingGumResponse, addedChwingGum, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });
  }

  async fetchFirstChewingGum(user: UserPlayloadResponse): Promise<ChewingGum> {
    return await this.prismaService.chewingGum.findFirst({
      where: {
        patientId: user.patientId,
      },
    });
  }

  async updateChewingGum(user: UserPlayloadResponse, updateChewingGumDto: UpdateChewingGumDto, id: number): Promise<ChewingGumResponse> {
    await this.isChewingGumOwner(user, id);

    const updatedChewingGum = await this.prismaService.chewingGum.update({
      where: { id },
      data: {
        ...updateChewingGumDto,
      },
      select: {
        id: true,
        takeDate: true,
        number: true,
        nicotinePerPiece: true,
        chewedPeriod: true,
        cravingLevel: true,
        symptom: true,
        createdAt: true,
      },
    });

    return plainToInstance(ChewingGumResponse, updatedChewingGum, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });
  }

  async deleteChewingGum(user: UserPlayloadResponse, id: number): Promise<ChewingGumResponse> {
    await this.isChewingGumOwner(user, id);

    const deletedChewingGum = await this.prismaService.chewingGum.delete({
      where: { id },
      select: {
        id: true,
        takeDate: true,
        number: true,
        nicotinePerPiece: true,
        chewedPeriod: true,
        cravingLevel: true,
        symptom: true,
        createdAt: true,
      },
    });

    return plainToInstance(ChewingGumResponse, deletedChewingGum, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });
  }

  async isPatchByIdExists(id: number): Promise<ChewingGum> {
    const chewingGumByIdById = await this.prismaService.chewingGum.findUnique({
      where: {
        id,
      },
    });

    if (!chewingGumByIdById) throw new MultiLanguageException(LanguageErrorEnum.CHEWING_GUM_NOT_FOUND);
    return chewingGumByIdById;
  }

  async isChewingGumOwner(user: UserPlayloadResponse, id: number): Promise<boolean> {
    const chewingGumById = await this.isPatchByIdExists(id);

    if (user.patientId !== Number(chewingGumById.patientId)) throw new MultiLanguageException(LanguageErrorEnum.CHEWING_GUM_NOT_YOURS);

    return true;
  }
}
