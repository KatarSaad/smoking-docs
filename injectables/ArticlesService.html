  async fetchPublishedArticles(language: LanguageEnum, user: UserPlayloadResponse, fetchArticlesDto: FetchArticlesDto): Promise<PaginationArticles> {
    const { currentPage, sizePerPage, articleLabelId, search } = fetchArticlesDto;

    const where: Prisma.ArticleWhereInput = {
      status: {
        notIn: [ArticleStatusEnum.DRAFT, ArticleStatusEnum.REJECTED, ArticleStatusEnum.SCHEDULED],
      },
      articleLabelId,

      ...(!user.isSuperAdmin && {
        OR: [
          {
            publisherId: user.adminId,
          },
          {
            status: ArticleStatusEnum.APPROVED,
          },
        ],
      }),

      articlesTranslation: {
        some: {
          title: {
            startsWith: search,
          },
        },
      },
    };

    const [articles, paginationData] = await this.prismaService
      .extendPagination()
      .article.paginate({
        where,
        select: this.getArticleType(language),
        orderBy: {
          updatedAt: 'desc',
        },
      })
      .withPages({
        includePageCount: true,
        page: currentPage,
        limit: sizePerPage,
      });

    const serialized = plainToInstance(ArticleResponse, articles, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    const returnedArticles = { data: serialized, totalItems: paginationData.totalCount };

    return returnedArticles;
  }

  async createArticle(language: LanguageEnum, user: UserPlayloadResponse, addArticleDto: AddArticleDto, file: Express.Multer.File): Promise<ArticleResponse> {
    const { titleEn, titleAr, bodyAr, bodyEn, groupIds, labelId, genderFilter } = addArticleDto;
    const isGroupIds = groupIds?.length > 0;
    const titles = [titleEn, titleAr];
    let imageName = null;

    const createNotificationDto: CreateNotificationDto = {
      messageAr: 'مقال منشور جديد',
      messageEn: 'New published article',
      titleAr: 'مقال منشور جديد',
      titleEn: 'New published article',
      deliveryDate: new Date(),
      notificationType: 'Article',
      isCritical: false,
    };

    if ((bodyAr && titleAr) || (bodyEn && titleEn)) throw new MultiLanguageException(LanguageErrorEnum.PROVIDE_AT_LEAST_ONE_LANGUAGES_FIELDS);

    const articleNames = titles.filter((title) => title != null);

    //check if articles posted titles are duplicated
    const isDuplicateTitles = this.commonService.isArrayHasDuplicateValues(titles);
    if (isDuplicateTitles) throw new MultiLanguageException(LanguageErrorEnum.PROVIDE_DIFFERENT_VALUES_FOR_DIFFERENT_LANGUAGES);

    //check if article name exists
    const isArticleNameExists = !!(await this.getArticleByName(articleNames));
    if (isArticleNameExists) throw new MultiLanguageException(LanguageErrorEnum.ARTICLE_NAME_EXISTS);

    const articleLabel = await this.articlesLabelsService.getArticleLabelById(labelId);
    if (!articleLabel) throw new MultiLanguageException(LanguageErrorEnum.ARTICLE_LABEL_NOT_FOUND);

    if (file) imageName = await this.fileService.uploadImage(file);

    if (isGroupIds) await this.groupsService.isGroupsByIdExist(groupIds);

    const arabicWordCount = bodyAr.split(/\s+/).length;
    const arabicReadingTime = this.calculateReadingTime(arabicWordCount, ReadingSpeed.AR);

    const englishWordCount = bodyEn.split(/\s+/).length;
    const englishReadingTime = this.calculateReadingTime(englishWordCount, ReadingSpeed.EN);
    const article = await this.prismaService.article.create({
      data: {
        image: imageName,
        genderFilter,
        isGroupNull: !isGroupIds,
        statusList: {
          create: {
            adminId: user.adminId,
          },
        },
        targetGroups: {
          connect: groupIds,
        },
        publisher: {
          connect: {
            id: user.adminId,
          },
        },
        articleLabel: {
          connect: {
            id: labelId,
          },
        },
        articlesTranslation: {
          create: [
            {
              title: titleEn,
              readingTime: englishReadingTime,
              body: bodyEn,
              language: {
                connect: {
                  code: LanguageEnum.ENGLISH,
                },
              },
            },
            {
              title: titleAr,
              readingTime: arabicReadingTime,
              body: bodyAr,
              language: {
                connect: {
                  code: LanguageEnum.ARABIC,
                },
              },
            },
          ],
        },
      },
      select: this.getArticleType(language),
    });

    const superAdminId = await this.usersService.fetchSuperadminId();

    this.notificationService.createAdminNotification(
      user,
      createNotificationDto,
      BigInt(superAdminId),
      Number(article.id),
      SettingOptionNameEnum.NEW_ARTICLE_REVIEW_NOTIFICATION,
    );

    const serialized = plainToInstance(ArticleResponse, article, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    return serialized;
  }

  async getPatientArticlesBookMarked(language: LanguageEnum, user: UserPlayloadResponse, fetchArticlesDto: FetchArticlesDto): Promise<PaginationArticlesBookmarked> {
    const { currentPage, articleLabelId, sizePerPage, search } = fetchArticlesDto;

    const personalInformation = await this.usersService.getUserPersonalInformation(user.patientId);
    const userGroups = await this.groupsService.fetchPatientGroups(user.patientId);
    const userGroupsIds = userGroups.map((group) => group.id);

    const where: Prisma.ArticleWhereInput = {
      status: ArticleStatusEnum.APPROVED,
      AND: [
        {
          OR: [
            {
              genderFilter: null,
            },
            { genderFilter: personalInformation.gender },
          ],
        },
        {
          OR: [
            {
              isGroupNull: true,
            },
            {
              targetGroups: {
                some: {
                  id: {
                    in: userGroupsIds,
                  },
                },
              },
            },
          ],
        },
      ],

      articleLabelId,
      articlesTranslation: {
        some: {
          title: {
            startsWith: search,
          },
        },
      },
    };

    const [articles, paginationData] = await this.prismaService
      .extendPagination()
      .article.paginate({
        where,
        select: this.getArticleType(language),

        orderBy: {
          updatedAt: 'desc',
        },
      })
      .withPages({
        includePageCount: true,
        page: currentPage,
        limit: sizePerPage,
      });

    const bookmarksIds = await this.getFavoriteArticleIdsOfUser(user.id);
    const serialized = plainToInstance(ArticleBookmedResponse, articles, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    // Add isBookmarked field to each article
    const articlesWithBookmarks = serialized.map((article) => {
      const isBookmarked = bookmarksIds.includes(article.id);
      return { ...article, isBookmarked };
    });

    const returnedArticles = { data: articlesWithBookmarks, totalItems: paginationData.totalCount };

    return returnedArticles;
  }
  async getFavoriteArticleIdsOfUser(userId: number): Promise<bigint[]> {
    const user = await this.prismaService.user.findUnique({
      where: { id: userId },
      include: {
        favoriteArticles: {
          select: {
            id: true, // Select only the article IDs
          },
        },
      },
    });

    if (!user) {
      // Handle case where user is not found
      throw new MultiLanguageException(LanguageErrorEnum.USER_NOT_FOUND);
    }

    // Extract article IDs from the fetched data
    const articleIds = user.favoriteArticles.map((article) => article.id);

    return articleIds;
  }

    if (deletedArticle.image) this.fileService.deleteFile(deletedArticle.image);

    const serialized = plainToInstance(ArticleResponse, deletedArticle, {

      enableImplicitConversion: true,

  async scheduleArticle(language: LanguageEnum, user: UserPlayloadResponse, scheduleArticleDto: ScheduleArticleDto, id: number): Promise<ArticleResponse> {
    const { scheduleAt } = scheduleArticleDto;
    let approveArticleJob = null;
    this.commonService.isDateAfterNow(scheduleAt);

    const articleById = await this.isArticleByIdExists(
      [ArticleStatusEnum.SUBMITTED, ArticleStatusEnum.UPDATED, ArticleStatusEnum.REJECTED, ArticleStatusEnum.SCHEDULED],
      id,
    );
    const isArticleScheduled = articleById.status === ArticleStatusEnum.SCHEDULED;

    const delay = this.commonService.remainingTime(scheduleAt);

    if (articleById.status === ArticleStatusEnum.SCHEDULED) {
      const job = await this.approveArticleQueue.getJob(articleById.jobId);
      await job?.changeDelay(delay);
    } else {
      approveArticleJob = await this.approveArticleQueue.add(
        QueueEnum.APPROVE_ARTICLE_QUEUE,
        {
          id,
          adminId: user.adminId,
        },
        {
          removeOnComplete: 500,
          removeOnFail: 1000,
          delay,
        },
      );
    }

    const article = await this.prismaService.article.update({
      where: { id },
      data: {
        status: ArticleStatusEnum.SCHEDULED,
        ...(!isArticleScheduled && { jobId: approveArticleJob.id }),
        scheduledAt: scheduleAt,
        statusList: {
          create: {
            type: ArticleStatusEnum.SCHEDULED,
            scheduledAt: scheduleAt,
            adminId: user.adminId,
          },
        },
      },
      select: this.getArticleType(language),
    });

    const serialized = plainToInstance(ArticleResponse, article, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    return serialized;
  }

  async addFavoriteArticle(user: UserPlayloadResponse, language: LanguageEnum, articleId: number): Promise<ArticleResponse> {
    // Check if the article exists and meets the criteria for being favorited
    const article = await this.fetchArticleByIdSimple(language, user, articleId);

    // Check if the article is already favorited by the user
    const existingFavorite = await this.prismaService.user.findFirst({
      where: {
        id: user.id,

        favoriteArticles: {
          some: { status: ArticleStatusEnum.APPROVED, id: articleId },
        },
      },
    });

    if (existingFavorite) {
      // Article is already favorited by the user
      throw new MultiLanguageException(LanguageErrorEnum.ARTICLE_ALREADY_FAVORITE);
    }

    // Add the article to the user's favorites
    await this.prismaService.user.update({
      where: { id: user.id },
      data: { favoriteArticles: { connect: { id: articleId } } },
    });

    // Return the favorited article
    return article;
  }

  async removeFavoriteArticle(user: UserPlayloadResponse, articleId: number, language: LanguageEnum): Promise<any> {
    const article = await this.fetchArticleByIdSimple(language, user, articleId);

    // Check if the article is favorited by the user
    const existingFavorite = await this.prismaService.user.findFirst({
      where: {
        id: user.id,
        favoriteArticles: { some: { id: articleId } },
      },
    });

    if (!existingFavorite) {
      // Article is not favorited by the user
      throw new MultiLanguageException(LanguageErrorEnum.ARTILCE_IS_NOT_FAVORITE);
    }

    // Remove the article from the user's favorites
    await this.prismaService.user.update({
      where: { id: user.id },
      data: { favoriteArticles: { disconnect: { id: articleId } } },
    });

    return article;
  }

        id: {
          in: idFlated,

      _count: {

      },

  async checkIfRead(language: LanguageEnum, userId: number, articleId: number, checkIfRead: CheckIfReadDto): Promise<boolean> {
    // Retrieve the article translation from the database
    const articleTranslation = await this.prismaService.articleTranslation.findFirst({
      where: {
        articleId,
        language: { code: language }, // Filter by language
      },
      select: { readingTime: true }, // Select only the readingTime field
    });

    if (!articleTranslation) {
      throw new MultiLanguageException(LanguageErrorEnum.TRANSLATION_NOT_FOUND);
    }

    // Compare the reading time of the article with the specified threshold
    if (articleTranslation.readingTime && articleTranslation.readingTime < checkIfRead.readingTime) {
      // If the reading time is less than the threshold, mark the article as read for the user
      await this.createUserArticleRead(userId, articleId);
      return true;
    }
    return false;
  }

  async createUserArticleRead(userId: number, articleId: number): Promise<User> {

      });

      // Optionally, you might want to return the updated user or the newly created readArticle
      // For demonstration, returning the user
      return prisma.user.findUnique({
        where: { id: userId },
      });
    });

    return user;
  }

  async articleReadCount(userId: bigint, timePeriod: TimePeriodEnum): Promise<number> {
    const oneWeekAgo = new Date();
    oneWeekAgo.setDate(oneWeekAgo.getDate() - 7);

    const oneMonthAgo = new Date();
    oneMonthAgo.setMonth(oneMonthAgo.getMonth() - 1);

    let whereCondition: Prisma.ReadArticleWhereInput = { userId: userId };

    if (timePeriod === TimePeriodEnum.WEEK) {
      whereCondition = { ...whereCondition, createdAt: { gte: oneWeekAgo } };
    } else if (timePeriod === TimePeriodEnum.MONTH) {
      whereCondition = { ...whereCondition, createdAt: { gte: oneMonthAgo } };
    }

    const readArticleCount = await this.prismaService.readArticle.count({
      where: whereCondition,
    });

    return readArticleCount; // Return the count of read articles
  }

  async recommandedArticle(user: UserPlayloadResponse, language: LanguageEnum): Promise<ArticleResponse> {
    let randomArticleId = null;
    const fetchPubishedRandomdArticlesId = `fetchPubisheRandomdArticlesId/${language}`;

    randomArticleId = await this.redisService.getResponseKey(fetchPubishedRandomdArticlesId);
    if (!randomArticleId) {
      const cacheTime = this.commonService.milliSecondsUntilEndOfDay();

      randomArticleId = await this.fetchRandomPubishedArticleId();

      this.redisService.setResponseKey(fetchPubishedRandomdArticlesId, randomArticleId, cacheTime);
    }

    return await this.fetchArticleById({ language }, user, randomArticleId);
  }

  async fetchRandomPubishedArticleId(): Promise<bigint> {
    const articleIds = await this.prismaService.article.findMany({
      where: {
        status: ArticleStatusEnum.APPROVED,
      },
      select: { id: true },
    });

    if (articleIds.length === 0) throw new MultiLanguageException(LanguageErrorEnum.ARTICLE_NOT_FOUND);

    const randomArticleId = articleIds[Math.floor(Math.random() * articleIds.length)].id;

    return randomArticleId;
  }
