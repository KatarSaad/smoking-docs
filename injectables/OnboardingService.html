import { Injectable } from '@nestjs/common';
import { plainToInstance } from 'class-transformer';
import { PrismaService } from 'src/common/services/prisma.service';
import { UpdateHabitInfoDto } from './dto/update-habit-info.dto';
import { UpdatePersonalInfoDto } from './dto/update-personal-info.dto';
import { PersonalInformationResponse } from './response/personal-info.response';
import { CommonService } from 'src/common/services/common.service';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { NotificationService } from '../notification/notification.service';
import { PrismaTransactionClient } from 'src/common/types/transaction.type';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';

@Injectable()
export class OnboardingService {
  constructor(
    private readonly prismaService: PrismaService,
    private commonService: CommonService,
    private notificationService: NotificationService,
  ) {}

  async updateHabitInfo(patientId: number, updateHabitInfoDto: UpdateHabitInfoDto, transaction?: PrismaTransactionClient) {
    const transactionOperation = transaction ?? this.prismaService;

    const oldHabitInformation = await this.prismaService.habitInformation.findFirst({
      where: {
        patientId,
      },
      select: {
        quitDate: true,
      },
    });

    const updatedHabitInformation = await transactionOperation.habitInformation.update({
      where: {
        patientId,
      },
      data: {
        ...updateHabitInfoDto,
      },
    });
    //automated preparedness
    if (new Date(oldHabitInformation.quitDate) != new Date(updateHabitInfoDto.quitDate)) {
      const patient = await this.prismaService.patient.findUnique({
        where: {
          id: patientId,
        },
        select: {
          userId: true,
        },
      });
      const notifsToCancel = await this.prismaService.notification.findMany({
        where: {
          notificationType: {
            contains: 'automatedWeekly',
          },
        },
        select: {
          id: true,
        },
      });
      await this.notificationService.cancelNotificationCron([
        `schedulePreparedness:${patient.userId}`,
        ...notifsToCancel.map((notif) => `scheduleNotification:${notif.id}:${patient.userId}`),
      ]);
      if (new Date(updateHabitInfoDto.quitDate) > new Date()) {
        this.notificationService.launchNotificationCron(
          'schedulePreparedness',
          {
            userId: Number(patient.userId),
            identifier: `schedulePreparedness:${patient.userId}`,
            endDate: new Date(updateHabitInfoDto.quitDate),
          },
          {
            repeat: {
              pattern: '0 0 * * *',
            },
          },
        );
      } else {
        this.notificationService.launchNotificationCron('scheduleWeekly', { userId: Number(patient.userId) }, { delay: 0 });
      }
    }
    return updatedHabitInformation;
  }
  async updatePersonalInfo(
    user: UserPlayloadResponse,
    updatePersonalInfoDto: UpdatePersonalInfoDto,
    transaction?: PrismaTransactionClient,
  ): Promise<PersonalInformationResponse> {
    const { id, patientId } = user;
    const transactionOperation = transaction ?? this.prismaService;
    if (updatePersonalInfoDto.dateOfBirth) this.commonService.isDateBeforeNow(updatePersonalInfoDto.dateOfBirth);

    const patient = await transactionOperation.patient.update({
      where: {
        id: patientId,
      },
      select: {
        userId: true,
        user: {
          include: {
            personalInformation: true,
            patient: {
              select: {
                id: true,
              },
            },
          },
        },
      },
      data: {
        user: {
          update: {
            where: {
              id,
            },
            data: {
              personalInformation: {
                update: {
                  ...updatePersonalInfoDto,
                },
              },
            },
          },
        },
      },
    });

    const serialized = plainToInstance(PersonalInformationResponse, patient.user.personalInformation, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });
    return serialized;
  }
  //TODO add return type
  async getPersonalInfo(patientId: number) {
    return await this.prismaService.personalInformation.findFirst({
      where: {
        user: {
          patient: {
            id: patientId,
          },
        },
      },
      select: {
        user: {
          select: {
            patient: {
              select: {
                id: true,
              },
            },
          },
        },
      },
    });
  }

  async getHabitInfo(patientId: number) {
    const habitInfo = await this.prismaService.habitInformation.findUnique({
      where: {
        patientId,
      },
    });

    return habitInfo;
  }

  async isEmailNotExist(email: string, userId: number): Promise<boolean> {
    const userEmail = await this.prismaService.personalInformation.findUnique({
      where: {
        email,
        ...(userId && {
          user: {
            id: { not: userId },
          },
        }),
      },
      select: { id: true },
    });

    if (userEmail) throw new MultiLanguageException(LanguageErrorEnum.EMAIL_EXISTS);
    return true;
  }

  async isEmailExist(email: string): Promise<boolean> {
    const userEmail = await this.prismaService.personalInformation.findUnique({
      where: {
        email,
      },
      select: { id: true },
    });

    if (!userEmail) throw new MultiLanguageException(LanguageErrorEnum.EMAIL_NOT_FOUND);
    return true;
  }
}
