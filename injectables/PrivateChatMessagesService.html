  async addMessageToPrivateChatHtttp(user: UserPlayloadResponse, messagePrivateDto: MessagePrivateDto, file?: Express.Multer.File): Promise<PrivateChatMessageResponse> {
    const { privateChatId, message, replyId, socketId } = messagePrivateDto;
    let fileData: FileResponse = null;
    const room = String(privateChatId);

    const privateChat = await this.privateChatsService.userPrivateChat(user.id, privateChatId);
    if (file) fileData = await this.fileService.uploadFile(file);

    //if the sender is patient
    if (user.isPatient && message) {
      //patient cant send messsage is chat is disabled
      if (privateChat.isDisabled) throw new MultiLanguageException(LanguageErrorEnum.CHAT_DISABLED);

      const isUrlNotDetected = this.commonService.isUrlNotDetected(message);
      if (isUrlNotDetected) throw new MultiLanguageException(LanguageErrorEnum.URL_NOT_ALLOWED);
    }

    if (replyId) await this.isMessageByIdExists(replyId);

    const receiverId = user.id !== Number(privateChat.healthSpecialistUserId) ? privateChat.healthSpecialistUserId : privateChat.patientUserId;

    const privateChatMessage = await this.prismaService.privateChatMessages.create({
      data: {
        message,
        privateChatId,
        senderId: user.id,
        receiverId,
        replyId,
        ...(fileData && { file: { create: fileData } }),
      },
      select: {
        id: true,
        message: true,
        file: {
          select: {
            name: true,
            blobName: true,
            extension: true,
            size: true,
          },
        },
        reply: {
          select: {
            id: true,
            file: {
              select: {
                blobName: true,
                extension: true,
                name: true,
              },
            },
            sender: {
              select: {
                id: true,
                personalInformation: { select: { firstName: true, lastName: true } },
              },
            },
            message: true,
            createdAt: true,
          },
        },
        sender: { select: { id: true, personalInformation: { select: { firstName: true, lastName: true } } } },
        receiverId: true,
        createdAt: true,
      },
    });

    this.socketService.socket
      .to(`private_chats_${room}`)
      .emit(SocketEventEnum.MESSAGE_PRIVATE_CHATS, this.commonService.objectParser({ privateChatId, message, file: fileData, senderId: user.id }));

    this.socketService.socket
      .to(`private_chat_${room}`)
      .except(socketId) //make this dynamic
      .emit(SocketEventEnum.MESSAGE_PRIVATE_CHAT, this.commonService.objectParser(privateChatMessage));

    const senderName = `${privateChatMessage.sender.personalInformation.firstName} ${privateChatMessage.sender.personalInformation.lastName}`;
    this.privateChatNotification(user, privateChatMessage.receiverId, privateChatId, SettingOptionNameEnum.ALL_MESSAGES_IN_PRIVATE_CHAT, senderName);

    return privateChatMessage;
  }

  async PatientFetchPrivateChatMessages(
    user: UserPlayloadResponse,
    searchWithCursorPaginationDto: SearchWithCursorPaginationDto,
  ): Promise<PaginationCursorPrivateChatMessageResponse> {
    const { oldCursorId, sizePerPage, search } = searchWithCursorPaginationDto;

    const fullTextSearch = search ? this.commonService.addLeadingPlusAndTrailingWildcard(search) : undefined;

    const userPrivateChat = await this.privateChatsService.privateChatByUserId(user.id);
    if (!userPrivateChat) throw new MultiLanguageException(LanguageErrorEnum.USER_NOT_IN_PRIVATE_CHAT);

    const [messages, messagePagination] = await this.prismaService
      .extendPagination()
      .privateChatMessages.paginate({
        where: {
          privateChatId: userPrivateChat.id,
          message: { search: fullTextSearch },
        },
        select: {
          id: true,
          message: true,
          file: true,
          reply: {
            select: {
              id: true,
              file: {
                select: {
                  blobName: true,
                  extension: true,
                  name: true,
                },
              },
              sender: {
                select: {
                  id: true,
                  personalInformation: { select: { firstName: true, lastName: true } },
                },
              },
              message: true,
              createdAt: true,
            },
          },
          sender: {
            select: {
              id: true,
              personalInformation: {
                select: {
                  firstName: true,
                  lastName: true,
                },
              },
            },
          },
          receiverId: true,
          createdAt: true,
        },
        orderBy: {
          createdAt: 'desc',
        },
      })
      .withCursor({
        limit: sizePerPage,
        after: oldCursorId ? `${oldCursorId}` : null,
        getCursor({ id }) {
          return id.toString();
        },
      });

    const { hasNextPage, hasPreviousPage, endCursor, startCursor } = messagePagination;

    return { data: messages, hasNextPage, hasPreviousPage, endCursor: Number(endCursor), startCursor: Number(startCursor) };
  }

  async isMessageByIdExists(id: number): Promise<boolean> {
    const privateMessage = await this.prismaService.privateChatMessages.findUnique({ where: { id }, select: { id: true } });

    if (!privateMessage) throw new MultiLanguageException(LanguageErrorEnum.MESSAGE_NOT_FOUND);
    return true;
  }

  async deletePrivateMessage(user: UserPlayloadResponse, id: number): Promise<PrivateChatMessageResponse> {
    await this.isMessageByIdExists(id);
    await this.isPrivateMessageSender(user, id);

    const deletedMessage = await this.prismaService.privateChatMessages.delete({
      where: { id },
      select: {
        id: true,
        message: true,
        file: {
          select: {
            blobName: true,
            extension: true,
            name: true,
          },
        },
        reply: {
          select: {
            id: true,
            file: {
              select: {
                blobName: true,
                extension: true,
                name: true,
              },
            },
            sender: {
              select: {
                id: true,
                personalInformation: { select: { firstName: true, lastName: true } },
              },
            },
            message: true,
            createdAt: true,
          },
        },
        sender: { select: { id: true, personalInformation: { select: { firstName: true, lastName: true } } } },
        receiverId: true,
        createdAt: true,
      },
    });

    if (deletedMessage.file) this.fileService.deleteFile(deletedMessage.file.blobName);

    return deletedMessage;
  }

  async isPrivateMessageSender(user: UserPlayloadResponse, id: number): Promise<boolean> {
    const message = await this.prismaService.privateChatMessages.findUnique({ where: { id, senderId: user.id } });

    if (!message) throw new MultiLanguageException(LanguageErrorEnum.MESSAGE_NOT_YOURS);
    return true;
  }

  async privateChatNotification(
    user: UserPlayloadResponse,
    receiverId: bigint,
    privateChatId: number,
    settingOptionNameEnum: SettingOptionNameEnum,
    senderName: string,
  ): Promise<boolean> {
    const notificationMessage = `${senderName} sent you a message!`;
    const notificationMessageAr = `${senderName}   !`;
    const createNotificationDto: CreateNotificationDto = {
      messageAr: notificationMessageAr,
      messageEn: notificationMessage,
      titleAr: ' !',
      titleEn: 'New message!',
      deliveryDate: new Date(),
      notificationType: 'PrivateChat',
      isCritical: false,
    };

    if (user.isHS) this.notificationService.sendPushNotif([{ id: receiverId }], settingOptionNameEnum, notificationMessage);
    else this.notificationService.createAdminNotification(user, createNotificationDto, receiverId, privateChatId, settingOptionNameEnum);
    return true;
  }
}
