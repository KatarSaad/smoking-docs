import { Inject, Injectable, forwardRef } from '@nestjs/common';
import { LanguageEnum } from '@prisma/client';
import { DateDto } from 'src/common/dto/date.dto';
import { CommonService } from 'src/common/services/common.service';
import { ArticlesService } from 'src/modules/articles/articles.service';
import { ChewingGumsService } from 'src/modules/chewing-gums/chewing-gums.service';
import { CravingsService } from 'src/modules/cravings/cravings.service';
import { HasEntryResponse } from 'src/modules/diaries/response/has-entry.response';
import { DiaryType } from 'src/modules/diaries/types/diary.type';
import { MoodsService } from 'src/modules/moods/moods.service';
import { PatchesService } from 'src/modules/patches/patches.service';
import { PatientService } from 'src/modules/patients/patient.service';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';
import { RelapsesService } from 'src/modules/relapses/relapses.service';
import { SymptomsService } from 'src/modules/symptoms/symptoms.service';
import { AchievementsService } from '../achievements/achievements.service';
import { ChallengesService } from '../challenges/challenges.service';

@Injectable()
export class DiariesService {
  constructor(
    private chewingGumsService: ChewingGumsService,
    @Inject(forwardRef(() => PatientService))
    private patientService: PatientService,
    private cravingsService: CravingsService,
    private patchesService: PatchesService,
    private moodsService: MoodsService,
    private symptomsService: SymptomsService,
    private articlesService: ArticlesService,
    private relapsesService: RelapsesService,
    private commonService: CommonService,
    private achievementsService: AchievementsService,
    @Inject(forwardRef(() => ChallengesService))
    private challengesService: ChallengesService,
  ) {}

  async fetchDiaries(dateDto: DateDto, userId: number, language: LanguageEnum): Promise<DiaryType[]> {
    const isLastThreeRow = false;

    const patientId = await this.patientService.findPatientIdByUserId(userId, null);

    return await this.fetchPatientDiaries(dateDto, patientId, userId, language, isLastThreeRow);
  }

  async fetchPatientDiaries(dateDto: DateDto, patientId: number, userId: number, language: LanguageEnum, isLastThreeRow: boolean): Promise<DiaryType[]> {
    //we need all possible moods so making this empty will force fetchPatientMoods to not use any filter on moods
    const [symptoms, moods, patches, chewinGums, cravings, articles, relapses, achievements, challenges] = await Promise.all([
      this.symptomsService.fetchSymptoms(patientId, dateDto, isLastThreeRow),
      this.moodsService.fetchPatientMoods(patientId, dateDto, isLastThreeRow),
      this.patchesService.fetchPatches(patientId, dateDto, isLastThreeRow),
      this.chewingGumsService.fetchChewingGums(patientId, dateDto, isLastThreeRow),
      this.cravingsService.fetchCravings(patientId, dateDto, isLastThreeRow),
      this.articlesService.fetchReadArticlesByDate(userId, dateDto, isLastThreeRow, language),
      this.relapsesService.fetchRelapsesByDate(userId, dateDto, isLastThreeRow),
      this.achievementsService.getPatientAchievement(patientId, dateDto, isLastThreeRow, language),
      this.challengesService.getCommpletedChallenges(patientId, dateDto, isLastThreeRow),
    ]);

    //sort data
    const diaryEntries = [...chewinGums, ...cravings, ...patches, ...moods, ...symptoms, ...articles, ...relapses, ...achievements, ...challenges].sort(
      (entry1, entry2) => {
        if (entry2.sortField > entry1.sortField) {
          return 1;
        }
        if (entry1.sortField > entry2.sortField) {
          return -1;
        }
        return 0;
      },
    );

    return diaryEntries;
  }
  //this fetches true if user has ever made entry to any of the diary entries
  async userHasEntry(user: UserPlayloadResponse): Promise<HasEntryResponse> {
    const hasEntryTrue = { hasEntry: true };

    const firstCraving = await this.cravingsService.fetchFirstCraving(user);
    if (firstCraving) return hasEntryTrue;

    const firstSymptom = await this.symptomsService.fetchFirstSymptom(user);
    if (firstSymptom) return hasEntryTrue;

    const firstMood = await this.moodsService.fetchFirstMood(user);
    if (firstMood) return hasEntryTrue;

    const firstPatch = await this.patchesService.fetchFirstPatch(user);
    if (firstPatch) return hasEntryTrue;

    const firstChewingGum = await this.chewingGumsService.fetchFirstChewingGum(user);
    if (firstChewingGum) return hasEntryTrue;

    return { hasEntry: false };
  }

  async fetchLastThreeDiaries(userId: number, language: LanguageEnum): Promise<DiaryType[]> {
    const isLastThreeRow = true;

    //in this case date wont be used in the fetch functions because isLastThreeRow is true
    const date = this.commonService.dateStartAndEndTime(new Date());

    const patientId = await this.patientService.findPatientIdByUserId(userId, null);

    const [first, second, third] = await this.fetchPatientDiaries(date, patientId, userId, language, isLastThreeRow);
    return [first, second, third];
  }

  //this fetches true if user today made entry to any of the diary entries

  async checkIsTodayHasDiaryEntry(user: UserPlayloadResponse): Promise<HasEntryResponse> {
    const currentDay = this.commonService.dateStartAndEndTime(new Date());

    const fetchPatientDiaries = await this.fetchPatientDiaries(currentDay, user.patientId, user.id, LanguageEnum.ENGLISH, false);

    if (fetchPatientDiaries.length > 0)
      return {
        hasEntry: true,
      };
    return { hasEntry: false };
  }
