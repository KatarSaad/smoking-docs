import { Inject, Injectable, forwardRef } from '@nestjs/common';
import { DailyProgressTypeEnum, LanguageEnum } from '@prisma/client';
import { SearchWithPaginationDto } from 'src/common/dto/search-with-pagination.dto';
import { CommonService } from 'src/common/services/common.service';
import { PrismaService } from 'src/common/services/prisma.service';
import { ArticlesService } from 'src/modules/articles/articles.service';
import { CommunityMessagesService } from 'src/modules/community-messages/community-messages.service';
import { DiariesService } from 'src/modules/diaries/diaries.service';
import { HasEntryResponse } from 'src/modules/diaries/response/has-entry.response';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';
import { TodayScreenResponse } from 'src/modules/today-progresses/response/today-progress.response';

@Injectable()
export class TodayProgressesService {
  constructor(
    private prismaService: PrismaService,
    private commonService: CommonService,
    private diariesService: DiariesService,
    @Inject(forwardRef(() => CommunityMessagesService))
    private communityMessagesService: CommunityMessagesService,
    private articlesService: ArticlesService,
  ) {}

  async fetchTodayProgress(user: UserPlayloadResponse, language: LanguageEnum): Promise<TodayScreenResponse> {
    const communityChatMessagesLastTwo: SearchWithPaginationDto = { sizePerPage: 2, currentPage: 1, search: null };

    const [todayDiaryEntry, todayNotifcationChecked, todayArticleChecked, todayChallengeChecked, communityChatMessages, recommendedArticle] = await Promise.all([
      this.diariesService.checkIsTodayHasDiaryEntry(user),
      this.checkTodayDailyProgress(user.id, DailyProgressTypeEnum.NOTIFICATION),
      this.checkTodayDailyProgress(user.id, DailyProgressTypeEnum.ARTICLE),
      this.checkTodayDailyProgress(user.id, DailyProgressTypeEnum.CHALLENGE),
      this.communityMessagesService.fetchCommmunityChatOffsetPagination(user, communityChatMessagesLastTwo),
      this.articlesService.recommandedArticle(user, language),
    ]);

    return {
      todayQuote: 'Code is like humor. When you have to explain it, itâ€™s bad', //TODO: todayQuote should be dynamic
      recommendedArticle,
      todayScreen: {
        todayDiaryEntry,
        todayNotifcationChecked,
        todayChallengeChecked,
        todayArticleChecked,
      },
      communityChatMessages: communityChatMessages.data,
    };
  }

  async addCheckToTodaysProgress(userId: number, dailyProgressType: DailyProgressTypeEnum): Promise<boolean> {
    await this.prismaService.dailyProgress.create({
      data: {
        userId,
        dailyProgressType,
      },
    });

    return true;
  }

  async checkTodayDailyProgress(userId: number, dailyProgressType: DailyProgressTypeEnum): Promise<HasEntryResponse> {
    const { dateEndTime, dateStartTime } = this.commonService.dateStartAndEndTime(new Date());

    const isTodayNotifcationsChecked = await this.prismaService.dailyProgress.findFirst({
      where: {
        userId,
        dailyProgressType,
        createdAt: {
          gt: dateStartTime,
          lt: dateEndTime,
        },
      },
    });

    if (!isTodayNotifcationsChecked) return { hasEntry: false };
    return { hasEntry: true };
  }
}
