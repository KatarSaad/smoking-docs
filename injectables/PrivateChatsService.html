import { Inject, Injectable, forwardRef } from '@nestjs/common';
import { Prisma, PrivateChats } from '@prisma/client';
import { plainToInstance } from 'class-transformer';
import { SearchWithCursorPaginationDto } from 'src/common/dto/search-with-cursor-pagination.dto';
import { SearchWithOptionalPaginationDto } from 'src/common/dto/search-with-optional-pagination.dto';
import { SearchWithPaginationDto } from 'src/common/dto/search-with-pagination.dto';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';
import { PaginationFlatPrivateChatListResponse, PaginationPrivateChatMessageResponse } from 'src/common/responses/Paggination-data.response';
import { PaginationCursorPrivateChatMessageResponse } from 'src/common/responses/paggination-cursor.response';
import { CommonService } from 'src/common/services/common.service';
import { PrismaService } from 'src/common/services/prisma.service';
import { PrismaTransactionClient } from 'src/common/types/transaction.type';
import { HealthFacilityService } from 'src/modules/facilities/health-facility.service';
import { PatientService } from 'src/modules/patients/patient.service';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';
import { CreatePrivateMessageDto } from 'src/modules/private-chats/dto/create-private-message.dto';
import { FlatPrivateChatListResponse } from 'src/modules/private-chats/response/flat-private-chat-list.response';
import { PrivateChatByIdResponse } from 'src/modules/private-chats/response/private-chat-by-is.response';

@Injectable()
export class PrivateChatsService {
  constructor(
    private prismaService: PrismaService,
    @Inject(forwardRef(() => PatientService))
    private patientService: PatientService,
    private healthFacilityService: HealthFacilityService,
    @Inject(forwardRef(() => UsersService))
    private usersService: UsersService,
    private commonService: CommonService,
  ) {}

  async createPrivateChat(user: UserPlayloadResponse, createPrivateMessageDto: CreatePrivateMessageDto): Promise<PrivateChatSimpleResponse> {
    const { userId } = createPrivateMessageDto;

    await this.patientService.isUserPatient(userId);

    const patient = await this.usersService.findPatientByUserId(userId);

    await this.healthFacilityService.isPatientAndHealthSpecialistSameFacility(user.healthSpecialistId, Number(patient.patient.id));

    await this.isPrivateChatByUserIdNotExist(userId);

    return this.prismaService.privateChats.create({
      data: {
        healthSpecialistUserId: user.id,
        patientUserId: userId,
      },
      select: {
        id: true,
        isDisabled: true,
      },
    });
  }

  async fetchPrivateChats(user: UserPlayloadResponse, searchWithPaginationDto: SearchWithPaginationDto): Promise<PaginationFlatPrivateChatListResponse> {
    const { currentPage, sizePerPage, search } = searchWithPaginationDto;
    const incrementedPage = currentPage - 1;
    const fullTextSearch = search ? this.commonService.addLeadingPlusAndTrailingWildcard(search) : undefined;
    const searchQID = `${search}%`; //this is a trick to pass the search value

    const isFiltringSearch = Prisma.sql`${
      search
        ? Prisma.sql`AND (MATCH (personalInfomationsList.firstName , personalInfomationsList.lastName) AGAINST (${fullTextSearch} IN BOOLEAN MODE)
      OR qid LIKE ${searchQID}) `
        : Prisma.empty
    }`;

    const authedUserId = user.id;
    const privateChats: any[] = await this.prismaService.$queryRaw`
    SELECT privateChats.id AS id, privateChatMessages.message AS message,files.id AS fileId,privateChatMessages.createdAt AS createdAt,
    usersList.id AS userId,
    personalInfomationsList.firstName AS firstName,personalInfomationsList.lastName AS lastName,
    patients.id AS patientId,

    privateChats.isDisabled As isDisabled,
    privateChats.isCraving As isCraving,

    (SELECT messagesonusersonprivatechats.updatedAt From messagesonusersonprivatechats 
        WHERE privateChats.id=messagesonusersonprivatechats.privateChatId AND ${authedUserId}=messagesonusersonprivatechats.participantId
    ) AS lastSeenMessageUpdatedAt
    
    ,(SELECT COUNT(CASE WHEN lastSeenMessageUpdatedAt IS NOT NULL then 1 ELSE NULL END) from privatechatmessages WHERE privatechatmessages.createdAt>lastSeenMessageUpdatedAt AND privateChats.id=privateChatMessages.privateChatId) AS unseenMessagesNumber
    ,(SELECT COUNT(CASE WHEN lastSeenMessageUpdatedAt IS NULL then 1 ELSE NULL END) from privatechatmessages WHERE privateChats.id=privateChatMessages.privateChatId) AS messagesNumber

    from privatechats privateChats 
    
    LEFT JOIN privatechatmessages privateChatMessages ON privateChats.id=privateChatMessages.privateChatId 

    LEFT OUTER JOIN privatechatmessages privateChatMessages2 ON( privateChats.id=privateChatMessages2.privateChatId 
    AND privateChatMessages.createdAt < privateChatMessages2.createdAt OR (privateChatMessages.createdAt = privateChatMessages2.createdAt AND privateChatMessages.id < privateChatMessages2.id))

    LEFT JOIN files files ON files.privateChatMessageId=privateChatMessages.id 

    LEFT JOIN user usersList ON usersList.id = privateChats.patientUserId

    LEFT JOIN patient patients ON patients.userId = usersList.id

    LEFT JOIN personalinformation personalInfomationsList ON usersList.id = personalInfomationsList.userId

    WHERE privateChatMessages2.id IS NULL AND privateChats.healthSpecialistUserId=${authedUserId} ${isFiltringSearch}

    ORDER BY -privateChatMessages.createdAt ASC
    LIMIT ${sizePerPage} OFFSET ${incrementedPage * sizePerPage};`;

    //counting all possible rows
    const privateChatsCount = await this.prismaService.$queryRaw`
    SELECT COUNT(*) as totalCount
    FROM  privatechats privateChats 
  
    WHERE ${authedUserId}=privatechats.healthSpecialistUserId;`;

    //hidding unneccessairy data
    const serializedPrivateChatList = plainToInstance(FlatPrivateChatListResponse, privateChats, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    const totalItems = privateChatsCount[0].totalCount;

    return { data: serializedPrivateChatList, totalItems };
  }

  async fetchPrivateChatById(id: number): Promise<PrivateChatByIdResponse> {
    const privateChat = await this.prismaService.privateChats.findUnique({
      where: { id },
      select: {
        id: true,
        isCraving: true,
        isDisabled: true,
        patient: { select: { id: true, personalInformation: { select: { firstName: true, lastName: true } } } },
        createdAt: true,
        updatedAt: true,
      },
    });

    if (!privateChat) throw new MultiLanguageException(LanguageErrorEnum.PRIVATE_CHAT_NOT_FOUND);

    return privateChat;
  }

  async deletePrivateChatById(user: UserPlayloadResponse, id: number): Promise<PrivateChatByIdResponse> {
    await this.isHealthSpecialistInPrivateChat(id, user.id);
    const privateChat = await this.prismaService.privateChats.delete({
      where: { id },
      select: {
        id: true,
        isDisabled: true,
        isCraving: true,
        patient: { select: { id: true, personalInformation: { select: { firstName: true, lastName: true } } } },
        createdAt: true,
        updatedAt: true,
      },
    });

    if (!privateChat) throw new MultiLanguageException(LanguageErrorEnum.PRIVATE_CHAT_NOT_FOUND);

    return privateChat;
  }

  async healthSpecialistFetchPrivateChatMessages(
    user: UserPlayloadResponse,
    searchWithCursorPaginationDto: SearchWithCursorPaginationDto,
    id: number,
  ): Promise<PaginationCursorPrivateChatMessageResponse> {
    const { oldCursorId, sizePerPage, search } = searchWithCursorPaginationDto;

    const fullTextSearch = search ? this.commonService.addLeadingPlusAndTrailingWildcard(search) : undefined;

    await this.isHealthSpecialistInPrivateChat(id, user.id);

    const [messages, messagePagination] = await this.prismaService
      .extendPagination()
      .privateChatMessages.paginate({
        where: {
          privateChatId: id,
          message: { search: fullTextSearch },
        },
        select: {
          id: true,
          message: true,
          file: true,
          reply: {
            select: {
              id: true,
              file: {
                select: {
                  blobName: true,
                  extension: true,
                  name: true,
                },
              },
              sender: {
                select: {
                  id: true,
                  personalInformation: { select: { firstName: true, lastName: true } },
                },
              },
              message: true,
              createdAt: true,
            },
          },
          sender: {
            select: {
              id: true,
              personalInformation: {
                select: {
                  firstName: true,
                  lastName: true,
                },
              },
            },
          },
          receiverId: true,
          createdAt: true,
        },
        orderBy: {
          createdAt: 'desc',
        },
      })
      .withCursor({
        limit: sizePerPage,
        after: oldCursorId ? `${oldCursorId}` : null,
        getCursor({ id }) {
          return id.toString();
        },
      });

    this.messageViewedPrivateChat(id, user.id);

    const { hasNextPage, hasPreviousPage, endCursor, startCursor } = messagePagination;

    return { data: messages, hasNextPage, hasPreviousPage, endCursor: Number(endCursor), startCursor: Number(startCursor) };
  }

  async disablePrivateChat(user: UserPlayloadResponse, id: number): Promise<PrivateChatSimpleResponse> {
    await this.isHealthSpecialistInPrivateChat(id, user.id);

    return await this.prismaService.privateChats.update({
      where: { id },
      select: {
        id: true,
        isDisabled: true,
      },
      data: {
        isDisabled: true,
      },
    });
  }

  async enablePrivateChat(user: UserPlayloadResponse, id: number): Promise<PrivateChatSimpleResponse> {
    await this.isHealthSpecialistInPrivateChat(id, user.id);

    return await this.prismaService.privateChats.update({
      where: { id },
      select: {
        id: true,
        isDisabled: true,
      },
      data: {
        isCraving: false,
        isDisabled: false,
      },
    });
  }

  async updatePrivateChatToCraving(user: UserPlayloadResponse, id: number, transaction: PrismaTransactionClient): Promise<PrivateChatSimpleResponse> {
    const transactionOperation = transaction ?? this.prismaService;
    await this.isPatientInPrivateChat(user.id, id, transactionOperation);

    return await transactionOperation.privateChats.update({
      where: { id },
      data: { isCraving: true },
      select: {
        id: true,
        isDisabled: true,
      },
    });
  }

  async isHealthSpecialistInPrivateChat(privateChatId: number, healthSpecialistUserId: number): Promise<boolean> {
    await this.isPrivateChatByIdExists(privateChatId, null);

    const privateChat = await this.prismaService.privateChats.findUnique({
      where: {
        id: privateChatId,
        healthSpecialistUserId,
      },
    });
    if (!privateChat) throw new MultiLanguageException(LanguageErrorEnum.HS_NOT_IN_PRIVATE_CHAT);
    return true;
  }

  async isHealthSpecialistInPrivateChats(privateChatIds: number[], healthSpecialistUserId: number): Promise<boolean> {
    const privateChat = await this.prismaService.privateChats.aggregate({
      where: {
        id: { in: privateChatIds },
        healthSpecialistUserId,
      },
      _count: {
        _all: true,
      },
    });
    if (privateChat._count._all < privateChatIds.length) throw new MultiLanguageException(LanguageErrorEnum.HS_NOT_IN_PRIVATE_CHAT);

    return true;
  }

  async privateChatByUserId(userId: number): Promise<PrivateChatResponse> {
    const userPrivateChat = await this.prismaService.user.findUnique({
      where: {
        id: userId,
      },
      select: {
        patientPrivateChat: {
          select: {
            id: true,
            isDisabled: true,
            patientUserId: true,
            isCraving: true,
            healthSpecialist: {
              select: {
                id: true,
                personalInformation: { select: { firstName: true, lastName: true } },
              },
            },
            createdAt: true,
            updatedAt: true,
          },
        },
      },
    });

    return userPrivateChat.patientPrivateChat;
  }

  async isPrivateChatByIdExists(chatId: number, transaction: PrismaTransactionClient): Promise<PrivateChats> {
    const transactionOperation = transaction ?? this.prismaService;

    const privateChat = await transactionOperation.privateChats.findUnique({
      where: {
        id: chatId,
      },
    });

    if (!privateChat) throw new MultiLanguageException(LanguageErrorEnum.PRIVATE_CHAT_NOT_FOUND);

    return privateChat;
  }

  async isPrivateChatByUserIdNotExist(userId: number): Promise<boolean> {
    const isPrivateChatExists = await this.prismaService.privateChats.findFirst({
      where: {
        patientUserId: userId,
      },
    });

    if (isPrivateChatExists) throw new MultiLanguageException(LanguageErrorEnum.PRIVATE_CHAT_WITH_USER_EXISTS);
    return true;
  }

  async isPatientInPrivateChat(userId: number, chatId: number, transaction: PrismaTransactionClient): Promise<boolean> {
    const transactionOperation = transaction ?? this.prismaService;

    await this.isPrivateChatByIdExists(chatId, transactionOperation);

    const privateChat = await transactionOperation.privateChats.findUnique({
      where: {
        id: chatId,
        patientUserId: userId,
      },
      select: { id: true },
    });
    if (!privateChat) throw new MultiLanguageException(LanguageErrorEnum.USER_NOT_IN_PRIVATE_CHAT);
    return true;
  }

  async userPrivateChat(userId: number, chatId: number): Promise<PrivateChats> {
    await this.isPrivateChatByIdExists(chatId, null);

    const privateChat = await this.prismaService.privateChats.findUnique({
      where: {
        id: chatId,
        OR: [
          {
            patientUserId: userId,
          },
          {
            healthSpecialistUserId: userId,
          },
        ],
      },
    });

    if (!privateChat) throw new MultiLanguageException(LanguageErrorEnum.USER_NOT_IN_PRIVATE_CHAT);
    return privateChat;
  }

  async messageViewedPrivateChat(privateChatId: number, participantId: number): Promise<boolean> {
    const isViewed = await this.prismaService.messagesOnUsersOnPrivateChats.findFirst({
      where: {
        participantId,
        privateChatId,
      },
    });
    if (isViewed) {
      await this.prismaService.$transaction([
        this.prismaService.messagesOnUsersOnPrivateChats.delete({
          where: {
            participantId_privateChatId: {
              participantId,
              privateChatId,
            },
          },
        }),

        this.prismaService.messagesOnUsersOnPrivateChats.create({
          data: {
            participantId,
            privateChatId,
          },
        }),
      ]);
    } else {
      await this.prismaService.messagesOnUsersOnPrivateChats.create({
        data: {
          participantId,
          privateChatId,
        },
      });
    }

    return true;
  }
}
