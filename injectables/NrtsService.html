import { NrtsDetailService } from 'src/modules/nrts-detail/nrts-detail.service';
import { InjectQueue } from '@nestjs/bullmq';
import { Inject, Injectable, forwardRef } from '@nestjs/common';
import { Queue } from 'bullmq';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { QueueEnum } from 'src/common/enums/queues.enum';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';
import { CommonService } from 'src/common/services/common.service';
import { PrismaService } from 'src/common/services/prisma.service';
import { GroupsService } from 'src/modules/groups/groups.service';
import { AddNrtDto } from 'src/modules/nrts/dto/add-nrt.dto';
import { UpdateNrtDto } from 'src/modules/nrts/dto/update-nrt.dto';
import { UpdatePriorityDto } from 'src/modules/nrts/dto/update-priority.dto';
import { NrtResponse } from 'src/modules/nrts/response/nrt.response';
import { RedisService } from 'src/common/services/redis.service';
import { LanguageEnum, Prisma } from '@prisma/client';
import { EnvironmentService } from 'src/common/services/environment.service';
import { plainToInstance } from 'class-transformer';
import { NrtOnlyPriorityResponse } from 'src/modules/sc-plans/response/nrt-only-priority.response';
import { PatientService } from 'src/modules/patients/patient.service';
import { UsersService } from 'src/modules/users/users.service';
import { NrtDetailResponse } from 'src/modules/nrts-detail/response/nrt-detail.response';

@Injectable()
export class NrtsService {
  constructor(
    private readonly prismaService: PrismaService,
    @Inject(forwardRef(() => GroupsService)) private readonly groupsService: GroupsService,
    private readonly commonService: CommonService,
    @InjectQueue(QueueEnum.RESET_NRT_PRIORITIES_QUEUE) private resetNrtPrioritiesQueue: Queue,
    private readonly nrtsDetailService: NrtsDetailService,
    private redisService: RedisService,
    private environmentService: EnvironmentService,
    @Inject(forwardRef(() => PatientService)) private patientService: PatientService,
    @Inject(forwardRef(() => UsersService)) private usersService: UsersService,
  ) {}

  async onModuleInit() {
    await this.resetNrtPrioritiesQueue.drain();

    await this.resetNrtPrioritiesQueue.add(
      QueueEnum.RESET_NRT_PRIORITIES_QUEUE,
      {},

      {
        //repeat every Saturday At 00:00
        repeat: {
          pattern: '0 0 0 * * 6',
        },
      },
    );
  }

  async nrtList(language: LanguageEnum): Promise<NrtResponse[]> {
    const nrts = await this.prismaService.nrt.findMany({
      select: {
        id: true,
        nrtDetail: {
          select: {
            id: true,
            nrtType: true,
            nrtDetailsTranslation: {
              where: {
                language: {
                  code: language,
                },
              },
              select: {
                gum: true,
                patch: true,
              },
            },
          },
        },
        priority: true,
        isEditable: true,
        targetGroup: {
          select: {
            id: true,
            name: true,
          },
        },
      },
      orderBy: {
        priority: 'desc',
      },
    });

    const serializedNrts = plainToInstance(NrtResponse, nrts, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    return serializedNrts;
  }

  async addNrt(language: LanguageEnum, addNrtDto: AddNrtDto): Promise<NrtResponse> {
    const { targetGroupId, nrtTypeId } = addNrtDto;
    let newPriority = null;

    const groupNrt = await this.groupsService.groupNrt(language, targetGroupId, null);
    if (groupNrt?.nrts) throw new MultiLanguageException(LanguageErrorEnum.GROUP_HAS_NRT);

    await this.nrtsDetailService.getNrtDetailById(language, nrtTypeId);

    const higherPriority = await this.prismaService.nrt.aggregate({
      _max: {
        priority: true,
      },
    });

    if (Number.isInteger(higherPriority._max.priority)) newPriority = Math.ceil(higherPriority._max.priority) + 1;
    else newPriority = Math.ceil(higherPriority._max.priority);

    const addedNrt = await this.prismaService.nrt.create({
      data: {
        nrtDetail: {
          connect: {
            id: nrtTypeId,
          },
        },
        priority: newPriority,
        targetGroup: {
          connect: { id: targetGroupId },
        },
      },
      select: {
        id: true,
        nrtDetail: {
          select: {
            id: true,
            nrtType: true,
            nrtDetailsTranslation: {
              where: {
                language: {
                  code: language,
                },
              },
              select: {
                gum: true,
                patch: true,
              },
            },
          },
        },
        priority: true,
        isEditable: true,
        targetGroup: {
          select: {
            id: true,
            name: true,
          },
        },
      },
    });

    const serializedNrt = plainToInstance(NrtResponse, addedNrt, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    return serializedNrt;
  }

  async updateNrt(language: LanguageEnum, updateNrtDto: UpdateNrtDto, id: number): Promise<NrtResponse> {
    const { targetGroupId, nrtTypeId } = updateNrtDto;

    if (nrtTypeId) await this.nrtsDetailService.getNrtDetailById(language, nrtTypeId);

    if (targetGroupId) {
      const groupNrt = await this.groupsService.groupNrt(language, targetGroupId, id);

      if (groupNrt) throw new MultiLanguageException(LanguageErrorEnum.GROUP_HAS_NRT);
      // if (!groupNrt && Number(groupNrt?.targetGroup?.id) !== targetGroupId) throw new MultiLanguageException(LanguageErrorEnum.GROUP_HAS_NRT);
    }

    const nrt = await this.prismaService.nrt.update({
      where: { id },
      data: {
        ...(nrtTypeId && {
          nrtDetail: {
            connect: { id: nrtTypeId },
          },
        }),

        targetGroup: {
          connect: { id: targetGroupId },
        },
      },
      select: {
        id: true,
        nrtDetail: {
          select: {
            id: true,
            nrtType: true,
            nrtDetailsTranslation: {
              where: {
                language: {
                  code: language,
                },
              },
              select: {
                gum: true,
                patch: true,
              },
            },
          },
        },
        priority: true,
        isEditable: true,
        targetGroup: {
          select: {
            id: true,
            name: true,
          },
        },
      },
    });

    const serializedNrt = plainToInstance(NrtResponse, nrt, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    return serializedNrt;
  }

  async userNrt(language: LanguageEnum, userId: number): Promise<NrtDetailResponse> {
    await this.patientService.isUserPatient(userId);
    const user = await this.usersService.findPatientByUserId(userId);

    //manual nrt if admin asigned nrt from dashboard manually
    let currentPatientNrtDetail = await this.patientSignedNrt(language, Number(user.patient.id));

    //if patient assigned to nrt we fetch it from db otherwise we query thepriorities to find out his most relevant nrt
    if (!currentPatientNrtDetail) currentPatientNrtDetail = await this.QueryPatientNrt(language, Number(user.patient.id));

    return currentPatientNrtDetail;
  }

  async updateNrtPriority(language: LanguageEnum, updatePriorityDto: UpdatePriorityDto, id: number): Promise<NrtResponse> {
    const { topNrtId, bottomNrtId } = updatePriorityDto;
    let newPriority = null;
    let topPriorityNrt = null;

    const nrtById = await this.getNrtById(language, id);
    if (!nrtById) throw new MultiLanguageException(LanguageErrorEnum.NRT_NOT_FOUND);

    return this.prismaService.$transaction(async (tx) => {
      const bottomPriorityNrt = await tx.nrt.findUnique({
        where: { id: bottomNrtId },
        select: {
          id: true,
          priority: true,
        },
      });
      if (topNrtId) {
        topPriorityNrt = await tx.nrt.findUnique({
          where: { id: topNrtId },
          select: {
            id: true,
            priority: true,
          },
        });
        if (!topPriorityNrt) throw new MultiLanguageException(LanguageErrorEnum.NRT_NOT_FOUND);
      }

      if (!bottomPriorityNrt) throw new MultiLanguageException(LanguageErrorEnum.NRT_NOT_FOUND);

      if (topNrtId && topPriorityNrt?.priority < bottomPriorityNrt.priority) throw new MultiLanguageException(LanguageErrorEnum.TOP_PRIORITY_MUST_HIGHER);

      const bottomPriority = bottomPriorityNrt.priority;
      const topPriority = topPriorityNrt?.priority;

      if (topPriority) newPriority = this.commonService.reorderItem(bottomPriority, topPriority);
      else {
        //if topPriority is null it means the item is replaced to the top
        //if the NRT moved to the top we want the priority to be the first integer after the biggest priority
        if (Number.isInteger(bottomPriority)) newPriority = Math.ceil(bottomPriority) + 1;
        else newPriority = Math.ceil(bottomPriority);
      }

      const nrt = await tx.nrt.update({
        where: { id },
        data: {
          priority: newPriority,
        },
        select: {
          id: true,
          nrtDetail: {
            select: {
              id: true,
              nrtType: true,
              nrtDetailsTranslation: {
                where: {
                  language: {
                    code: language,
                  },
                },
                select: {
                  gum: true,
                  patch: true,
                },
              },
            },
          },
          priority: true,
          isEditable: true,
          targetGroup: {
            select: {
              id: true,
              name: true,
            },
          },
        },
      });

      const serializedNrt = plainToInstance(NrtResponse, nrt, {
        excludeExtraneousValues: true,
        enableImplicitConversion: true,
      });

      return serializedNrt;
    });
  }

  async updatePatientNrt(language: LanguageEnum, id: number, userId: number): Promise<NrtDetailResponse> {
    await this.patientService.isUserPatient(userId);

    const nrt = await this.nrtsDetailService.getNrtDetailById(language, id);

    await this.prismaService.user.update({
      where: {
        id: userId,
      },
      data: {
        patient: {
          update: {
            nrtDetail: {
              connect: {
                id,
              },
            },
          },
        },
      },
    });

    return nrt;
  }

  async getDefaultNrt(language: LanguageEnum): Promise<NrtResponse> {
    const getDefaultNrtRedisKey = `getDefaultNrt?language${language}`;

    const cacheResults = await this.redisService.getResponseKey(getDefaultNrtRedisKey);
    if (cacheResults) return cacheResults;

    const nrt = await this.prismaService.nrt.findFirst({
      where: { isEditable: true },
      select: {
        id: true,
        nrtDetail: {
          select: {
            id: true,
            nrtType: true,
            nrtDetailsTranslation: {
              where: {
                language: {
                  code: language,
                },
              },
              select: {
                gum: true,
                patch: true,
              },
            },
          },
        },
        priority: true,
        isEditable: true,
        targetGroup: {
          select: {
            id: true,
            name: true,
          },
        },
      },
    });

    if (!nrt) throw new MultiLanguageException(LanguageErrorEnum.NRT_NOT_FOUND);

    const serializedNrt = plainToInstance(NrtResponse, nrt, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    this.redisService.setResponseKey(getDefaultNrtRedisKey, serializedNrt, this.environmentService.redisConfig.longerTimeToLive);

    return serializedNrt;
  }

  //this the result of the asigned nrt we get from the db relation
  async patientSignedNrt(language: LanguageEnum, patientId: number): Promise<NrtDetailResponse> {
    const isPatientHasNrtRedisKey = `isPatientHasNrt?patientId${patientId}`;
    let nrt: NrtDetailResponse = null;

    const cacheResults = await this.redisService.getResponseKey(isPatientHasNrtRedisKey);
    if (cacheResults) return cacheResults;

    const patientNrt = await this.prismaService.patient.findFirst({
      where: { id: patientId },
      select: {
        nrtDetailId: true,
      },
    });

    if (patientNrt.nrtDetailId) nrt = await this.nrtsDetailService.getNrtDetailById(language, Number(patientNrt.nrtDetailId));

    this.redisService.setResponseKey(isPatientHasNrtRedisKey, nrt, this.environmentService.redisConfig.longTimeToLive);

    return nrt;
  }

  //this is needed when the patient has no nrt linked to
  async QueryPatientNrt(language: LanguageEnum, patientId: number): Promise<NrtDetailResponse> {
    const patientGroups = await this.groupsService.fetchPatientGroups(patientId);

    const patientGroupsIds = patientGroups.map((patientGroup) => patientGroup.id);
    const isFiltringTargetGroup = Prisma.sql`${patientGroupsIds.length > 0 ? Prisma.sql`OR targetGroupId IN (${Prisma.join(patientGroupsIds)})` : Prisma.empty}`;

    //i m using sql here because prisma  cant aggregate by _MAX(priority) only and then fetch its id
    const nrtMaxPriority: NrtOnlyPriorityResponse[] = await this.prismaService.$queryRaw`
      SELECT id, MAX(priority) as maxPriority,targetGroupId,isEditable

      FROM NRT nrt

      WHERE isEditable=0   ${isFiltringTargetGroup}

      GROUP BY id
      ORDER BY maxPriority DESC
      LIMIT 1;`;
    const nrt = await this.getNrtById(language, Number(nrtMaxPriority[0].id));
    return nrt.nrtDetail;
  }

  async isNrtEditable(id: number): Promise<boolean> {
    const nrt = await this.prismaService.nrt.findUnique({
      where: { id },
      select: {
        id: true,
        isEditable: true,
      },
    });
    if (!nrt) throw new MultiLanguageException(LanguageErrorEnum.NRT_NOT_FOUND);
    if (!nrt.isEditable) throw new MultiLanguageException(LanguageErrorEnum.ITEM_NOT_EDIABLE);
    return true;
  }
}
