import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';
import { PrismaService } from '../../common/services/prisma.service';
import { Inject, Injectable, forwardRef } from '@nestjs/common';
import { UserAdminResponse } from 'src/modules/admins/response/fetch-admin.response';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { LanguageEnum, RoleRestrictionEnum } from '@prisma/client';
import { CreateFacilityAdminDto } from './dtos/create-facility-admin.dto';
import { BcryptService } from '../auth/bcrypt.service';
import { RedisService } from 'src/common/services/redis.service';
import { EnvironmentService } from 'src/common/services/environment.service';
import { UpdateFacilityAdminDto } from 'src/modules/admins/dtos/update-facility-admin.dto';
import { OnboardingService } from 'src/modules/onboarding/onboarding.service';
import { PermissionsService } from 'src/modules/permissions/permissions.service';
import { CreateSystemAdminDto } from 'src/modules/admins/dtos/create-system-admin.dto';
import { RoleEnum } from 'src/common/enums/role.enum';
import { UsersService } from 'src/modules/users/users.service';
import { FacilityRoleService } from 'src/modules/facility-role/facility-role.service';
import { FacilityGroupService } from 'src/modules/facility-group/facility-group.service';
import { UpdateSystemAdminDto } from 'src/modules/admins/dtos/update-system-admin.dto';
import { SettingsData } from 'src/common/consts/settings-data';
import { SettingService } from '../settings/settings.service';
import { AdminSimpleResponse } from './response/fetch-admins-simple.response';

@Injectable()
export class AdminsService {
  constructor(
    private settingsService: SettingService,
    private prismaService: PrismaService,
    private bcryptService: BcryptService,
    private redisService: RedisService,
    private environmentService: EnvironmentService,
    private onboardingService: OnboardingService,
    private permissionsService: PermissionsService,
    @Inject(forwardRef(() => UsersService)) private usersService: UsersService,
    @Inject(forwardRef(() => FacilityRoleService)) private facilityRoleService: FacilityRoleService,
    private facilityGroupService: FacilityGroupService,
  ) {}

  async fetchAdminProfile(language: LanguageEnum, user: UserPlayloadResponse): Promise<UserAdminResponse> {
    const authedUser = await this.prismaService.user.findUnique({
      where: {
        id: user.id,
      },
      select: {
        id: true,
        qid: true,
        personalInformation: {
          select: {
            firstName: true,
            lastName: true,
          },
        },
        admin: {
          select: {
            department: true,
            role: {
              select: {
                id: true,
                name: true,
                isActive: true,
                restriction: true,
              },
            },
          },
        },
      },
    });

    const scopes = await this.permissionsService.fetchScopesByRole(language, Number(authedUser.admin.role.id));

    const scopesResult = {
      ...authedUser,
      admin: { ...authedUser.admin, role: { ...authedUser.admin.role, scopes } },
      personalInformation: { ...authedUser.personalInformation },
    };

    return scopesResult;
  }

  async createFacilityAdmin(language: LanguageEnum, createFacilityAdminDto: CreateFacilityAdminDto): Promise<UserAdminResponse> {
    const { email, location, personalInformation, phone, qid, departement, roleId } = createFacilityAdminDto;

    await this.usersService.isQidNotExist(qid, null);
    await this.onboardingService.isEmailNotExist(email, null);

    const HashedPassword = await this.bcryptService.randomPassword(5);

    if (roleId) {
      const userRole = await this.facilityRoleService.fetchFacilityRole(language, Number(roleId));
      if (userRole.restriction !== RoleRestrictionEnum.CUSTOM) throw new MultiLanguageException(LanguageErrorEnum.FORBIDDEN_RESOURCES);
    }

    return await this.prismaService.$transaction(async (prisma) => {
      const newAdmin = await prisma.user.create({
        data: {
          qid: qid,
          password: HashedPassword,
          location: location,
          personalInformation: {
            create: {
              ...personalInformation,
              email: email,
              phoneNumber: phone,
            },
          },
          admin: {
            create: {
              roleId: roleId,
              department: departement,
            },
          },
        },
        select: {
          id: true,
          admin: {
            select: {
              id: true,
              department: true,
              role: {
                select: {
                  id: true,
                  isActive: true,
                  name: true,
                  restriction: true,
                },
              },
            },
          },
          personalInformation: {
            select: {
              firstName: true,
              lastName: true,
            },
          },
          qid: true,
        },
      });

      // Assign settings to the new admin
      const settingData = SettingsData.admin;
      await this.settingsService.assignSettings(settingData, newAdmin.id, prisma);

      const scopes = await this.permissionsService.fetchScopesByRole(language, Number(roleId));

      return {
        admin: {
          ...newAdmin.admin,
          role: roleId ? { ...newAdmin.admin.role, scopes } : {},
        },
        ...newAdmin,
        personalInformation: {
          firstName: newAdmin.personalInformation.firstName,
          lastName: newAdmin.personalInformation.lastName,
        },
      };
    });
  }

  async createSystemAdmin(language: LanguageEnum, createSystemAdminDto: CreateSystemAdminDto): Promise<UserAdminResponse> {
    const { email, location, personalInformation, phone, qid, departement, facilityGroupId } = createSystemAdminDto;

    if (qid) await this.usersService.isQidNotExist(qid, null);
    if (email) await this.onboardingService.isEmailNotExist(email, null);

    await this.facilityGroupService.isFacilityGroupByIdExists(Number(facilityGroupId));

    const HashedPassword = await this.bcryptService.randomPassword(5);

    return this.prismaService.$transaction(async (prisma) => {
      const newAdmin = await prisma.user.create({
        data: {
          qid: qid,
          password: HashedPassword,
          location: location,
          personalInformation: {
            create: {
              ...personalInformation,
              email: email,
              phoneNumber: phone,
            },
          },
          admin: {
            create: {
              role: {
                connect: {
                  name: RoleEnum.SYSADMIN,
                },
              },
              department: departement,
            },
          },
        },
        select: {
          // Correctly placed outside the data object
          admin: {
            select: {
              id: true,
              department: true,
              role: {
                select: {
                  id: true,
                  isActive: true,
                  name: true,
                  restriction: true,
                },
              },
            },
          },
          personalInformation: {
            select: {
              firstName: true,
              lastName: true,
            },
          },
          qid: true,
          id: true,
        },
      });

      // Assign settings to the new admin
      const settingData = SettingsData.admin;
      await this.settingsService.assignSettings(settingData, newAdmin.id, prisma);

      const scopes = await this.permissionsService.fetchSystemAdminScopes(language);

      return {
        admin: {
          ...newAdmin.admin,
          role: facilityGroupId ? { ...newAdmin.admin.role, scopes } : {},
        },
        ...newAdmin,
        personalInformation: {
          firstName: newAdmin.personalInformation.firstName,
          lastName: newAdmin.personalInformation.lastName,
        },
      };
    });
  }

  async isAdmin(userId: number | bigint): Promise<boolean> {
    const isAdminRedisKey = `isAdmin/${userId}`;
    const cacheResults = await this.redisService.getResponseKey(isAdminRedisKey);
    if (cacheResults) return true;

    const user = await this.prismaService.user.findUnique({
      where: {
        id: userId,
      },
      select: {
        admin: {
          select: {
            id: true,
          },
        },
      },
    });
    if (!user) throw new MultiLanguageException(LanguageErrorEnum.USER_NOT_FOUND);

    if (!user?.admin) throw new MultiLanguageException(LanguageErrorEnum.FORBIDDEN_RESOURCES);

    this.redisService.setResponseKey(isAdminRedisKey, 'true', this.environmentService.redisConfig.longerTimeToLive);

    return true;
  }

  async isAdminHasRole(adminId: number): Promise<boolean> {
    const roleByAdminId = await this.prismaService.admin.findUnique({
      where: { id: adminId },
      select: {
        id: true,
        roleId: true,
      },
    });
    if (!roleByAdminId.roleId) throw new MultiLanguageException(LanguageErrorEnum.ADMIN_HAS_NO_ROLE);

    return true;
  }
  async fetchAdminsSimple(user: UserPlayloadResponse): Promise<AdminSimpleResponse[]> {
    const isAdmin = await this.isAdmin(user.id);
    if (!isAdmin) throw new MultiLanguageException(LanguageErrorEnum.UNAUTHORIZED);
    const admins = await this.prismaService.admin.findMany({
      where: {
        healthSpecialist: null,
      },
      select: {
        user: {
          select: {
            id: true,

            personalInformation: {
              select: {
                firstName: true,
                lastName: true,
              },
            },
          },
        },
      },
    });
    return admins;
  }
}
