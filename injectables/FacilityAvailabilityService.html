import { Injectable } from '@nestjs/common';
import { PrismaService } from 'src/common/services/prisma.service';
import { AvailabilitiesService } from '../availabilities/availabilities.service';
import { EditFacilityAvailabilitiesDto } from './dto/facility.availability.dto';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { EditAvailability } from './dto/edit.facility.availability.dto';
import { DaysEnum } from '@prisma/client';
import { TimeSlotDto } from '../availabilities/dto/add-availability-per-day';
import { PrismaTransactionClient } from 'src/common/types/transaction.type';

@Injectable()
export class FacilityAvailabilityService {
  constructor(
    private readonly prismaService: PrismaService,
    private readonly hsAvailability: AvailabilitiesService,
  ) {}

  async editAvailabilities(availabilitiesDto: EditFacilityAvailabilitiesDto, facilityId: bigint): Promise<EditFacilityAvailabilitiesDto> {
    for (const availability of availabilitiesDto.availabilities) {
      if (await !this.hsAvailability.validateTimeSlots(availability)) throw new MultiLanguageException(LanguageErrorEnum.NON_VALID_TIME_SLOTS);

      //const availabilityId = await this.findAvailabilityId(availability.dayOfWeek, facilityId); //add costom exception

      const { dayOfWeek, timeSlots } = availability;
      const closed = timeSlots.length === 0 || !timeSlots ? true : false;

      await this.prismaService.facilityAvailability.upsert({
        where: {
          facilityId_dayOfWeek: {
            facilityId,
            dayOfWeek: dayOfWeek,
          },
        },
        update: {
          isClosed: closed,
          timeSlots: {
            deleteMany: {}, // Delete existing time slots
            create: timeSlots.map((timeSlot) => ({
              startTime: timeSlot.startTime,
              endTime: timeSlot.endTime,
            })),
          },
        },
        create: {
          dayOfWeek,
          isClosed: closed,
          facility: { connect: { id: facilityId } },
          timeSlots: {
            create: timeSlots.map((timeSlot) => ({
              startTime: timeSlot.startTime,
              endTime: timeSlot.endTime,
            })),
          },
        },
      });
    }
    return await this.findAvailabilities(facilityId);
  }

  async updateOnlyIsClosed(availabilityId: bigint, isClosed: boolean) {
    await this.prismaService.facilityAvailability.update({
      where: { id: availabilityId },
      data: { isClosed },
    });
  }

  async findAvailability(dayOfWeek: DaysEnum, facilityId: bigint): Promise<EditAvailability> {
    // Find existing availability for the facility

    const existingAvailability = await this.prismaService.facilityAvailability.findFirst({
      where: { facilityId: facilityId, dayOfWeek },
      include: { timeSlots: true }, // Include associated timeSlots
    });
    return existingAvailability;
  }
  async findAvailabilities(facilityId: bigint): Promise<EditFacilityAvailabilitiesDto> {
    // Find existing availability for the facility
    const availabilities = await this.prismaService.facilityAvailability.findMany({
      where: {
        facilityId,
      },
      include: { timeSlots: { orderBy: { startTime: 'asc' } } },
    });
    availabilities.forEach((availability) => {
      availability.timeSlots.sort((a, b) => {
        // Assuming startTime is in the format 'HH:mm'
        // Convert time strings to Date objects
        const timeA = new Date(`2000-01-01T${a.startTime}`);
        const timeB = new Date(`2000-01-01T${b.startTime}`);

        // Compare the Date objects
        if (timeA < timeB) return -1;
        if (timeA > timeB) return 1;
        return 0;
      });
    });
    return { availabilities };
  }
  async findAvailabilityId(dayOfWeek: DaysEnum, facilityId: bigint): Promise<bigint> {
    // Find existing availability for the facility
    const existingAvailability = await this.prismaService.facilityAvailability.findFirst({
      where: { facilityId, dayOfWeek },
      include: { timeSlots: true }, // Include associated timeSlots
    });
    return existingAvailability?.id;
  }
  async seedAvailabilities(id: bigint, transaction?: PrismaTransactionClient): Promise<boolean> {
    const startTime = '07:00';
    const endTime = '23:00';
    const timeSlots: TimeSlotDto[] = [{ startTime, endTime }];
    let data;
    const dayValues = Object.values(DaysEnum);

    // Use the transaction if available, otherwise use the default Prisma client
    const prismaClient = transaction ?? this.prismaService;

    for (const dayOfWeek of dayValues) {
      if (dayOfWeek === 'FRIDAY' || dayOfWeek === 'SATURDAY') {
        data = {
          dayOfWeek: dayOfWeek,
          is24H: false,
          isClosed: true,
          facility: { connect: { id } },
        };
      } else {
        data = {
          dayOfWeek: dayOfWeek,
          is24H: false,
          isClosed: false,
          facility: { connect: { id } },
          timeSlots: {
            create: timeSlots.map((timeSlot) => ({
              startTime: timeSlot.startTime,
              endTime: timeSlot.endTime,
            })),
          },
        };
      }

      const existingRecord = await prismaClient.facilityAvailability.findFirst({
        where: {
          dayOfWeek,
          facilityId: id,
        },
      });

      // If the record exists, use its id for the upsert operation
      if (existingRecord) {
        await prismaClient.facilityAvailability.update({
          where: {
            id: existingRecord.id, // Use the id of the existing record
          },
          data,
        });
      } else {
        // If the record does not exist, create it
        await prismaClient.facilityAvailability.create({
          data,
        });
      }
    }
    return true;
  }
}
