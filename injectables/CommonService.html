import { Injectable } from '@nestjs/common';
import { MassBodyEnum } from '@prisma/client';
import { DateDto } from 'src/common/dto/date.dto';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';
import { randomBytes, randomInt } from 'crypto';

@Injectable()
export class CommonService {
  // constructor(private environmentService: EnvironmentService) {}

  addLeadingPlusAndTrailingWildcard(sentence: string): string | undefined {
    //this clean unsupported symbols from
    const filteredSentence = this.stripMysqlBooleanOperators(sentence);

    // this splits the search string by space
    const words = filteredSentence.split(' ');

    //to filter out spaces(empty strings)
    const filteredWords = words.filter((word) => word);

    //handling empty sentence
    if (filteredSentence.trim().length === 0) return undefined;

    //if we have multiple words in the sentence
    if (filteredWords.length > 1) {
      const result = filteredWords.join('* +');
      return `+${result}*`.trim();
    }

    //if we have 1 word in the sentence
    return `+${filteredSentence}*`.trim();
    //this adds regex to the full text search
    //* to make the search works like a prefix
    //+ indicates that this word must be present in each row that is returned
    //documentation link https://dev.mysql.com/doc/refman/8.0/en/fulltext-boolean.html
  }

  stripMysqlBooleanOperators(str: string) {
    return str.split('').reduce((acc, char) => {
      return ['+', '-', '>', '<', '@', '(', ')', '~', '*'].includes(char) ? acc + ' ' : acc + char;
    }, '');
  }

  objectParser(object: any) {
    return JSON.parse(
      JSON.stringify(
        object,
        (_, value) => (typeof value === 'bigint' ? Number(value) : value), //return bigInt=>number & everything else unchanged
      ),
    );
  }

  BigIntParser(numberToParse: bigint): number {
    return Number.parseInt(numberToParse.toString());
  }

  getMax(array: number[]): number {
    const reduceMax = (a, b) => Math.max(a, b);

    const max = array.reduce(reduceMax, array[0]);
    return max;
  }

  getMin(array: number[]): number {
    const getMaxA = (a, b) => Math.min(a, b);

    const min = array.reduce(getMaxA, array[0]);
    return min;
  }

  isArrayHasDuplicateValues(array: string[]): boolean {
    return new Set(array).size !== array.length;
  }

  BMI(height: number, weight: number): MassBodyEnum {
    //we get height with centimetre
    const heightMeter = height / 100;
    const massBodyIndex = weight / (heightMeter * heightMeter);

    switch (true) {
      case massBodyIndex < 18.5:
        return MassBodyEnum.UNDERWEIGHT;

      case massBodyIndex < 24.9:
        return MassBodyEnum.HEALTHY;

      case massBodyIndex < 29.9:
        return MassBodyEnum.OVERWEIGHT;

      default:
        return MassBodyEnum.OBESE;
    }
  }

  getCurrentUtcDate(): Date {
    const currentTime = new Date();

    const yearUtc = currentTime.getUTCFullYear();
    const monthUtc = currentTime.getUTCMonth();
    const dayUtc = currentTime.getUTCDate();
    const hoursUtc = currentTime.getUTCHours();
    const minutesUtc = currentTime.getUTCMinutes();
    const secondsUtc = currentTime.getUTCSeconds();
    //milliseconds? const millisecondsUtc = currentTime.getUTCSeconds();
    const date = new Date(Date.UTC(yearUtc, monthUtc, dayUtc, hoursUtc, minutesUtc, secondsUtc));

    return date;
  }

  adjustDate(): Date {
    const givenDate = new Date();

    givenDate.setTime(givenDate.getTime() - new Date().getTimezoneOffset() * 60 * 1000);
    return givenDate;
  }

  isDateBeforeNow(dateOfBirth: Date): boolean {
    if (dateOfBirth > this.adjustDate()) throw new MultiLanguageException(LanguageErrorEnum.NOT_VALID_DATE);
    return true;
  }

  isDateAfterNow(date: Date): boolean {
    if (date < this.adjustDate()) throw new MultiLanguageException(LanguageErrorEnum.NOT_VALID_DATE);
    return true;
  }
  //return the remaining time in milliseconds from now to the given date
  remainingTime(date: Date): number {
    const targetTime = new Date(date);
    const delay = Math.abs(Number(targetTime) - Number(this.adjustDate()));
    return delay;
  }

  //This function is used to determin the priority(weight) of an item(like NRT) in a list
  reorderItem(previousIndex: number, nextIndex: number): number {
    return (nextIndex - previousIndex) / 2 + previousIndex;
  }

  datesDifferenceInDays(date1: Date, date2: Date): number {
    const Difference_In_Time = date2.getTime() - date1.getTime();

    // To calculate the difference by days
    return Math.round(Difference_In_Time / (1000 * 3600 * 24));
  }

  dateOfBirthToAge(date: Date): number {
    const currentDate = new Date();
    currentDate.setFullYear(currentDate.getFullYear() - new Date(date).getFullYear());
    return currentDate.getFullYear();
  }

  dateStartAndEndTime(date: Date): DateDto {
    const dateStartTime = new Date(date.setUTCHours(0, 0, 0, 0));
    const dateEndTime = new Date(date.setUTCHours(23, 59, 59, 999));

    return {
      date,
      dateEndTime,
      dateStartTime,
    };
  }

  isUrlNotDetected(text: string): boolean {
    const urlRegex = /(\b(https?|ftp|file):\/\/[-A-Z0-9+&@#/%?=~_|!:,.;]*[-A-Z0-9+&@#/%=~_|])/gi;

    const isUrldetected = text.match(urlRegex);
    if (isUrldetected) return true;
    return false;
  }

  randomString(halfPasswordLength: number): string {
    return randomBytes(halfPasswordLength).toString('hex');
  }

  randomNumber(): number {
    return randomInt(100000, 999999);
  }

  //how much milliseconds left of end of the day
  milliSecondsUntilEndOfDay(): number {
    const d = new Date();
    const h = d.getHours();
    const m = d.getMinutes();
    const s = d.getSeconds();
    const secondsUntilEndOfDate = 24 * 60 * 60 - h * 60 * 60 - m * 60 - s;
    const milliSecondsUntilEndOfDate = secondsUntilEndOfDate * 1000;
    return milliSecondsUntilEndOfDate;
  }

  mapObject(object: Record<any, boolean>) {
    const results = [];

    Object.keys(object).forEach(function (key) {
      if (object[key])
        results.push({
          [key]: true,
        });
    });

    return results;
  }
}
