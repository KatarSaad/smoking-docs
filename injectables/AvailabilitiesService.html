import { Injectable } from '@nestjs/common';
import { PrismaService } from 'src/common/services/prisma.service';
import { AddAvailabilitiesDto, AvailabilityPerDayDto } from './dto/add-availability-per-day';
import { Day, DaysEnum, HealthSpecialist } from '@prisma/client';
import { UserPlayloadResponse } from '../patients/response/user-playload.response';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { HealthSpecialistService } from '../health-specialists/health-specialist.service';
import { EditAvailability } from '../facility-availability/dto/edit.facility.availability.dto';

@Injectable()
export class AvailabilitiesService {
  constructor(
    private readonly prismaService: PrismaService,

    private healthSpecialistService: HealthSpecialistService,
  ) {}
  async addAvailability(user: UserPlayloadResponse, availabilitiesDto: AddAvailabilitiesDto): Promise<AvailabilityPerDayDto[]> {
    // Iterate through each availability object in availabilitiesDto
    for (const availabilityDto of availabilitiesDto.availabilities) {
      // Apply your logic to determine if you should update only non-working day
      if (this.shouldUpdateOnlyNonWorkingDay(availabilityDto)) {
        // Perform update for non-working day
        await this.updateNonWorkingDay(user, availabilityDto);
      } else {
        // Perform update for health specialist availability
        await this.updateHealthSpecialistAvailability(user, availabilityDto);
      }
    }
    // After updating, fetch updated availability
    const updatedAvailabilities = await this.getHealthSpecialistAvailabilities(user);

    return updatedAvailabilities;
  }

  async getHealthSpecialistAvailabilities(user: UserPlayloadResponse): Promise<AvailabilityPerDayDto[]> {
    const healthSpecialistId = user.healthSpecialistId;

    const availabilities = await this.prismaService.availabilityPerDay.findMany({
      where: { healthSpecialistId },
      select: {
        nonWorkingDay: true,

        dayOfWeek: true,
        timeSlots: { select: { endTime: true, startTime: true } },
      },
    });

    // Sort the time slots for each availability by start time
    availabilities.forEach((availability) => {
      availability.timeSlots.sort((a, b) => {
        if (a.startTime < b.startTime) return -1;
        if (a.startTime > b.startTime) return 1;
        return 0;
      });
    });

    return availabilities;
  }
  async getCurrentDayAvailability(userId: number): Promise<AvailabilityPerDayDto[] | null> {
    const healthSpecialistId = await this.healthSpecialistService.healthSpecialistIdByUserId(userId);

    // Get the current day of the week (0 for Sunday, 1 for Monday, ..., 6 for Saturday)
    const daysOfWeek = ['SUNDAY', 'MONDAY', 'TUESDAY', 'WEDNESDAY', 'THURSDAY', 'FRIDAY', 'SATURDAY'];
    const currentDayIndex = new Date().getDay();
    const currentDay = daysOfWeek[currentDayIndex];
    const currentDayOfWeek: DaysEnum = DaysEnum[currentDay];
    // Find the availability for the current day
    const availabilities = await this.prismaService.availabilityPerDay.findMany({
      where: { healthSpecialistId, dayOfWeek: currentDayOfWeek },
      select: {
        nonWorkingDay: true,

        dayOfWeek: true,
        healthSpecialistId: true,
        timeSlots: true,
      },
    });

    // Sort the time slots for each availability by start time
    availabilities.forEach((availability) => {
      availability.timeSlots.sort((a, b) => {
        if (a.startTime < b.startTime) return -1;
        if (a.startTime > b.startTime) return 1;
        return 0;
      });
    });

    return availabilities;
  }

  private async updateHealthSpecialistAvailability(user: UserPlayloadResponse, availabilityDto: AvailabilityPerDayDto): Promise<HealthSpecialist> {
    const healthSpecialistId = user.healthSpecialistId;
    if (!this.validateTimeSlots(availabilityDto)) {
      throw new MultiLanguageException(LanguageErrorEnum.NON_VALID_TIME_SLOTS);
    }

    if (!healthSpecialistId) {
      throw new MultiLanguageException(LanguageErrorEnum.NOT_HEALTH_SPECIALIST);
    }
    const existingAvailability = await this.prismaService.availabilityPerDay.findFirst({
      where: {
        healthSpecialistId,
        dayOfWeek: availabilityDto.dayOfWeek,
      },
    });

    if (existingAvailability) {
      // Update the existing availability
      return await this.prismaService.healthSpecialist.update({
        where: { id: healthSpecialistId },
        data: {
          availabilities: {
            update: {
              where: { id: existingAvailability.id },
              data: {
                nonWorkingDay: availabilityDto.nonWorkingDay,
                timeSlots: {
                  // Delete all existing time slots
                  deleteMany: {},
                  // Create new time slots if it's not a non-working day and time slots are provided
                  create: !availabilityDto.timeSlots
                    ? []
                    : availabilityDto.timeSlots.map((timeSlotDto) => ({
                        startTime: timeSlotDto.startTime,
                        endTime: timeSlotDto.endTime,
                      })),
                },
              },
            },
          },
        },
      });
    } else {
      const day = await this.getDayByName(availabilityDto.dayOfWeek);
      // Create a new availability
      return await this.prismaService.healthSpecialist.update({
        where: { id: healthSpecialistId },
        data: {
          availabilities: {
            create: {
              dayId: day.id,
              dayOfWeek: availabilityDto.dayOfWeek,
              nonWorkingDay: availabilityDto.nonWorkingDay,
              timeSlots: {
                // Only create time slots if time slots are provided
                create: availabilityDto.timeSlots
                  ? availabilityDto.timeSlots.map((timeSlotDto) => ({
                      startTime: timeSlotDto.startTime,
                      endTime: timeSlotDto.endTime,
                    }))
                  : [],
              },
            },
          },
        },
      });
    }
  }

  async getDayByName(name: DaysEnum): Promise<Day | null> {
    return await this.prismaService.day.findFirst({
      where: {
        name: name,
      },
    });
  }

  private shouldUpdateOnlyNonWorkingDay(availabilityDto: AvailabilityPerDayDto): boolean {
    return availabilityDto.nonWorkingDay;
  }
  private async updateNonWorkingDay(user: UserPlayloadResponse, availabilityDto: AvailabilityPerDayDto): Promise<void> {
    const healthSpecialistId = user.healthSpecialistId;
    const existingAvailability = await this.prismaService.availabilityPerDay.findFirst({
      where: {
        healthSpecialistId,
        dayOfWeek: availabilityDto.dayOfWeek,
      },
    });

    if (existingAvailability) {
      await this.prismaService.healthSpecialist.update({
        where: { id: healthSpecialistId },
        data: {
          availabilities: {
            update: {
              where: { id: existingAvailability.id },
              data: {
                nonWorkingDay: availabilityDto.nonWorkingDay,
              },
            },
          },
        },
      });
    } else {
      // Create a new availability with only nonWorkingDay updated
      // Assuming getDayByName method is available to get the Day entity by name
      const day = await this.getDayByName(availabilityDto.dayOfWeek);
      await this.prismaService.healthSpecialist.update({
        where: { id: healthSpecialistId },
        data: {
          availabilities: {
            create: {
              dayId: day.id,
              dayOfWeek: availabilityDto.dayOfWeek,
              nonWorkingDay: availabilityDto.nonWorkingDay,
            },
          },
        },
      });
    }
  }
  validateTimeSlots(availability: AvailabilityPerDayDto | EditAvailability): boolean {
    for (const timeSlot of availability.timeSlots) {
      const startTime = this.convertTimeToMinutes(timeSlot.startTime);
      const endTime = this.convertTimeToMinutes(timeSlot.endTime);

      if (startTime >= endTime) {
        return false;
      }
    }

    // If all time slots pass the validation, return true
    return true;
  }

  private convertTimeToMinutes(time: string): number {
    const [hours, minutes] = time.split(':').map(Number);
    return hours * 60 + minutes;
  }
}
