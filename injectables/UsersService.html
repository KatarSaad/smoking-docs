import { Inject, Injectable, forwardRef } from '@nestjs/common';
import { Prisma, User } from '@prisma/client';
import { plainToInstance } from 'class-transformer';
import { AdminsService } from 'src/modules/admins/admins.service';
import { HealthSpecialistService } from 'src/modules/health-specialists/health-specialist.service';
import { PatientService } from 'src/modules/patients/patient.service';
import { FetchAdminByIdResponse } from 'src/modules/users/response/fetch-admin-by-id.response';
import { FetchHealthSpecialistByIdResponse } from 'src/modules/users/response/fetch-health-specialist-by-id.response';
import { FetchUserByIdResponse } from 'src/modules/users/response/fetch-user-by-id.response';
import { PrismaService } from './../../common/services/prisma.service';

@Injectable()
export class UsersService {
  constructor(
    private readonly prismaService: PrismaService,
    @Inject(forwardRef(() => PatientService)) private patientService: PatientService,
    private healthSpecialistService: HealthSpecialistService,
    @Inject(forwardRef(() => AdminsService)) private adminService: AdminsService,
    private readonly commonService: CommonService,
    private readonly redisService: RedisService,
    private readonly environmentService: EnvironmentService,
    @Inject(forwardRef(() => OnboardingService)) private onboardingService: OnboardingService,
  ) {}

  async findAllUsers(filterUsersDto: FilterUsersDto): Promise<PaginatedUsersResponseDto> {
    const { sizePerPage, currentPage, sortColumn, isSorted, name, userType } = filterUsersDto;
    const sortDirection = isSorted ? 'desc' : 'asc';

    const fullTextSearch = name ? this.commonService.addLeadingPlusAndTrailingWildcard(name) : undefined;

    const where: Prisma.UserWhereInput = {
      ...(name && {
        OR: [
          {
            personalInformation: {
              firstName: {
                search: fullTextSearch,
              },
              lastName: {
                search: fullTextSearch,
              },
            },
          },
          {
            qid: { startsWith: name },
          },
        ],
      }),
      ...(userType === UserType.PATIENT && { patient: { isNot: null } }),
      ...(userType === UserType.ADMIN && { admin: { isNot: null } } && { admin: { healthSpecialist: null } }),
      ...(userType === UserType.HS && { admin: { isNot: null } } && { admin: { healthSpecialist: { isNot: null } } }),
    };

    const [query, pginationData] = await this.prismaService
      .extendPagination()
      .user.paginate({
        select: {
          id: true,
          qid: true,
          createdAt: true,
          lastLogin: true,
          patient: {
            select: {
              facility: {
                select: {
                  id: true,
                  name: true,
                },
              },
            },
          },
          personalInformation: {
            select: {
              firstName: true,
              lastName: true,
            },
          },
          admin: {
            select: {
              healthSpecialist: {
                select: {
                  facility: {
                    select: {
                      id: true,
                      name: true,
                    },
                  },
                },
              },
            },
          },
        },

        where,
        orderBy: [
          {
            ...(sortColumn === sortColumnUsersEnum.QID && { qid: sortDirection }),
            ...(sortColumn === sortColumnUsersEnum.LASTLOGIN && { lastLogin: sortDirection }),
            ...(sortColumn === sortColumnUsersEnum.FULLNAME && {
              personalInformation: {
                firstName: sortDirection,
              },
            }),

            ...(sortColumn === sortColumnUsersEnum.USERID && { id: sortDirection }),
            ...(sortColumn === sortColumnUsersEnum.FACILITY && {
              patient: {
                facility: {
                  name: sortDirection,
                },
              },
            }),
            ...(sortColumn === sortColumnUsersEnum.CREATEDAT && { createdAt: sortDirection }),
          },
          {
            ...(sortColumn === sortColumnUsersEnum.FULLNAME && {
              personalInformation: {
                lastName: sortDirection,
              },
            }),
          },
        ],
      })
      .withPages({
        includePageCount: true,
        page: currentPage,
        limit: sizePerPage,
      });

    const serialized = plainToInstance(UsersListResponse, query, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    return { data: serialized, totalItems: pginationData.totalCount };
  }

  async findPatientByUserId(userId: number): Promise<FetchUserByIdResponse> {
    await this.patientService.isUserPatient(userId);
    const user = await this.prismaService.user.findUnique({
      where: {
        id: userId,
      },
      select: {
        id: true,
        qid: true,
        location: true,
        personalInformation: {
          select: {
            email: true,
            firstName: true,
            lastName: true,
            gender: true,
            dateOfBirth: true,
            phoneNumber: true,
          },
        },
        patient: {
          select: {
            id: true,
            username: true,
          },
        },
      },
    });
    return user;
  }

  async findHealthSpecialistByUserId(userId: number): Promise<FetchHealthSpecialistByIdResponse> {
    await this.healthSpecialistService.healthSpecialistIdByUserId(userId);
    return await this.prismaService.user.findUnique({
      where: {
        id: userId,
      },
      select: {
        id: true,
        qid: true,
        location: true,
        personalInformation: {
          select: {
            phoneNumber: true,
            email: true,
            firstName: true,
            lastName: true,
            gender: true,
            dateOfBirth: true,
          },
        },
        admin: {
          select: {
            healthSpecialist: {
              select: {
                speciality: true,
                licenseNumber: true,
                joiningDate: true,
                facility: {
                  select: { id: true, name: true },
                },
              },
            },
          },
        },
      },
    });
  }

  async findAdminByUserId(userId: number): Promise<FetchAdminByIdResponse> {
    await this.adminService.isAdmin(userId);

    const findAdminByUserId = await this.prismaService.user.findUnique({
      where: {
        id: userId,
      },
      select: {
        id: true,
        qid: true,
        location: true,
        personalInformation: {
          select: {
            firstName: true,
            lastName: true,
            gender: true,
            dateOfBirth: true,
            phoneNumber: true,
            email: true,
          },
        },
        admin: {
          select: {
            department: true,
            adminFacilityGroups: {
              select: {
                id: true,
                name: true,
              },
            },
            role: {
              select: {
                id: true,
                name: true,
                restriction: true,
                isActive: true,
                facilityGroups: {
                  select: {
                    id: true,
                    name: true,
                  },
                },
              },
            },
          },
        },
      },
    });

    const findAdminByUserIdSerialized = plainToInstance(FetchAdminByIdResponse, findAdminByUserId, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });

    return findAdminByUserIdSerialized;
  }

  async getUserIdByAdminId(adminId: number): Promise<bigint> {
    const admin = await this.prismaService.admin.findFirst({
      where: {
        id: adminId,
      },
      select: { userId: true },
    });

    if (!admin) throw new MultiLanguageException(LanguageErrorEnum.USER_NOT_FOUND);

    return admin.userId;
  }

  async fetchSuperadminId(): Promise<number> {
    const superadmin = await this.prismaService.user.findFirst({
      where: {
        admin: {
          role: {
            name: RoleEnum.SUPERADMIN,
          },
        },
      },
      select: {
        id: true,
      },
    });
    return Number(superadmin.id);
  }

  async getUserPersonalInformation(userId: number): Promise<PersonalInformationResponse | null> {
    const getUserPersonalInformationRedisKey = `getUserPersonalInformation/${userId}`;
    const cacheResults = await this.redisService.getResponseKey(getUserPersonalInformationRedisKey);
    if (cacheResults) return cacheResults;

    const personalInformation = await this.prismaService.personalInformation.findFirst({
      where: {
        user: {
          id: userId,
        },
      },
      select: {
        firstName: true,
        lastName: true,
        dateOfBirth: true,
        gender: true,
        height: true,
        weight: true,
      },
    });

    this.redisService.setResponseKey(getUserPersonalInformationRedisKey, personalInformation, this.environmentService.redisConfig.longTimeToLive);

    return personalInformation;
  }

  async getUserPersonalInformationByEmail(email: string): Promise<UserForPersonalInfoFull> {
    const getUserPersonalInformationRedisKey = `getUserPersonalInformationByEmail/${email}`;
    const cacheResults = await this.redisService.getResponseKey(getUserPersonalInformationRedisKey);
    if (cacheResults) return cacheResults;

    await this.onboardingService.isEmailExist(email);

    const personalInformation = await this.prismaService.user.findFirst({
      where: {
        personalInformation: {
          email,
        },
      },
      select: {
        id: true,
        personalInformation: {
          select: {
            firstName: true,
            lastName: true,
            dateOfBirth: true,
            gender: true,
            height: true,
            weight: true,
          },
        },
      },
    });

    this.redisService.setResponseKey(getUserPersonalInformationRedisKey, personalInformation, this.environmentService.redisConfig.longTimeToLive);

    return personalInformation;
  }

  async findById(id: bigint): Promise<User> {
    const findUserByIdRedisKey = `findUserById/${id}`;
    const cacheResults = await this.redisService.getResponseKey(findUserByIdRedisKey);

    if (cacheResults) return cacheResults;

    const user = await this.prismaService.user.findUnique({
      where: { id },
      include: {
        patient: true,
        admin: {
          include: {
            role: true,
            healthSpecialist: true,
          },
        },
      },
    });

    if (!user) throw new MultiLanguageException(LanguageErrorEnum.USER_NOT_FOUND);

    this.redisService.setResponseKey(findUserByIdRedisKey, user, this.environmentService.redisConfig.longerTimeToLive);

    return user;
  }

  async isQidNotExist(qid: string, userId: number): Promise<boolean> {
    const userQid = await this.prismaService.user.findFirst({
      where: {
        qid: qid,
        ...(userId && {
          id: {
            not: userId,
          },
        }),
      },
      select: { id: true },
    });

    if (userQid) throw new MultiLanguageException(LanguageErrorEnum.QID_EXISTS);
    return true;
  }

  async isUsersByIdExist(userIds: number[]): Promise<boolean> {
    const users = await this.prismaService.user.aggregate({
      where: { id: { in: userIds } },
      _count: { _all: true },
    });

    if (users._count._all < userIds.length) throw new MultiLanguageException(LanguageErrorEnum.USER_NOT_FOUND);
    return true;
  }

  async isUserByIdExists(userId: number, transaction: PrismaTransactionClient): Promise<boolean> {
    const prismaOperation = transaction ?? this.prismaService;

    const user = await prismaOperation.user.findUnique({
      where: { id: userId },
      select: { id: true },
    });

    if (!user) throw new MultiLanguageException(LanguageErrorEnum.USER_NOT_FOUND);
    return true;
  }

  async isCodeByUserExists(id: number, code: string): Promise<boolean> {
    const isEmailAndCodeExists = await this.prismaService.user.findFirst({
      where: {
        id,
        signInCode: {
          code,
        },
      },
      select: {
        signInCode: {
          select: {
            updatedAt: true,
            isUsed: true,
          },
        },
      },
    });

    if (!isEmailAndCodeExists) throw new MultiLanguageException(LanguageErrorEnum.SIGNIN_CODE_NOT_FOUND);

    if (isEmailAndCodeExists.signInCode.isUsed) throw new MultiLanguageException(LanguageErrorEnum.SIGNIN_CODE_EXPIRED);

    const datesDifferenceInDays = this.commonService.datesDifferenceInDays(isEmailAndCodeExists.signInCode.updatedAt, new Date());
    if (datesDifferenceInDays > 1) throw new MultiLanguageException(LanguageErrorEnum.SIGNIN_CODE_EXPIRED);
    return true;
  }
}
