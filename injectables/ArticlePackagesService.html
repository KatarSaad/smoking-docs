import { Injectable } from '@nestjs/common';
import { ArticleStatusEnum, LanguageEnum } from '@prisma/client';
import { plainToInstance } from 'class-transformer';
import { IdDto } from 'src/common/dto/id.dto';
import { LanguageErrorEnum } from 'src/common/enums/language-error.enum';
import { MultiLanguageException } from 'src/common/exception-filters/multi-language.exception';
import { EnvironmentService } from 'src/common/services/environment.service';
import { PrismaService } from 'src/common/services/prisma.service';
import { RedisService } from 'src/common/services/redis.service';
import { AddArticlePackageDto } from 'src/modules/article-packages/dto/add-article-package.dto';
import { AddRecommendedArticleDto } from 'src/modules/article-packages/dto/add-recommended-article.Dto';
import { UpdateArticlePackage } from 'src/modules/article-packages/dto/update-article-package.dto';
import { UpdateRecommendedArticleDto } from 'src/modules/article-packages/dto/update-recommended-article.Dto';
import { ArticlePackageWithCountResponse } from 'src/modules/article-packages/response/article-package.response';
import { FetchArticleById } from 'src/modules/article-packages/response/fetch-article-by-id.response';
import { ArticlesService } from 'src/modules/articles/articles.service';
import { GroupsService } from 'src/modules/groups/groups.service';

@Injectable()
export class ArticlePackagesService {
  constructor(
    private prismaService: PrismaService,
    private articlesService: ArticlesService,
    private groupsService: GroupsService,
    private redisService: RedisService,
    private environmentService: EnvironmentService,
  ) {}

  async articlePackages(): Promise<ArticlePackageWithCountResponse[]> {
    return await this.prismaService.articlePackage.findMany({
      select: {
        id: true,
        isEditable: true,
        targetGroup: {
          select: {
            id: true,
            name: true,
          },
        },
        _count: {
          select: {
            recommendedArticles: true,
          },
        },
      },
      orderBy: [
        {
          isEditable: 'desc',
        },
        {
          recommendedArticles: {
            _count: 'desc',
          },
        },
      ],
    });
  }

  async addArticlePackage(addArticlePackageDto: AddArticlePackageDto): Promise<ArticlePackageWithCountResponse> {
    const { recommendedArticles, targetGroupId } = addArticlePackageDto;
    let articlesToConnect = null;

    const groupArticlePackage = await this.groupsService.groupArticlePackage(targetGroupId);
    if (groupArticlePackage) throw new MultiLanguageException(LanguageErrorEnum.GROUP_HAS_ARTICLE_PACKAGE);

    if (recommendedArticles?.length > 0) {
      await this.articlesService.isArticlesByIdExists(recommendedArticles, ArticleStatusEnum.APPROVED);

      articlesToConnect = recommendedArticles.map((recommendedArticle) => {
        return {
          articleId: recommendedArticle.id,
          displayDaysAfterQuitting: recommendedArticle?.displayDaysAfterQuitting,
        };
      });
    }

    return await this.prismaService.articlePackage.create({
      data: {
        targetGroup: {
          connect: {
            id: targetGroupId,
          },
        },
        ...(recommendedArticles && {
          recommendedArticles: {
            create: articlesToConnect,
          },
        }),
      },
      select: {
        id: true,
        isEditable: true,
        targetGroup: {
          select: {
            id: true,
            name: true,
          },
        },
        _count: {
          select: {
            recommendedArticles: true,
          },
        },
      },
    });
  }

  async articlePackageById(language: LanguageEnum, id: number): Promise<FetchArticleById> {
    const articlePackageById = await this.prismaService.articlePackage.findUnique({
      where: { id },
      select: {
        id: true,
        isEditable: true,
        targetGroup: {
          select: {
            id: true,
            name: true,
          },
        },
        recommendedArticles: {
          select: {
            article: {
              select: {
                id: true,
                image: true,
                articlesTranslation: {
                  where: {
                    language: {
                      code: language,
                    },
                  },
                  select: {
                    title: true,
                    body: true,
                  },
                },
              },
            },
            displayDaysAfterQuitting: true,
          },
        },
      },
    });
    if (!articlePackageById) throw new MultiLanguageException(LanguageErrorEnum.ARTICLE_PACKAGE_NOT_FOUND);
    const serializedArticlePackageById = plainToInstance(FetchArticleById, articlePackageById, {
      excludeExtraneousValues: true,
      enableImplicitConversion: true,
    });
    return serializedArticlePackageById;
  }

  async updateArticlePackage(updateArticlePackage: UpdateArticlePackage, id: number): Promise<ArticlePackageWithCountResponse> {
    const { recommendedArticles, targetGroupId } = updateArticlePackage;
    await this.isArticlePackageByIdExists(id);

    if (targetGroupId) {
      const groupArticlePackage = await this.groupsService.groupArticlePackage(targetGroupId);
      if (groupArticlePackage) throw new MultiLanguageException(LanguageErrorEnum.GROUP_HAS_ARTICLE_PACKAGE);
    }

    const articlesToConnect = recommendedArticles.map((recommendedArticle) => {
      return {
        articleId: recommendedArticle.id,
        displayDaysAfterQuitting: recommendedArticle?.displayDaysAfterQuitting,
      };
    });

    const articlePackageById = await this.prismaService.articlePackage.update({
      where: { id },
      data: {
        ...(targetGroupId && {
          targetGroup: {
            connect: {
              id: targetGroupId,
            },
          },
        }),
        recommendedArticles: {
          deleteMany: {},
          create: articlesToConnect,
        },
      },
      select: {
        id: true,
        isEditable: true,
        targetGroup: {
          select: {
            id: true,
            name: true,
          },
        },
        _count: {
          select: {
            recommendedArticles: true,
          },
        },
      },
    });

    return articlePackageById;
  }

  async deletePackage(id: number): Promise<ArticlePackageWithCountResponse> {
    await this.isArticlePackageByIdExists(id);
    await this.isArticlePackageEditable(id);

    return await this.prismaService.articlePackage.delete({
      where: { id },
      select: {
        id: true,
        isEditable: true,
        targetGroup: {
          select: {
            id: true,
            name: true,
          },
        },
        _count: {
          select: {
            recommendedArticles: true,
          },
        },
      },
    });
  }

  async addArticleToArticlePackage(addRecommendedArticleDto: AddRecommendedArticleDto, packageId: number): Promise<ArticlePackageWithCountResponse> {
    const { displayDaysAfterQuitting, id } = addRecommendedArticleDto;

    await this.isArticlePackageByIdExists(packageId);

    await this.articlesService.isArticleByIdExists([ArticleStatusEnum.APPROVED], Number(id));

    await this.isArticleNotExistInPackag(packageId, Number(id));

    return await this.prismaService.articlePackage.update({
      where: { id: packageId },
      data: {
        recommendedArticles: {
          create: {
            displayDaysAfterQuitting,
            articleId: id,
          },
        },
      },
      select: {
        id: true,
        isEditable: true,
        targetGroup: {
          select: {
            id: true,
            name: true,
          },
        },
        _count: {
          select: {
            recommendedArticles: true,
          },
        },
      },
    });
  }

  async updateArticleInArticlePackage(
    updateRecommendedArticleDto: UpdateRecommendedArticleDto,
    packageId: number,
    articleId: number,
  ): Promise<ArticlePackageWithCountResponse> {
    const { displayDaysAfterQuitting, id } = updateRecommendedArticleDto;

    await this.isArticlePackageByIdExists(packageId);

    await this.articlesService.isArticleByIdExists(null, Number(id));

    await this.isArticleExistsInPackag(packageId, Number(articleId));

    return await this.prismaService.articlePackage.update({
      where: { id: packageId },
      data: {
        recommendedArticles: {
          delete: {
            articlePackageId_articleId: {
              articleId,
              articlePackageId: packageId,
            },
          },
          create: {
            displayDaysAfterQuitting,
            articleId: id,
          },
        },
      },
      select: {
        id: true,
        isEditable: true,
        targetGroup: {
          select: {
            id: true,
            name: true,
          },
        },
        _count: {
          select: {
            recommendedArticles: true,
          },
        },
      },
    });
  }

  async deleteArticleFromArticlePackage(id: number, articleId: number): Promise<ArticlePackageWithCountResponse> {
    await this.isArticlePackageByIdExists(id);

    await this.articlesService.isArticleByIdExists(null, articleId);

    await this.isArticleExistsInPackag(id, articleId);

    return await this.prismaService.articlePackage.update({
      where: { id },
      data: {
        recommendedArticles: {
          delete: {
            articlePackageId_articleId: {
              articleId,
              articlePackageId: id,
            },
          },
        },
      },
      select: {
        id: true,
        isEditable: true,
        targetGroup: {
          select: {
            id: true,
            name: true,
          },
        },
        _count: {
          select: {
            recommendedArticles: true,
          },
        },
      },
    });
  }

  async defaultPackageArticlesIds(): Promise<IdDto[]> {
    const defaultPackageArticleIdsRedisKey = `defaultPackageArticleIds`;
    const cacheResults = await this.redisService.getResponseKey(defaultPackageArticleIdsRedisKey);
    if (cacheResults) return cacheResults;

    const defaultArticlePackages = await this.prismaService.articlePackage.findMany({
      where: {
        isEditable: false,
      },
      select: { id: true },
    });
    this.redisService.setResponseKey(defaultPackageArticleIdsRedisKey, defaultArticlePackages, this.environmentService.redisConfig.TTL);

    return defaultArticlePackages;
  }

  async isArticlePackageByIdExists(id: number): Promise<boolean> {
    const articlePackageById = await this.prismaService.articlePackage.findUnique({
      where: { id },
      select: { id: true },
    });
    if (!articlePackageById) throw new MultiLanguageException(LanguageErrorEnum.ARTICLE_PACKAGE_NOT_FOUND);
    return true;
  }

  async isArticleExistsInPackag(articlePackageId: number, articleId: number): Promise<boolean> {
    const packageOnArticle = await this.prismaService.articlePackagesOnArticles.findFirst({
      where: {
        articleId,
        articlePackageId,
      },
    });
    if (!packageOnArticle) throw new MultiLanguageException(LanguageErrorEnum.ARTICLE_NOT_FOUND_IN_PACKAGE);
    return true;
  }

  async isArticleNotExistInPackag(articlePackageId: number, articleId: number): Promise<boolean> {
    const packageOnArticle = await this.prismaService.articlePackagesOnArticles.findFirst({
      where: {
        articleId,
        articlePackageId,
      },
    });
    if (packageOnArticle) throw new MultiLanguageException(LanguageErrorEnum.ARTICLE_EXISTS_IN_PACKAGE);
    return true;
  }

  async isArticlePackageEditable(id: number): Promise<boolean> {
    const articlePackage = await this.prismaService.articlePackage.findUnique({
      where: { id },
      select: {
        id: true,
        isEditable: true,
      },
    });
    if (!articlePackage) throw new MultiLanguageException(LanguageErrorEnum.ARTICLE_PACKAGE_NOT_FOUND);
    if (!articlePackage.isEditable) throw new MultiLanguageException(LanguageErrorEnum.ITEM_NOT_EDIABLE);
    return true;
  }
}
