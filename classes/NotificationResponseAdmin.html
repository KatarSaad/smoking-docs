import { ApiProperty, ApiPropertyOptional, PickType } from '@nestjs/swagger';
import { NotificationStatusEnum } from '@prisma/client';
import { Expose } from 'class-transformer';
import { IsOptional } from 'class-validator';
import { CommonFieldsResponse } from 'src/common/responses/common-fields.response';
import { NameAndIdResponse } from 'src/common/responses/id-and-name.response';

export class NotificationResponse extends PickType(CommonFieldsResponse, ['id'] as const) {
  @Expose()
  @ApiProperty()
  senderId: bigint;
  @Expose()
  @ApiProperty()
  title: string;
  @Expose()
  @ApiProperty()
  body: string;
  @Expose()
  @ApiProperty({ type: NameAndIdResponse, isArray: true })
  targetGroupFilters: NameAndIdResponse[] | [];
  @Expose()
  @ApiProperty()
  imageLink: string;
  @Expose()
  @ApiProperty()
  isCritical: boolean;
  @Expose()
  @ApiProperty()
  deliveryDate: Date;
  @Expose()
  @ApiProperty()
  notificationType: string;
  @Expose()
  @ApiProperty()
  status: NotificationStatusEnum;
  @Expose()
  @IsOptional()
  @ApiPropertyOptional()
  recipientId: bigint;
  @IsOptional()
  @ApiPropertyOptional()
  subjectId?: bigint;
  @IsOptional()
  @ApiPropertyOptional()
  seen?: boolean;
  @IsOptional()
  @ApiPropertyOptional()
  senderFullName?: string;
  @IsOptional()
  @ApiPropertyOptional()
  notificationPackageId?: bigint;
  @IsOptional()
  @ApiPropertyOptional()
  daysAfterQuitting?: number;
}
