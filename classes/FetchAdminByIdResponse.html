import { ApiProperty, PickType } from '@nestjs/swagger';
import { RoleRestrictionEnum } from '@prisma/client';
import { Expose, Transform, Type } from 'class-transformer';
import { CommonFieldsResponse } from 'src/common/responses/common-fields.response';
import { NameAndIdResponse } from 'src/common/responses/id-and-name.response';
import { FacilityRoleResponse } from 'src/modules/facility-role/response/created-facility-role.response';
import { CommonPersonalInformationResponse } from 'src/modules/onboarding/response/common-personal-information.response';
import { AdminType } from 'src/modules/users/enum/admin-type.enum';

class AdminResponse {
  @Expose()
  @ApiProperty()
  department: string;

  @Expose()
  @Type(() => FacilityRoleResponse)
  @ApiProperty({ type: FacilityRoleResponse })
  role: FacilityRoleResponse;

  // @Expose()
  @Type(() => NameAndIdResponse)
  // @ApiProperty({ type: [NameAndIdResponse] })
  adminFacilityGroups: NameAndIdResponse[];

  @Expose()
  @Transform(({ obj }) => getFirstFacilityGroup(obj))
  @Type(() => NameAndIdResponse)
  @ApiProperty({ type: NameAndIdResponse })
  adminFacilityGroup: NameAndIdResponse;
}

export class FetchAdminByIdResponse extends PickType(CommonFieldsResponse, ['id'] as const) {
  @Expose()
  @ApiProperty()
  qid: string;

  @Expose()
  @ApiProperty()
  location: string;

  @Expose()
  @Type(() => CommonPersonalInformationResponse)
  @ApiProperty({ type: CommonPersonalInformationResponse })
  personalInformation: CommonPersonalInformationResponse;

  @Expose()
  @Type(() => AdminResponse)
  @ApiProperty({ type: AdminResponse })
  admin: AdminResponse;

  @Expose()
  @Transform(({ obj }) => returnAdminType(obj))
  @ApiProperty({ enum: AdminType })
  adminType: AdminType;
}

const getFirstFacilityGroup = (obj: AdminResponse): NameAndIdResponse => {
  return obj.adminFacilityGroups[0];
};

const returnAdminType = (obj: FetchAdminByIdResponse): AdminType => {
  if (!obj.admin.role) return null;
  if (obj.admin.role.restriction === RoleRestrictionEnum.SYSTEM_ADMIN) return AdminType.SYSADMIN;
  if (obj.admin.role.restriction === RoleRestrictionEnum.CUSTOM) return AdminType.FACILITY_ADMIN;
  else return null;
};
