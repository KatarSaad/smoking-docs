import { ApiProperty, PickType } from '@nestjs/swagger';
import { Expose, Transform, Type } from 'class-transformer';
import { DistractionOptionResponse } from 'src/modules/cravings/response/distraction-option.response';
import { StressFactorOptionResponse } from 'src/modules/cravings/response/stress-factor-option.response';
import { DiaryEntryEnum } from 'src/modules/diaries/enums/diary-entry.enum';
import { PatchResponse } from 'src/modules/patches/response/patch.response';

export class CravingResponse extends PickType(PatchResponse, ['id', 'createdAt', 'cravingLevel'] as const) {
  @Expose()
  @ApiProperty()
  cravingDate: Date;

  @Type(() => DistractionOptionResponse)
  distractionOptions: [DistractionOptionResponse];

  @Type(() => StressFactorOptionResponse)
  stressFactorOptions: StressFactorOptionResponse[];

  @Expose()
  @Transform(({ obj }: { obj: CravingResponse }) => getDistractions(obj))
  @ApiProperty()
  distractions: string[];

  @Expose()
  @Transform(({ obj }: { obj: CravingResponse }) => getStressFactors(obj))
  @ApiProperty()
  stressFactors: string[];

  @Expose()
  @ApiProperty()
  latitude: number;

  @Expose()
  @ApiProperty()
  longitude: number;

  @Expose()
  @ApiProperty()
  location: string;

  @Expose()
  @Transform(() => DiaryEntryEnum.CRAVING)
  @ApiProperty({ enum: DiaryEntryEnum })
  diaryEntryEnum: DiaryEntryEnum;

  @Expose()
  @Transform(({ obj }) => getSortField(obj))
  sortField: Date;
}

function getSortField(craving: CravingResponse): Date {
  return craving.cravingDate;
}

function getDistractions(distraction: CravingResponse): string[] {
  const isDistractions = distraction?.distractionOptions.length > 0;

  if (!isDistractions) return [];

  return distraction.distractionOptions.map((distractionOption) => distractionOption.distraction);
}

function getStressFactors(stressFactor: CravingResponse): string[] {
  const isStressFactor = stressFactor?.stressFactorOptions.length > 0;

  if (!isStressFactor) return [];

  return stressFactor.stressFactorOptions.map((stressFactor) => stressFactor.stressFactor);
}
