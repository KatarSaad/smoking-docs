import {
  WebSocketGateway,
  OnGatewayInit,
  OnGatewayConnection,
  OnGatewayDisconnect,
  WebSocketServer,
  ConnectedSocket,
  SubscribeMessage,
  MessageBody,
} from '@nestjs/websockets';
import { Logger, UseFilters, UsePipes } from '@nestjs/common';
import { Server, Socket } from 'socket.io';
import { SocketService } from 'src/modules/socket/socket.service';
import { JwtAuth } from 'src/common/decorators/jwt-auth.decorator';
import { RoleDecoratorEnum } from 'src/common/enums/role-decorator.enum';
import { SocketEventEnum } from 'src/common/enums/socket-event.enum';
import { UrgentSupportDto } from 'src/modules/private-chat-messages/dto/urgent-support.dto';
import { UserPlayloadResponse } from 'src/modules/patients/response/user-playload.response';
import { User } from 'src/common/decorators/auth.decorator';
import { SuccessResponse } from 'src/common/responses/socket.response';
import { SocketExceptionFilter } from 'src/common/exception-filters/socket-exception.filter';
import { WSValidationPipe } from 'src/common/pipes/socket-validation.pipe';
import { CravingsListener } from 'src/modules/cravings/cravings.listener';
import { PrivateChatsListener } from 'src/modules/private-chats/private-chat.listener';
import { PrivateChatInteractionSocketResponse } from 'src/modules/private-chat-messages/response/private-chat-interaction-socket.response';
import { PrivateChatInteractionIdDto } from 'src/modules/private-chat-messages/dto/private-chat-interaction.dto';
import { MessagesListener } from 'src/modules/messages/messages.listener';
import { DeleteMessageDto } from 'src/modules/private-chat-messages/dto/delete-message.dto';
import { MessageGroupChatSocketResponse } from 'src/modules/messages/response/message-group-chat.response';
import { HealthSpecialistJoinRoomDto } from 'src/modules/messages/dto/health-specialist-join-room.dto';
import { PatientJoinRoomDto } from 'src/modules/messages/dto/patient-join-room.dto';
import { SocketMessagePrivateDto } from 'src/modules/private-chat-messages/dto/socket-message-private.dto';
import { PrivateChatMessageSocketResponse } from 'src/modules/private-chat-messages/response/private-chat-message-socket.response';
import { ParticipantJoinPrivateChatRoomDto } from 'src/modules/private-chat-messages/dto/participant-join-private-chat-room.dto';
import { PrivateChatMessagesListener } from 'src/modules/private-chat-messages/private-chat-messages.listener';
import { JoinHealthSpecialistPrivateChatRoomDto } from 'src/modules/private-chat-messages/dto/health-specialist-join-private-chat-room.dto';
import { CommunityChatSocketResponse } from 'src/modules/community-messages/response/community-chat-socket.response';
import { CommunityMessageDto } from 'src/modules/community-messages/dto/community-messages.dto';
import { CommunityMessagesListener } from 'src/modules/community-messages/community-messages.listener';
import { JoinHealthSpecialistGroupChatRoomsDto } from 'src/modules/messages/response/health-specialist-join-group-chat-room.dto';
import { SocketMessageGroupChatDto } from 'src/modules/messages/dto/socket-message-group-chat.dto';

@WebSocketGateway({
  cors: {
    origin: '*',
  },
})
@UseFilters(new SocketExceptionFilter())
@UsePipes(new WSValidationPipe())
export class SocketListener implements OnGatewayInit, OnGatewayConnection, OnGatewayDisconnect {
  constructor(
    private privateChatsListener: PrivateChatsListener,
    private cravingsListener: CravingsListener,
    private messagesListener: MessagesListener,
    private privateChatMessagesListener: PrivateChatMessagesListener,
    private communityMessagesListener: CommunityMessagesListener,
    private socketService: SocketService,
  ) {}

  @WebSocketServer() public server: Server;

  private logger: Logger = new Logger('SocketListener');

  afterInit(server: Server) {
    this.socketService.socket = server;
  }

  //group chats

  @JwtAuth()
  @SubscribeMessage(SocketEventEnum.MESSAGE_GROUP_CHAT)
  async sendGroupChatMessage(
    @MessageBody()
    socketMessageGroupChatDto: SocketMessageGroupChatDto,
    @ConnectedSocket() client: Socket,
    @User() user: UserPlayloadResponse,
  ): Promise<MessageGroupChatSocketResponse> {
    return await this.messagesListener.sendGroupChatMessage(socketMessageGroupChatDto, client, user, this.server);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @SubscribeMessage(SocketEventEnum.PARTICIPANT_JOIN_GROUP)
  async participantJoinGroupChatRoom(
    @MessageBody()
    patientJoinRoomDto: PatientJoinRoomDto,
    @User() user: UserPlayloadResponse,
  ) {
    return await this.messagesListener.participantJoinGroupChatRoom(patientJoinRoomDto, user, this.server);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @SubscribeMessage(SocketEventEnum.ADMIN_JOIN_GROUP)
  async adminJoinGroupChatRoom(
    @MessageBody()
    healthSpecialistJoinRoomDto: HealthSpecialistJoinRoomDto,
    @User() user: UserPlayloadResponse,
  ) {
    return await this.messagesListener.adminJoinGroupChatRoom(healthSpecialistJoinRoomDto, user, this.server);
  }

  @JwtAuth([RoleDecoratorEnum.SYSADMIN])
  @SubscribeMessage(SocketEventEnum.ADMIN_JOIN_GROUP_CHATS)
  async adminJoinGroupChatRooms(
    @MessageBody()
    healthSpecialistJoinRoomDto: JoinHealthSpecialistGroupChatRoomsDto,
    @User() user: UserPlayloadResponse,
  ) {
    return await this.messagesListener.adminJoinGroupChatRooms(healthSpecialistJoinRoomDto, user, this.server);
  }

  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST, RoleDecoratorEnum.USER])
  @SubscribeMessage(SocketEventEnum.DELETE_MESSAGE_GROUP_CHAT)
  async deleteGroupChatMessage(
    @MessageBody()
    deleteMessageDto: DeleteMessageDto,
    @ConnectedSocket() client: Socket,
    @User() user: UserPlayloadResponse,
  ): Promise<MessageGroupChatSocketResponse> {
    return await this.messagesListener.deleteGroupChatMessage(deleteMessageDto, client, user);
  }

  //Community  chat

  @JwtAuth([RoleDecoratorEnum.USER])
  @SubscribeMessage(SocketEventEnum.MESSAGE_COMMUNITY_CHAT)
  async handleChatEvent(
    @MessageBody()
    communityMessageDto: CommunityMessageDto,
    @ConnectedSocket() client: Socket,
    @User() user: UserPlayloadResponse,
  ): Promise<CommunityChatSocketResponse> {
    return await this.communityMessagesListener.sendCommunityChatMessage(communityMessageDto, client, user);
  }

  @JwtAuth()
  @SubscribeMessage(SocketEventEnum.DELETE_MESSAGE_COMMUNITY_CHAT)
  async handleChatDeleteEvent(
    @MessageBody()
    deleteMessageDto: DeleteMessageDto,
    @ConnectedSocket() client: Socket,
    @User() user: UserPlayloadResponse,
  ): Promise<CommunityChatSocketResponse> {
    return await this.communityMessagesListener.handleChatDeleteEvent(deleteMessageDto, client, user);
  }
  //private chat

  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST, RoleDecoratorEnum.USER])
  @SubscribeMessage(SocketEventEnum.MESSAGE_PRIVATE_CHAT)
  async sendPrivateChatMessage(
    @MessageBody()
    socketMessagePrivateDto: SocketMessagePrivateDto,
    @ConnectedSocket() client: Socket,
    @User() user: UserPlayloadResponse,
  ): Promise<PrivateChatMessageSocketResponse> {
    return this.privateChatMessagesListener.sendPrivateChatMessage(socketMessagePrivateDto, client, user, this.server);
  }

  @JwtAuth([RoleDecoratorEnum.USER])
  @SubscribeMessage(SocketEventEnum.PATIENT_JOIN_PRIVATE_CHAT)
  async userJoinPrivateChatRoom(
    @MessageBody()
    participantJoinPrivateChatRoomDto: ParticipantJoinPrivateChatRoomDto,
    @User() user: UserPlayloadResponse,
  ): Promise<SuccessResponse> {
    return this.privateChatMessagesListener.userJoinPrivateChatRoom(participantJoinPrivateChatRoomDto, user, this.server);
  }

  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST])
  @SubscribeMessage(SocketEventEnum.HS_JOIN_PRIVATE_CHAT)
  async healthSpecialistJoinPrivateChatRoom(
    @MessageBody()
    participantJoinPrivateChatRoomDto: ParticipantJoinPrivateChatRoomDto,
    @User() user: UserPlayloadResponse,
  ): Promise<SuccessResponse> {
    return this.privateChatMessagesListener.healthSpecialistJoinPrivateChatRoom(participantJoinPrivateChatRoomDto, user, this.server);
  }

  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST])
  @SubscribeMessage(SocketEventEnum.HS_JOIN_PRIVATE_CHATS)
  async healthSpecialistJoinPrivateChatRooms(
    @MessageBody()
    joinHealthSpecialistPrivateChatRoomDto: JoinHealthSpecialistPrivateChatRoomDto,
    @User() user: UserPlayloadResponse,
  ): Promise<SuccessResponse> {
    return this.privateChatMessagesListener.healthSpecialistJoinPrivateChatRooms(joinHealthSpecialistPrivateChatRoomDto, user, this.server);
  }

  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST, RoleDecoratorEnum.USER])
  @SubscribeMessage(SocketEventEnum.DELETE_MESSAGE_PRIVATE_CHAT)
  async deletePrivateChatMessage(
    @MessageBody()
    deleteMessageDto: DeleteMessageDto,
    @ConnectedSocket() client: Socket,
    @User() user: UserPlayloadResponse,
  ): Promise<PrivateChatMessageSocketResponse> {
    return this.privateChatMessagesListener.deletePrivateChatMessage(deleteMessageDto, client, user);
  }

  //interaction for private chat
  //this event to enable/disable chat
  @JwtAuth([RoleDecoratorEnum.HEALTH_SPECIALIST])
  @SubscribeMessage(SocketEventEnum.PRIVATE_CHAT_INTERACTION)
  async privateChatStatusChange(
    @MessageBody()
    privateChatInteractionIdDto: PrivateChatInteractionIdDto,
    @ConnectedSocket() client: Socket,
    @User() user: UserPlayloadResponse,
  ): Promise<PrivateChatInteractionSocketResponse> {
    return await this.privateChatsListener.privateChatStatusChange(privateChatInteractionIdDto, client, user);
  }

  // Craving
  @JwtAuth([RoleDecoratorEnum.USER])
  @SubscribeMessage(SocketEventEnum.CRAVING)
  async cravingEvent(
    @MessageBody()
    urgentSupportDto: UrgentSupportDto,
    @ConnectedSocket() client: Socket,
    @User() user: UserPlayloadResponse,
  ): Promise<SuccessResponse> {
    return await this.cravingsListener.cravingEvent(urgentSupportDto, client, user);
  }

  handleDisconnect(client: Socket) {
    this.logger.log(`Client disconnected: ${client.id}`);
  }

  handleConnection(client: Socket) {
    this.logger.log(`Client connected: ${client.id}`);
  }
}
