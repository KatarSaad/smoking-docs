import { ApiProperty, PickType } from '@nestjs/swagger';
import { StatusEnum, TicketCategoryEnum } from '@prisma/client';
import { CommentResponse } from 'src/modules/comments/response/comment.response';
import { PriorityEnum } from '../dto/update-ticket-by-admin.dto';
import { CommonFieldsResponse } from 'src/common/responses/common-fields.response';

export class TicketResponse extends PickType(CommonFieldsResponse, ['id', 'createdAt', 'updatedAt']) {
  @ApiProperty({
    description: 'The subject of the ticket.',
    type: 'string',
  })
  subject: string;

  @ApiProperty({ enum: TicketCategoryEnum })
  category: TicketCategoryEnum;

  @ApiProperty({
    description: 'The description of the ticket.',
    type: 'string',
    nullable: true,
  })
  description: string | null;

  @ApiProperty({
    description: 'The status of the ticket.',
    enum: StatusEnum,
  })
  status: StatusEnum;

  @ApiProperty()
  assignedToFirstName: string | null;

  @ApiProperty()
  assignedToLastName: string | null;

  @ApiProperty({
    description: 'The ID of the admin assigned to the ticket.',
    type: 'integer',
    format: 'int64',
    nullable: true,
  })
  assignedToId: bigint | null;

  @ApiProperty({
    description: 'The priority of the ticket.',
    enum: PriorityEnum,
    nullable: true,
  })
  priority: PriorityEnum | null;

  @ApiProperty()
  reporterFirstName: string;

  @ApiProperty()
  reporterLastName: string;

  @ApiProperty({
    description: 'The ID of the reporter.',
    type: 'integer',
    format: 'int64',
  })
  reporterId: bigint;

  @ApiProperty({
    description: 'The comments associated with the ticket.',
    type: () => [CommentResponse],
  })
  comments: CommentResponse[];

  @ApiProperty({
    description: 'The attachments associated with the ticket.',
    type: () => [AttachmentResponse],
  })
  attachments: AttachmentResponse[];
}

class AttachmentResponse {
  @ApiProperty({
    description: 'The unique identifier of the attachment.',
    type: 'integer',
    format: 'int64',
  })
  id: bigint;

  @ApiProperty({
    description: 'The name of the image file.',
    type: 'string',
  })
  imageName: string;

  @ApiProperty({
    description: 'The ID of the ticket the attachment is associated with.',
    type: 'integer',
    format: 'int64',
  })
  ticketId: bigint;
}
