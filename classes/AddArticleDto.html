import { ApiProperty, ApiPropertyOptional } from '@nestjs/swagger';
import { GenderTypeEnum } from '@prisma/client';
import { Transform } from 'class-transformer';
import { IsEnum, IsInt, IsOptional, IsString, MaxLength, Min, MinLength } from 'class-validator';
import * as Joi from 'joi';
import { JsonParseAndValidate } from 'src/common/custom-validators/parse-validate.decorator';
import { IdDto } from 'src/common/dto/id.dto';

function emptyStringToNull(string: string): string {
  if (string === '') return null;
  return string;
}

export class AddArticleDto {
  @MaxLength(50)
  @MinLength(1)
  @IsString()
  @IsOptional()
  @Transform(({ value }) => emptyStringToNull(value))
  @ApiPropertyOptional()
  titleAr?: string;

  @MaxLength(50)
  @MinLength(1)
  @IsString()
  @IsOptional()
  @Transform(({ value }) => emptyStringToNull(value))
  @ApiPropertyOptional()
  titleEn?: string;

  @MaxLength(7000)
  @MinLength(100)
  @IsString()
  @IsOptional()
  @Transform(({ value }) => emptyStringToNull(value))
  @ApiPropertyOptional({
    example:
      'منذ خمسينيات القرن الماضي على الأقل، ربط خبراء الصحة التدخين بسرطان الرئة. تستمر الأبحاث في تحديد المزيد من الطرق التي يضر بها التبغ بصحتك، بدءًا من السرطان وحتى الأمراض المزمنة (طويلة الأمد).',
  })
  bodyAr?: string;

  @MaxLength(7000)
  @MinLength(100)
  @IsString()
  @IsOptional()
  @Transform(({ value }) => emptyStringToNull(value))
  @ApiPropertyOptional({
    example:
      'Since at least the 1950s, health experts have linked smoking to lung cancer. Research continues to pinpoint more ways tobacco harms your health, from cancers to chronic (long-term) diseases.',
  })
  bodyEn?: string;

  @IsOptional()
  @ApiPropertyOptional({ type: 'string', format: 'binary' })
  file?: Express.Multer.File;

  @IsOptional()
  @Transform(({ value }) => emptyStringToNull(value))
  @IsEnum(GenderTypeEnum)
  @ApiPropertyOptional({ enum: GenderTypeEnum })
  genderFilter?: GenderTypeEnum;

  @JsonParseAndValidate(Joi.object({ id: Joi.number().required() }))
  @IsOptional()
  @ApiPropertyOptional({ type: 'string' })
  groupIds?: IdDto[] | [];

  @Transform(({ value }) => Number(value))
  @IsInt()
  @Min(1)
  @ApiProperty()
  labelId: number;
}
