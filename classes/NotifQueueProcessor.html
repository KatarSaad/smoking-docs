import { NotificationService } from '../notification.service';

        const patients = job.data.userId
          ? [{ id: job.data.userId }]
          : await this.prismaService.user.findMany({
              where: {
                patient: {
                  groups: {
                    some: {
                      id: { in: notification.targetGroupFilters.map((item) => item.id) },
                    },
                  },
                },
              },
              select: {
                id: true,
              },
            });
        this.notificationService.sendNotifsAsync(patients, notification);
        if (!job.data.notifKind || job.data.notifKind != 'automatic') await this.notificationService.updateNotification('ENGLISH', {}, notification.id, null, 'SENT');
        break;
      case 'schedulePreparedness':
        if (new Date(job.data.endDate) <= new Date()) {

          });

            'scheduleNotification',

              id: Number(notif.id),
              identifier: `scheduleNotification:${notif.id}:${job.data.userId}`,
              userId: job.data.userId,
