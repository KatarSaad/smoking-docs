import { Module, forwardRef } from '@nestjs/common';
import { CommonModule } from 'src/common/common.module';
import { NotificationListener } from './notification.listener';
import { NotificationService } from './notification.service';
import { NotificationController } from './notification.controller';
import { NotificationHubService } from './notification-hub.service';
import { BullModule } from '@nestjs/bullmq';
import { QueueEnum } from 'src/common/enums/queues.enum';
import { NotifQueueProcessor } from './job-processors/notification-job-processor';
import { SettingModule } from 'src/modules/settings/settings.module';
import { UsersModule } from 'src/modules/users/users.module';
import { TodayProgressesModule } from 'src/modules/today-progresses/today-progresses.module';

@Module({
  imports: [
    CommonModule,
    SettingModule,
    forwardRef(() => UsersModule),
    BullModule.registerQueue({
      name: QueueEnum.NOTIFICATION_QUEUE,
    }),
    forwardRef(() => TodayProgressesModule),
  ],
  controllers: [NotificationController],
  providers: [NotificationListener, NotificationService, NotificationHubService, NotifQueueProcessor],
  exports: [NotificationService, NotificationHubService],
})
export class NotificationModule {}
