import { Module, forwardRef } from '@nestjs/common';
import { AdminsController } from './admins.controller';
import { AdminsService } from './admins.service';
import { AuthModule } from '../auth/auth.module';
import { CommonModule } from 'src/common/common.module';
import { OnboardingModule } from 'src/modules/onboarding/onboarding.module';
import { PermissionsModule } from 'src/modules/permissions/permissions.module';
import { UsersModule } from 'src/modules/users/users.module';
import { FacilityRoleModule } from 'src/modules/facility-role/facility-role.module';
import { FacilityGroupModule } from 'src/modules/facility-group/facility-group.module';
import { SettingModule } from '../settings/settings.module';

@Module({
  imports: [AuthModule, CommonModule, OnboardingModule, PermissionsModule, forwardRef(() => UsersModule), FacilityRoleModule, FacilityGroupModule, SettingModule],
  controllers: [AdminsController],
  providers: [AdminsService],
  exports: [AdminsService],
})
export class AdminsModule {}
