import { Module, forwardRef } from '@nestjs/common';
import { MessagesController } from './messages.controller';
import { MessagesService } from './messages.service';
import { CommonModule } from 'src/common/common.module';
import { GroupChatsModule } from 'src/modules/group-chats/group-chats.module';
import { PatientModule } from 'src/modules/patients/patient.module';
import { MessagesListener } from 'src/modules/messages/messages.listener';
import { NotificationModule } from 'src/modules/notification/notification.module';
import { UsersModule } from 'src/modules/users/users.module';

@Module({
  controllers: [MessagesController],
  providers: [MessagesService, MessagesListener],
  exports: [MessagesService, MessagesListener],
  imports: [CommonModule, forwardRef(() => GroupChatsModule), PatientModule, NotificationModule, UsersModule],
})
export class MessagesModule {}
