import { Module, forwardRef } from '@nestjs/common';
import { ArticlesController } from './articles.controller';
import { ArticlesService } from './articles.service';
import { ArticlesLabelsModule } from 'src/modules/articles-labels/articles-labels.module';
import { CommonModule } from 'src/common/common.module';
import { GroupsModule } from 'src/modules/groups/groups.module';
import { UsersModule } from 'src/modules/users/users.module';
import { BullModule } from '@nestjs/bullmq';
import { QueueEnum } from 'src/common/enums/queues.enum';
import { ApproveArticleProcessor } from 'src/modules/articles/job-processors/approve-article.job';
import { NotificationModule } from 'src/modules/notification/notification.module';
import { TodayProgressesModule } from 'src/modules/today-progresses/today-progresses.module';

@Module({
  imports: [
    ArticlesLabelsModule,
    CommonModule,
    GroupsModule,
    forwardRef(() => UsersModule),
    forwardRef(() => NotificationModule),
    forwardRef(() => TodayProgressesModule),
    BullModule.registerQueue({
      name: QueueEnum.APPROVE_ARTICLE_QUEUE,
    }),
  ],
  controllers: [ArticlesController],
  providers: [ArticlesService, ApproveArticleProcessor],
  exports: [ArticlesService],
})
export class ArticlesModule {}
