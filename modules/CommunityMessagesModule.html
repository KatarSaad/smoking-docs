import { Module, forwardRef } from '@nestjs/common';
import { CommonModule } from 'src/common/common.module';
import { PatientModule } from 'src/modules/patients/patient.module';
import { CommunityMessagesService } from './community-messages.service';
import { CommunityMessagesController } from './community-messages.controller';
import { CommunityMessagesListener } from './community-messages.listener';
import { BlockedUsersModule } from 'src/modules/blocked-users/blocked-users.module';
import { UsersModule } from 'src/modules/users/users.module';
import { NotificationModule } from 'src/modules/notification/notification.module';

@Module({
  controllers: [CommunityMessagesController],
  providers: [CommunityMessagesService, CommunityMessagesListener],
  exports: [CommunityMessagesService, CommunityMessagesListener],
  imports: [CommonModule, forwardRef(() => PatientModule), forwardRef(() => BlockedUsersModule), forwardRef(() => UsersModule), forwardRef(() => NotificationModule)],
})
export class CommunityMessagesModule {}
