@Module({
  providers: [FileService],
  exports: [FileService],
  imports: [
    CacheModule.registerAsync<RedisClientOptions>({
      imports: [CommonModule],
      inject: [EnvironmentService],
      isGlobal: true,

      useFactory: async (environmentService: EnvironmentService) => ({
        ttl: environmentService.redisConfig.TTL,
        store: redisStore,
        password: environmentService.redisConfig.password,
        // environmentService.redisConfig.tls  && password: environmentService.redisConfig.password ,

        //a ping to the redis server every 9 minutes will be sent because the azure will kill the unused connection every 10 minutes
        //so sendiung the ping will make sure the connection stays warm
        pingInterval: 9 * 60 * 1000,
        socket: {
          host: environmentService.redisConfig.host,
          port: environmentService.redisConfig.port,

          //reconnectStrategy gradually increase the time between each retry attempt, giving the system some time to recover before trying again.
          //Exponential backoff helps to avoid overwhelming the system with frequent retries
          reconnectStrategy: function (times: number) {
            return Math.max(Math.min(Math.exp(times), 20000), 1000); // 1s, 2s, 4s, 8s, 16s, 20s
          },
        },
      }),
    }),

    BullModule.forRootAsync({
      imports: [CommonModule],
      inject: [EnvironmentService],
      useFactory: async (environmentService: EnvironmentService) => {
        return {
          settings: {
            //gradually increase the time between each retry attempt, giving the system some time to recover before trying again.
            //Exponential backoff helps to avoid overwhelming the system with frequent retries
            repeatStrategy: function (times: number) {
              return Math.max(Math.min(Math.exp(times), 20000), 1000); // 1s, 2s, 4s, 8s, 16s, 20s
            },
          },
          connection: {
            enableOfflineQueue: false, //fast fail on error in code level if server lose connection with redis
            maxRetriesPerRequest: null,
            host: environmentService.redisConfig.host,
            port: environmentService.redisConfig.port,
            password: environmentService.redisConfig.password,
          },
        };
      },
    }),
    AuthModule,
    CommonModule,
    ConfigModule.forRoot({
      isGlobal: true,
      validationSchema: Joi.object({
        SERVER_ENV: Joi.string()
          .valid(...Object.values(ServerEnvironmentEnum))
          .required(),
        PORT: Joi.number().required(),
        DATABASE_USER: Joi.string().required(), //DB
        DATABASE_PASSWORD: Joi.string().required(),
        DATABASE_NAME: Joi.string().required(),
        DATABASE_PORT: Joi.string().required(),
        DATABASE_URL: Joi.string().required(), //JWT
