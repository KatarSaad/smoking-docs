import { Module, forwardRef } from '@nestjs/common';
import { DiariesController } from './diaries.controller';
import { DiariesService } from './diaries.service';
import { CommonModule } from 'src/common/common.module';
import { ChewingGumsModule } from 'src/modules/chewing-gums/chewing-gums.module';
import { PatientModule } from 'src/modules/patients/patient.module';
import { CravingsModule } from 'src/modules/cravings/cravings.module';
import { PatchesModule } from 'src/modules/patches/patches.module';
import { MoodsModule } from 'src/modules/moods/moods.module';
import { SymptomsModule } from 'src/modules/symptoms/symptoms.module';
import { ArticlesModule } from 'src/modules/articles/articles.module';
import { RelapsesModule } from 'src/modules/relapses/relapses.module';
import { AchievementsModule } from '../achievements/achievements.module';
import { ChallengesModule } from '../challenges/challenges.module';

@Module({
  controllers: [DiariesController],
  providers: [DiariesService],
  exports: [DiariesService],
  imports: [
    CommonModule,
    ChewingGumsModule,
    forwardRef(() => PatientModule),
    CravingsModule,
    PatchesModule,
    MoodsModule,
    SymptomsModule,
    ArticlesModule,
    RelapsesModule,
    AchievementsModule,
    forwardRef(() => ChallengesModule),
  ],
})
export class DiariesModule {}
