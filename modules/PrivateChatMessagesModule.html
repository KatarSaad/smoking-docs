import { Module, forwardRef } from '@nestjs/common';
import { PrivateChatMessagesController } from './private-chat-messages.controller';
import { PrivateChatMessagesService } from './private-chat-messages.service';

import { PrivateChatMessagesListener } from 'src/modules/private-chat-messages/private-chat-messages.listener';
import { PrivateChatsModule } from 'src/modules/private-chats/private-chats.module';
import { NotificationModule } from 'src/modules/notification/notification.module';

@Module({
  controllers: [PrivateChatMessagesController],
  providers: [PrivateChatMessagesService, PrivateChatMessagesListener],
  imports: [CommonModule, forwardRef(() => PrivateChatsModule), forwardRef(() => NotificationModule)],
  exports: [PrivateChatMessagesService, PrivateChatMessagesListener],
})
export class PrivateChatMessagesModule {}
