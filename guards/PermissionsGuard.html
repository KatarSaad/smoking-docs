import { UserPlayloadResponse } from './../../modules/patients/response/user-playload.response';
import { CanActivate, ExecutionContext, Injectable } from '@nestjs/common';
import { Reflector } from '@nestjs/core';
import { PrismaService } from '../services/prisma.service';
import { PermissionEnum, RoleRestrictionEnum } from '@prisma/client';
import { EnvironmentService } from 'src/common/services/environment.service';
import { RedisService } from 'src/common/services/redis.service';

@Injectable()
export class PermissionsGuard implements CanActivate {
  constructor(
    private readonly reflector: Reflector,
    private readonly prismaService: PrismaService,
    private readonly environmentService: EnvironmentService,
    private readonly redisService: RedisService,
  ) {}

  async canActivate(context: ExecutionContext): Promise<boolean> {
    let facilityRole = null;
    const requiredPermissions = this.reflector.get<PermissionEnum[]>('permissions', context.getHandler());

    if (!requiredPermissions) return true; // No permissions required, allow access
    const request = context.switchToHttp().getRequest();
    const user: UserPlayloadResponse = request.user;

    //if patient,allow acess =>since patient has no permissions and also assign patient facility to header
    if (user.isPatient) return await this.patientFacilities(user, context);

    //if admin has no role should not access anything
    if (!user.roleId) return false;

    const facilityRoleRedisKey = `check-permission/facilityRole/${user.roleId}`;
    facilityRole = await this.redisService.getResponseKey(facilityRoleRedisKey);

    if (!facilityRole)
      facilityRole = await this.prismaService.facilityRole.findUnique({
        where: { id: user.roleId },
        include: {
          permissions: { select: { id: true, code: true } },
          facilities: { select: { id: true } },
          facilityGroups: {
            select: {
              id: true,
            },
          },
        },
      });

    this.redisService.setResponseKey(facilityRoleRedisKey, facilityRole, this.environmentService.redisConfig.longerTimeToLive);

    if (facilityRole.restriction === RoleRestrictionEnum.ALL) return true; //if role restriction true it means he has all access

    let adminFacilitiesIds = null;
    //we attach the facilities to the request so we can filter based on them
    if (facilityRole.restriction === RoleRestrictionEnum.HEALTHSPECIALIST) {
      const healthSpecialist = await this.prismaService.healthSpecialist.findUnique({
        where: { id: user.healthSpecialistId },
        select: { facilityId: true },
      });

      context.switchToHttp().getRequest().facilities = [healthSpecialist.facilityId];
    } else if (facilityRole.restriction === RoleRestrictionEnum.SYSTEM_ADMIN) {
      //checking if system admin role has grp of facilities

      const currentAdminFacility = await this.prismaService.admin.findUnique({
        where: { id: user.adminId },
        select: {
          adminFacilityGroups: { select: { id: true } },
          adminFacilities: {
            select: {
              id: true,
            },
          },
        },
      });

      if (currentAdminFacility.adminFacilityGroups[0].id) {
        adminFacilitiesIds = await this.facilityGroupFacilities(currentAdminFacility.adminFacilityGroups[0].id);
      }
      //system admin role has facilities
      else {
        adminFacilitiesIds = currentAdminFacility.adminFacilities.map((adminFacility) => adminFacility.id);
      }
      context.switchToHttp().getRequest().facilities = adminFacilitiesIds;
    }
    //this is for custom roles(facility admins)
    else {
      //checking if admin has grp of facilities
      if (facilityRole.facilityGroups[0]) {
        adminFacilitiesIds = await this.facilityGroupFacilities(facilityRole.facilityGroups[0].id);
      }
      //if admin has facilities
      else {
        adminFacilitiesIds = facilityRole.facilities.map((adminFacility) => adminFacility.id);
      }

      context.switchToHttp().getRequest().facilities = adminFacilitiesIds;
    }

    //format permissions as array of strings

    const userPermissions = facilityRole.permissions.map((_) => _.code);

    const isAllowed = requiredPermissions.some((permission) => userPermissions.includes(permission));
    return isAllowed;
  }

  async facilityGroupFacilities(facilityGroupId: bigint): Promise<bigint[]> {
    const facilityGroupFacilitiesRedisKey = `check-permission/facilityGroupFacilitiesRedisKey/${facilityGroupId}`;
    let facilities = null;

    facilities = await this.redisService.getResponseKey(facilityGroupFacilitiesRedisKey);

    if (!facilities)
      facilities = await this.prismaService.healthFacility.findMany({
        where: {
          facilityGroupId: facilityGroupId,
        },
        select: { id: true },
      });

    this.redisService.setResponseKey(facilityGroupFacilitiesRedisKey, facilities, this.environmentService.redisConfig.longerTimeToLive);

    const adminFacilitiesIds = facilities.map((facility) => facility.id);
    return adminFacilitiesIds;
  }

  async patientFacilities(user: UserPlayloadResponse, context: ExecutionContext) {
    const patientFacilityRedisKey = `check-permission/patientFacility/${user.id}`;
    const cacheResults = await this.redisService.getResponseKey(patientFacilityRedisKey);

    //if result exists in redis
    if (cacheResults) {
      context.switchToHttp().getRequest().facilities = cacheResults.facilityId ? [cacheResults.facilityId] : null;
    } else {
      const patientFacility = await this.prismaService.patient.findUnique({
        where: {
          id: user.patientId,
        },
        select: {
          facilityId: true,
        },
      });

      this.redisService.setResponseKey(patientFacilityRedisKey, patientFacility, this.environmentService.redisConfig.longerTimeToLive);

      context.switchToHttp().getRequest().facilities = patientFacility.facilityId ? [patientFacility.facilityId] : null;
    }

    return true;
  }
}
